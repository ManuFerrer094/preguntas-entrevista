Los **componentes renderless** (o headless components) son componentes que encapsulan **lógica y estado** pero **no renderizan ningún HTML propio**. Toda la presentación visual se delega al componente consumidor mediante scoped slots, logrando máxima flexibilidad y reutilización de la lógica.

## Concepto básico

```vue
<!-- RenderlessContador.vue — Solo lógica, sin HTML propio -->
<script setup>
import { ref } from 'vue';

const props = defineProps({
  valorInicial: { type: Number, default: 0 },
  paso: { type: Number, default: 1 },
  min: { type: Number, default: -Infinity },
  max: { type: Number, default: Infinity }
});

const contador = ref(props.valorInicial);

function incrementar() {
  if (contador.value + props.paso <= props.max) {
    contador.value += props.paso;
  }
}

function decrementar() {
  if (contador.value - props.paso >= props.min) {
    contador.value -= props.paso;
  }
}

function resetear() {
  contador.value = props.valorInicial;
}
</script>

<!-- Solo expone un scoped slot con toda la lógica -->
<template>
  <slot
    :contador="contador"
    :incrementar="incrementar"
    :decrementar="decrementar"
    :resetear="resetear"
    :estaEnMin="contador <= props.min"
    :estaEnMax="contador >= props.max"
  />
</template>
```

```vue
<!-- Consumidor A: UI de botones -->
<template>
  <RenderlessContador :min="0" :max="10" v-slot="{ contador, incrementar, decrementar, estaEnMin, estaEnMax }">
    <div class="contador-btns">
      <button :disabled="estaEnMin" @click="decrementar">−</button>
      <span class="valor">{{ contador }}</span>
      <button :disabled="estaEnMax" @click="incrementar">+</button>
    </div>
  </RenderlessContador>
</template>

<!-- Consumidor B: UI de slider -->
<template>
  <RenderlessContador :min="0" :max="100" :paso="5"
    v-slot="{ contador, incrementar, decrementar }">
    <div class="slider-ui">
      <input type="range" :value="contador" @input="..."  />
      <label>Valor: {{ contador }}%</label>
    </div>
  </RenderlessContador>
</template>
```

## Ejemplo práctico: componente de paginación renderless

```vue
<!-- RenderlessPaginacion.vue -->
<script setup>
import { ref, computed } from 'vue';

const props = defineProps({
  total: { type: Number, required: true },
  porPagina: { type: Number, default: 10 },
  paginaInicial: { type: Number, default: 1 }
});

const emit = defineEmits(['cambio-pagina']);

const paginaActual = ref(props.paginaInicial);

const totalPaginas = computed(() => Math.ceil(props.total / props.porPagina));

const rango = computed(() => {
  const inicio = (paginaActual.value - 1) * props.porPagina + 1;
  const fin = Math.min(paginaActual.value * props.porPagina, props.total);
  return { inicio, fin };
});

const paginas = computed(() => {
  const pages = [];
  for (let i = 1; i <= totalPaginas.value; i++) pages.push(i);
  return pages;
});

function irA(pagina) {
  if (pagina < 1 || pagina > totalPaginas.value) return;
  paginaActual.value = pagina;
  emit('cambio-pagina', pagina);
}
</script>

<template>
  <slot
    :pagina-actual="paginaActual"
    :total-paginas="totalPaginas"
    :paginas="paginas"
    :rango="rango"
    :es-primera="paginaActual === 1"
    :es-ultima="paginaActual === totalPaginas"
    :ir-a="irA"
    :anterior="() => irA(paginaActual - 1)"
    :siguiente="() => irA(paginaActual + 1)"
  />
</template>
```

```vue
<!-- UI de paginación: botones simples -->
<RenderlessPaginacion :total="200" :por-pagina="10"
  @cambio-pagina="cargarDatos"
  v-slot="{ paginaActual, totalPaginas, anterior, siguiente, esPrimera, esUltima, rango }">
  <div class="paginacion">
    <span>Mostrando {{ rango.inicio }}-{{ rango.fin }} de 200</span>
    <button :disabled="esPrimera" @click="anterior">← Anterior</button>
    <span>Página {{ paginaActual }} de {{ totalPaginas }}</span>
    <button :disabled="esUltima" @click="siguiente">Siguiente →</button>
  </div>
</RenderlessPaginacion>
```

## Renderless con Composable (patrón moderno)

```javascript
// useAutocompletado.js — Composable alternativo a componente renderless
export function useAutocompletado(buscarFn) {
  const query = ref('');
  const resultados = ref([]);
  const cargando = ref(false);
  const seleccionado = ref(null);
  const indiceActivo = ref(-1);

  const buscar = useDebounceFn(async (valor) => {
    if (!valor.trim()) { resultados.value = []; return; }
    cargando.value = true;
    resultados.value = await buscarFn(valor);
    cargando.value = false;
  }, 300);

  watch(query, buscar);

  function seleccionar(item) {
    seleccionado.value = item;
    query.value = item.texto;
    resultados.value = [];
  }

  return { query, resultados, cargando, seleccionado, seleccionar, indiceActivo };
}
```

## Renderless Components vs Composables

| Aspecto           | Renderless Component         | Composable                   |
|-------------------|------------------------------|------------------------------|
| Encapsulación     | En un `.vue`                 | En un `.js`/`.ts`            |
| Slot visual       | ✅ Sí                       | ❌ No                        |
| Reusabilidad      | Alta                         | ✅ Máxima                   |
| Anidación         | Posible con slots             | Composición de funciones     |
| Test              | Necesita montar el componente | ✅ Función pura testeable    |

**Consejo de entrevista**: Los componentes renderless son el patrón detrás de librerías como Headless UI, Radix Vue y VueUse. En Vue 3, los **Composables** han reemplazado en gran medida a los componentes renderless para lógica pura (sin slots), siendo más simples y testeables. Sin embargo, cuando necesitas la estructura de slots para que el consumidor controle completamente el HTML (por accesibilidad o diseño), los renderless siguen siendo la mejor opción.
