title: ¿Qué es provide/inject en Vue 3?
difficulty: medium

`provide` e `inject` son la implementación de inyección de dependencias en Vue 3. Permiten pasar datos de un componente ancestro a cualquier descendiente sin necesidad de prop drilling (pasar props por cada nivel intermedio).

## Sintaxis básica

```vue
<!-- ParentComponent.vue — proveedor -->
<script setup>
import { provide, ref } from 'vue';

const theme = ref('dark');
const user = ref({ name: 'Ana', role: 'admin' });

// Proveer valores — disponibles para todos los descendientes
provide('theme', theme);
provide('currentUser', user);

// Proveer un objeto con métodos (API completa)
provide('themeActions', {
  toggle: () => theme.value = theme.value === 'dark' ? 'light' : 'dark',
  set: (value) => theme.value = value
});
</script>
```

```vue
<!-- DeepChildComponent.vue — consumidor (cualquier nivel de profundidad) -->
<script setup>
import { inject } from 'vue';

// inject(key, defaultValue)
const theme = inject('theme', ref('light')); // Valor por defecto si no se provee
const user = inject('currentUser');
const themeActions = inject('themeActions');
</script>

<template>
  <div :class="`theme-${theme}`">
    <p>Usuario: {{ user?.name }}</p>
    <button @click="themeActions?.toggle()">
      Cambiar tema (actual: {{ theme }})
    </button>
  </div>
</template>
```

## Usar Symbol como clave (evitar colisiones)

```typescript
// keys.ts — claves únicas para provide/inject
export const THEME_KEY = Symbol('theme');
export const USER_KEY = Symbol('currentUser');
export const CONFIG_KEY = Symbol('config');

// ParentComponent.vue
import { THEME_KEY, USER_KEY } from './keys';
provide(THEME_KEY, theme);
provide(USER_KEY, user);

// ChildComponent.vue — sin riesgo de colisión con otras librerías
import { THEME_KEY } from './keys';
const theme = inject(THEME_KEY); // TypeScript deduce el tipo automáticamente
```

## Tipado con InjectionKey (TypeScript)

```typescript
// keys.ts — con tipado fuerte
import { InjectionKey, Ref } from 'vue';

interface User {
  id: string;
  name: string;
  role: 'user' | 'admin';
}

interface ThemeActions {
  toggle: () => void;
  set: (theme: 'light' | 'dark') => void;
}

export const THEME_KEY: InjectionKey<Ref<'light' | 'dark'>> = Symbol('theme');
export const USER_KEY: InjectionKey<Ref<User | null>> = Symbol('user');
export const THEME_ACTIONS_KEY: InjectionKey<ThemeActions> = Symbol('themeActions');
```

```vue
<!-- Con InjectionKey: TypeScript infiere los tipos correctamente -->
<script setup lang="ts">
import { inject } from 'vue';
import { THEME_KEY, USER_KEY } from './keys';

const theme = inject(THEME_KEY);      // Ref<'light' | 'dark'> | undefined
const user = inject(USER_KEY);        // Ref<User | null> | undefined

// Con valor por defecto (TypeScript lo infiere como no-undefined)
const safeTheme = inject(THEME_KEY, ref('light')); // Ref<'light' | 'dark'>
</script>
```

## Provide reactivo: readonly para evitar mutaciones accidentales

```vue
<script setup>
import { provide, ref, readonly } from 'vue';

const count = ref(0);

// Proveer readonly: los hijos pueden leer pero no mutar directamente
provide('count', readonly(count));

// Proveer una función para mutar (API controlada)
provide('incrementCount', () => count.value++);
</script>
```

## Composable con provide/inject (Plugin de Vue)

```typescript
// composables/useNotifications.ts
import { provide, inject, ref, InjectionKey } from 'vue';

interface Notification {
  id: string;
  message: string;
  type: 'success' | 'error' | 'info';
}

interface NotificationAPI {
  notifications: Readonly<Ref<Notification[]>>;
  notify: (message: string, type?: Notification['type']) => void;
  dismiss: (id: string) => void;
}

const NOTIFICATIONS_KEY: InjectionKey<NotificationAPI> = Symbol('notifications');

export function provideNotifications() {
  const notifications = ref<Notification[]>([]);

  const api: NotificationAPI = {
    notifications: readonly(notifications),
    notify(message, type = 'info') {
      const id = crypto.randomUUID();
      notifications.value.push({ id, message, type });
      setTimeout(() => api.dismiss(id), 5000); // Auto-dismiss
    },
    dismiss(id) {
      notifications.value = notifications.value.filter(n => n.id !== id);
    }
  };

  provide(NOTIFICATIONS_KEY, api);
  return api;
}

export function useNotifications() {
  const api = inject(NOTIFICATIONS_KEY);
  if (!api) throw new Error('useNotifications() debe usarse dentro de un componente con provideNotifications()');
  return api;
}
```

**Consejo de entrevista**: `provide/inject` es la alternativa a Pinia para estado local de un subárbol de componentes (no global). Es ideal para temas, configuración, servicios y APIs que deben estar disponibles en toda una sección de la app. Usa siempre `Symbol` como clave en librerías/plugins para evitar colisiones. Con `InjectionKey<T>` de TypeScript, el tipado se propaga automáticamente sin necesidad de aserciones de tipo manuales.
