---
title: ¿Qué son los plugins en Vue 3?
difficulty: medium
tags: [vue, plugins, global, install, app]
---

Los **plugins** en Vue 3 son objetos o funciones que permiten añadir **funcionalidades globales** a la aplicación: directivas globales, componentes globales, propiedades de instancia, lógica de inicialización, o integraciones con librerías externas.

## Estructura de un plugin

```javascript
// Un plugin puede ser un objeto con método install...
const MiPlugin = {
  install(app, opciones) {
    // app: la instancia de la aplicación Vue
    // opciones: configuración pasada al instalar el plugin
  }
};

// ...o directamente una función
function MiPlugin(app, opciones) { /* ... */ }
```

## Registro del plugin

```javascript
// main.js
import { createApp } from 'vue';
import App from './App.vue';
import MiPlugin from './plugins/mi-plugin';

const app = createApp(App);

// Instalar el plugin con opciones opcionales
app.use(MiPlugin, { opcion1: 'valor', debug: true });

app.mount('#app');
```

## Plugin con componentes y directivas globales

```javascript
// plugins/ui-components.js
import Boton from '../components/Boton.vue';
import Modal from '../components/Modal.vue';
import Spinner from '../components/Spinner.vue';
import vTooltip from '../directives/tooltip';

export const UIPlugin = {
  install(app) {
    // Registrar componentes globales
    app.component('AppBoton', Boton);
    app.component('AppModal', Modal);
    app.component('AppSpinner', Spinner);

    // Registrar directivas globales
    app.directive('tooltip', vTooltip);

    console.log('UI Plugin instalado');
  }
};

// Ahora en cualquier componente:
// <AppBoton>Click</AppBoton>  ← sin necesidad de importar
// <div v-tooltip="'Información'">Hover</div>
```

## Plugin con provide/inject para configuración global

```javascript
// plugins/api.plugin.js
export const ApiPlugin = {
  install(app, { baseUrl, timeout = 5000, interceptors = [] }) {

    // Crear cliente HTTP configurado
    const apiClient = {
      baseUrl,
      timeout,
      async get(endpoint) {
        const response = await fetch(`${baseUrl}${endpoint}`);
        if (!response.ok) throw new Error(`HTTP ${response.status}`);
        return response.json();
      },
      async post(endpoint, data) {
        const response = await fetch(`${baseUrl}${endpoint}`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(data)
        });
        return response.json();
      }
    };

    // Inyectar en toda la aplicación mediante provide/inject
    app.provide('apiClient', apiClient);

    // También disponible como propiedad global
    app.config.globalProperties.$api = apiClient;
  }
};

// main.js
app.use(ApiPlugin, {
  baseUrl: 'https://api.ejemplo.com',
  timeout: 10000
});
```

```vue
<!-- En cualquier componente con inject: -->
<script setup>
import { inject } from 'vue';

const api = inject('apiClient');

async function cargarUsuarios() {
  const usuarios = await api.get('/usuarios');
  return usuarios;
}
</script>

<!-- Con Options API (globalProperties): -->
<script>
export default {
  async mounted() {
    const usuarios = await this.$api.get('/usuarios');
  }
}
</script>
```

## Plugin de internacionalización (i18n)

```javascript
// plugins/i18n.plugin.js
export function crearI18nPlugin(traducciones) {
  return {
    install(app, { locale = 'es' } = {}) {
      const currentLocale = ref(locale);

      function t(clave, params = {}) {
        const mensajes = traducciones[currentLocale.value] ?? {};
        let texto = clave.split('.').reduce((obj, k) => obj?.[k], mensajes) ?? clave;

        // Interpolación: {{ nombre }} → params.nombre
        return texto.replace(/\{\{\s*(\w+)\s*\}\}/g,
          (_, key) => params[key] ?? `{${key}}`
        );
      }

      function cambiarIdioma(nuevoLocale) {
        currentLocale.value = nuevoLocale;
      }

      app.provide('i18n', { t, locale: currentLocale, cambiarIdioma });
      app.config.globalProperties.$t = t;
    }
  };
}

// main.js
const traducciones = {
  es: {
    bienvenida: { titulo: 'Hola, {{ nombre }}!', subtitulo: 'Bienvenido' },
    botones: { enviar: 'Enviar', cancelar: 'Cancelar' }
  },
  en: {
    bienvenida: { titulo: 'Hello, {{ nombre }}!', subtitulo: 'Welcome' },
    botones: { enviar: 'Submit', cancelar: 'Cancel' }
  }
};

app.use(crearI18nPlugin(traducciones), { locale: 'es' });
```

## Plugin de notificaciones globales

```javascript
// plugins/toast.plugin.js
import { createApp, ref } from 'vue';
import ToastContainer from '../components/ToastContainer.vue';

export const ToastPlugin = {
  install(app) {
    const toasts = ref([]);

    const toastService = {
      mostrar(mensaje, tipo = 'info', duracion = 3000) {
        const id = Date.now();
        toasts.value.push({ id, mensaje, tipo });
        setTimeout(() => {
          toasts.value = toasts.value.filter(t => t.id !== id);
        }, duracion);
      },
      exito: (msg) => toastService.mostrar(msg, 'exito'),
      error: (msg) => toastService.mostrar(msg, 'error'),
      advertencia: (msg) => toastService.mostrar(msg, 'advertencia'),
    };

    // Montar el contenedor de toasts en el body
    const containerApp = createApp(ToastContainer, { toasts });
    const div = document.createElement('div');
    document.body.appendChild(div);
    containerApp.mount(div);

    app.provide('toast', toastService);
    app.config.globalProperties.$toast = toastService;
  }
};
```

**Consejo de entrevista**: Los plugins son el mecanismo de extensión de Vue. Son la base de librerías como Vue Router (`app.use(router)`), Pinia (`app.use(pinia)`) y Vue I18n. La diferencia con los composables es el alcance: los plugins configuran la **aplicación** globalmente, los composables encapsulan **lógica de componente** localmente. En proyectos grandes, los plugins se usan para registrar componentes del Design System, configurar librerías de terceros e inyectar servicios globales.
