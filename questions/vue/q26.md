---
title: ¿Qué es defineExpose en Vue 3?
difficulty: medium
tags: [vue, defineExpose, script-setup, template-refs, composition-api]
---

`defineExpose` es un macro de `<script setup>` que permite **exponer explícitamente** propiedades y métodos del componente hijo para que el componente padre pueda acceder a ellos mediante una `ref` de template. Por defecto, los componentes con `<script setup>` son cerrados (nada es público).

## El problema: componentes con script setup son privados

```vue
<!-- Hijo.vue -->
<script setup>
import { ref } from 'vue';

const contador = ref(0);
function resetear() { contador.value = 0; }

// Sin defineExpose: el padre NO puede acceder a contador ni resetear
</script>

<template>
  <p>{{ contador }}</p>
</template>
```

```vue
<!-- Padre.vue -->
<script setup>
import { ref, onMounted } from 'vue';
import Hijo from './Hijo.vue';

const hijoRef = ref(null);

onMounted(() => {
  console.log(hijoRef.value.contador);  // undefined — no está expuesto
  hijoRef.value.resetear();             // Error — no existe
});
</script>

<template>
  <Hijo ref="hijoRef" />
</template>
```

## Solución con defineExpose

```vue
<!-- Hijo.vue -->
<script setup>
import { ref } from 'vue';

const contador = ref(0);
const mensaje = ref('Hola');

function incrementar() { contador.value++; }
function resetear() { contador.value = 0; }

// Exponer explícitamente lo que el padre puede usar
defineExpose({
  contador,     // El padre puede leer/escribir contador.value
  incrementar,  // El padre puede llamar a incrementar()
  resetear,
  // mensaje NO se expone → el padre no puede acceder
});
</script>
```

```vue
<!-- Padre.vue -->
<script setup>
import { ref, onMounted } from 'vue';
import Hijo from './Hijo.vue';

const hijoRef = ref(null);

onMounted(() => {
  console.log(hijoRef.value.contador);    // ✅ 0
  hijoRef.value.incrementar();            // ✅ Funciona
  hijoRef.value.resetear();               // ✅ Funciona
  // hijoRef.value.mensaje                // undefined — no expuesto
});

function incrementarDesdeAfuera() {
  hijoRef.value?.incrementar();
}
</script>

<template>
  <Hijo ref="hijoRef" />
  <button @click="incrementarDesdeAfuera">Incrementar desde el padre</button>
</template>
```

## Caso de uso: componente de modal

```vue
<!-- Modal.vue -->
<script setup>
import { ref } from 'vue';

const visible = ref(false);
const titulo = ref('');

function abrir(nuevoTitulo = 'Modal') {
  titulo.value = nuevoTitulo;
  visible.value = true;
}

function cerrar() {
  visible.value = false;
}

defineExpose({ abrir, cerrar });
</script>

<template>
  <Teleport to="body">
    <div v-if="visible" class="modal-overlay" @click.self="cerrar">
      <div class="modal">
        <h2>{{ titulo }}</h2>
        <slot />
        <button @click="cerrar">Cerrar</button>
      </div>
    </div>
  </Teleport>
</template>
```

```vue
<!-- App.vue -->
<script setup>
import { ref } from 'vue';
import Modal from './Modal.vue';

const modalRef = ref(null);

function mostrarConfirmacion() {
  modalRef.value.abrir('Confirmar acción');
}
</script>

<template>
  <button @click="mostrarConfirmacion">Abrir Modal</button>
  <Modal ref="modalRef">
    <p>¿Estás seguro de que quieres continuar?</p>
  </Modal>
</template>
```

## Caso de uso: componente de formulario con validación

```vue
<!-- FormularioContacto.vue -->
<script setup>
import { ref, reactive } from 'vue';

const formulario = reactive({ nombre: '', email: '', mensaje: '' });
const errores = ref({});

function validar() {
  errores.value = {};
  if (!formulario.nombre) errores.value.nombre = 'El nombre es requerido';
  if (!formulario.email.includes('@')) errores.value.email = 'Email inválido';
  return Object.keys(errores.value).length === 0;
}

function limpiar() {
  formulario.nombre = '';
  formulario.email = '';
  formulario.mensaje = '';
  errores.value = {};
}

function obtenerDatos() {
  return { ...formulario };
}

defineExpose({ validar, limpiar, obtenerDatos });
</script>
```

```vue
<!-- PaginaContacto.vue -->
<script setup>
import { ref } from 'vue';
import FormularioContacto from './FormularioContacto.vue';

const formularioRef = ref(null);

async function enviarFormulario() {
  const esValido = formularioRef.value.validar();
  if (!esValido) return;

  const datos = formularioRef.value.obtenerDatos();
  await enviarAlServidor(datos);
  formularioRef.value.limpiar();
}
</script>

<template>
  <FormularioContacto ref="formularioRef" />
  <button @click="enviarFormulario">Enviar</button>
</template>
```

## Diferencia entre defineExpose y defineProps/defineEmits

| API              | Propósito                                    | Dirección            |
|------------------|----------------------------------------------|----------------------|
| `defineProps`    | Recibir datos del padre                      | Padre → Hijo         |
| `defineEmits`    | Enviar eventos al padre                      | Hijo → Padre         |
| `defineExpose`   | Exponer API al padre via ref                 | Hijo → Padre (imperativo) |

**Consejo de entrevista**: `defineExpose` es para comunicación **imperativa** (el padre llama métodos del hijo). Para comunicación **declarativa**, usa `props` y `emits`. El uso excesivo de `defineExpose` es un anti-patrón —si te encuentras usándolo frecuentemente, considera si el estado debería elevarse al padre o a un store compartido. Los casos de uso legítimos son: modales, drawers, reproductores multimedia, formularios con validación externa.
