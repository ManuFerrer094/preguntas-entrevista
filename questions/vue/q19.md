Vue 3 ofrece varias aproximaciones para manejar formularios: validación manual con la Composition API, o librerías como VeeValidate (la más popular) combinada con Zod o Yup para el schema de validación.

## Validación manual con la Composition API

```vue
<script setup>
import { reactive, computed } from 'vue';

const form = reactive({
  email: '',
  password: '',
  confirmPassword: ''
});

const errors = reactive({
  email: '',
  password: '',
  confirmPassword: ''
});

const isSubmitting = ref(false);

function validateEmail(email) {
  if (!email) return 'El email es requerido';
  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) return 'Email inválido';
  return '';
}

function validatePassword(password) {
  if (!password) return 'La contraseña es requerida';
  if (password.length < 8) return 'Mínimo 8 caracteres';
  if (!/(?=.*[A-Z])/.test(password)) return 'Debe contener una mayúscula';
  return '';
}

function validate() {
  errors.email = validateEmail(form.email);
  errors.password = validatePassword(form.password);
  errors.confirmPassword = form.password !== form.confirmPassword
    ? 'Las contraseñas no coinciden' : '';

  return !errors.email && !errors.password && !errors.confirmPassword;
}

async function handleSubmit() {
  if (!validate()) return;

  isSubmitting.value = true;
  try {
    await registerUser(form);
    router.push('/dashboard');
  } catch (err) {
    errors.email = err.message;
  } finally {
    isSubmitting.value = false;
  }
}
</script>

<template>
  <form @submit.prevent="handleSubmit" novalidate>
    <div>
      <label for="email">Email</label>
      <input
        id="email"
        v-model="form.email"
        type="email"
        :class="{ 'error': errors.email }"
        @blur="errors.email = validateEmail(form.email)"
      />
      <span v-if="errors.email" class="error-text">{{ errors.email }}</span>
    </div>

    <button type="submit" :disabled="isSubmitting">
      {{ isSubmitting ? 'Enviando...' : 'Registrarse' }}
    </button>
  </form>
</template>
```

## Con VeeValidate y Zod (enfoque recomendado)

```vue
<script setup>
import { useForm } from 'vee-validate';
import { toTypedSchema } from '@vee-validate/zod';
import { z } from 'zod';

// Schema de validación con Zod
const schema = toTypedSchema(z.object({
  email: z
    .string()
    .min(1, 'El email es requerido')
    .email('Email inválido'),
  password: z
    .string()
    .min(8, 'Mínimo 8 caracteres')
    .regex(/(?=.*[A-Z])/, 'Debe contener una mayúscula')
    .regex(/(?=.*\d)/, 'Debe contener un número'),
  confirmPassword: z.string()
}).refine(
  data => data.password === data.confirmPassword,
  { message: 'Las contraseñas no coinciden', path: ['confirmPassword'] }
));

const { handleSubmit, isSubmitting, errors, defineField } = useForm({
  validationSchema: schema
});

// defineField conecta el campo con el sistema de validación de VeeValidate
const [email, emailAttrs] = defineField('email');
const [password, passwordAttrs] = defineField('password');
const [confirmPassword, confirmPasswordAttrs] = defineField('confirmPassword');

const onSubmit = handleSubmit(async (values) => {
  await registerUser(values);
  router.push('/dashboard');
});
</script>

<template>
  <form @submit.prevent="onSubmit" novalidate>
    <div class="field">
      <label>Email</label>
      <input v-model="email" v-bind="emailAttrs" type="email" />
      <span v-if="errors.email" class="error">{{ errors.email }}</span>
    </div>

    <div class="field">
      <label>Contraseña</label>
      <input v-model="password" v-bind="passwordAttrs" type="password" />
      <span v-if="errors.password" class="error">{{ errors.password }}</span>
    </div>

    <div class="field">
      <label>Confirmar contraseña</label>
      <input v-model="confirmPassword" v-bind="confirmPasswordAttrs" type="password" />
      <span v-if="errors.confirmPassword" class="error">{{ errors.confirmPassword }}</span>
    </div>

    <button type="submit" :disabled="isSubmitting">
      {{ isSubmitting ? 'Enviando...' : 'Registrarse' }}
    </button>
  </form>
</template>
```

## Composable de formulario reutilizable

```typescript
// composables/useFormField.ts
import { ref, computed } from 'vue';

export function useFormField(validationFn: (value: string) => string) {
  const value = ref('');
  const touched = ref(false);
  const error = computed(() => touched.value ? validationFn(value.value) : '');
  const isValid = computed(() => !error.value);

  function blur() { touched.value = true; }
  function reset() {
    value.value = '';
    touched.value = false;
  }

  return { value, error, isValid, touched, blur, reset };
}

// Uso en componentes
const emailField = useFormField(email => {
  if (!email) return 'Requerido';
  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) return 'Email inválido';
  return '';
});
```

**Consejo de entrevista**: Para formularios simples, la validación manual con `reactive` y computed es suficiente. Para formularios complejos (muchos campos, validación asíncrona, arrays de campos), VeeValidate + Zod es el estándar de la industria en Vue 3. Zod tiene la ventaja de compartir el schema entre el frontend y el backend (con Node.js). Mencionar la validación en `@blur` (al salir del campo) en lugar de en cada keystroke mejora la UX.
