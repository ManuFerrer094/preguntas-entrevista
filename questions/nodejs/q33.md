El **Circuit Breaker** (Disyuntor) es un patrón de diseño de resiliencia que previene fallos en cascada cuando un servicio externo falla. Funciona como un interruptor eléctrico: cuando detecta demasiados fallos, "abre" el circuito y rechaza peticiones temporalmente, dando tiempo al servicio externo para recuperarse.

## Cómo funciona

```
┌──────────────────────────────────────────────────┐
│                    Estados                        │
│                                                  │
│  CLOSED → Normal: las peticiones pasan           │
│     ↓ (umbral de fallos alcanzado)               │
│  OPEN → Abierto: rechaza todas las peticiones    │
│     ↓ (timeout de recuperación expira)           │
│  HALF-OPEN → Prueba: deja pasar UNA petición     │
│     ↓ (éxito) → CLOSED                           │
│     ↓ (fallo) → OPEN                             │
└──────────────────────────────────────────────────┘
```

## Implementación manual

```javascript
class CircuitBreaker {
  constructor(servicio, opciones = {}) {
    this.servicio = servicio;
    this.umbralFallos = opciones.umbralFallos ?? 5;
    this.timeoutRecuperacion = opciones.timeoutRecuperacion ?? 30000;  // 30s
    this.timeoutPeticion = opciones.timeoutPeticion ?? 5000;           // 5s

    this.estado = 'CLOSED';
    this.contadorFallos = 0;
    this.ultimoFallo = null;
  }

  async ejecutar(...args) {
    if (this.estado === 'OPEN') {
      if (Date.now() - this.ultimoFallo > this.timeoutRecuperacion) {
        this.estado = 'HALF-OPEN';
        console.log(`[CircuitBreaker] Probando recuperación...`);
      } else {
        throw new Error(`CircuitBreaker ABIERTO: servicio no disponible temporalmente`);
      }
    }

    try {
      // Ejecutar con timeout
      const resultado = await Promise.race([
        this.servicio(...args),
        new Promise((_, reject) =>
          setTimeout(() => reject(new Error('Timeout')), this.timeoutPeticion)
        )
      ]);

      // Éxito: reset del circuito
      if (this.estado === 'HALF-OPEN') {
        console.log(`[CircuitBreaker] Recuperado. Estado: CLOSED`);
      }
      this.contadorFallos = 0;
      this.estado = 'CLOSED';
      return resultado;

    } catch (error) {
      this.contadorFallos++;
      this.ultimoFallo = Date.now();

      if (this.contadorFallos >= this.umbralFallos || this.estado === 'HALF-OPEN') {
        this.estado = 'OPEN';
        console.error(
          `[CircuitBreaker] ABIERTO tras ${this.contadorFallos} fallos. ` +
          `Próximo reintento en ${this.timeoutRecuperacion / 1000}s`
        );
      }

      throw error;
    }
  }

  obtenerEstado() {
    return {
      estado: this.estado,
      contadorFallos: this.contadorFallos,
      ultimoFallo: this.ultimoFallo
        ? new Date(this.ultimoFallo).toISOString()
        : null
    };
  }
}
```

import CircuitBreaker from 'opossum';
import fetch from 'node-fetch';

// Función a proteger con circuit breaker
async function llamarAPIExterna(endpoint, datos) {
  const response = await fetch(`https://api.externa.com${endpoint}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(datos),
    signal: AbortSignal.timeout(5000)  // Timeout de 5s
  });

  if (!response.ok) throw new Error(`API respondió con ${response.status}`);
  return response.json();
}

// Configurar el Circuit Breaker
const breaker = new CircuitBreaker(llamarAPIExterna, {
  timeout: 5000,              // Tiempo máximo de espera
  errorThresholdPercentage: 50,  // Abrir si >50% de peticiones fallan
  resetTimeout: 30000,        // Tiempo antes de intentar HALF-OPEN (30s)
  volumeThreshold: 5,         // Mínimo de peticiones antes de calcular %
  rollingCountTimeout: 10000, // Ventana temporal para calcular estadísticas (10s)
});

// Función de fallback: respuesta alternativa cuando el circuito está abierto
breaker.fallback((endpoint, datos, error) => {
  console.warn(`[CircuitBreaker] Usando fallback para ${endpoint}:`, error.message);
  return { exito: false, datos: null, error: 'Servicio no disponible', usandoFallback: true };
});

// Eventos del Circuit Breaker
breaker.on('open', () =>
  console.error('🔴 Circuit Breaker ABIERTO — servicio externo no responde')
);
breaker.on('halfOpen', () =>
  console.warn('🟡 Circuit Breaker HALF-OPEN — probando recuperación')
);
breaker.on('close', () =>
  console.log('🟢 Circuit Breaker CERRADO — servicio recuperado')
);
breaker.on('fallback', (result) =>
  console.warn('⚠️ Fallback activado:', result)
);

// Uso en la aplicación
async function procesarPago(datosPago) {
  try {
    const resultado = await breaker.fire('/pagos', datosPago);
    if (resultado.usandoFallback) {
      // El servicio de pagos no está disponible, encolar para reintentar
      await colaPagos.add('pago-pendiente', datosPago);
      return { estado: 'pendiente', mensaje: 'Pago en cola para procesar' };
    }
    return resultado;
  } catch (error) {
    throw new Error('Error al procesar el pago');
  }
}

## Múltiples Circuit Breakers para diferentes servicios

```javascript
// services/resilient-clients.js
export class ClienteResilient {
  constructor() {
    this.breakers = new Map();
  }

  obtenerBreaker(nombreServicio, fnServicio, opciones = {}) {
    if (!this.breakers.has(nombreServicio)) {
      const breaker = new CircuitBreaker(fnServicio, {
        timeout: 5000,
        errorThresholdPercentage: 50,
        resetTimeout: 30000,
        ...opciones
      });

      this.breakers.set(nombreServicio, breaker);
    }
    return this.breakers.get(nombreServicio);
  }

  obtenerEstadoTodos() {
    const estado = {};
    for (const [nombre, breaker] of this.breakers) {
      estado[nombre] = {
        abierto: breaker.opened,
        cerrado: breaker.closed,
        stats: breaker.stats
      };
    }
    return estado;
  }
}

// Health check endpoint que muestra el estado de los circuit breakers
app.get('/health/circuit-breakers', (req, res) => {
  res.json(clienteResilient.obtenerEstadoTodos());
});
```

## Patrón Retry con Backoff exponencial

```javascript
// Combinar Circuit Breaker con reintentos automáticos
async function ejecutarConReintentosYCircuitBreaker(
  fn,
  maxReintentos = 3,
  baseDelay = 1000,
  circuitBreaker
) {
  for (let intento = 1; intento <= maxReintentos; intento++) {
    try {
      return circuitBreaker
        ? await circuitBreaker.fire(fn)
        : await fn();
    } catch (error) {
      if (intento === maxReintentos) throw error;

      const delay = baseDelay * Math.pow(2, intento - 1);  // 1s, 2s, 4s
      const jitter = Math.random() * 1000;                  // Añadir variación
      await new Promise(resolve => setTimeout(resolve, delay + jitter));

      console.log(`Reintento ${intento}/${maxReintentos} en ${delay}ms...`);
    }
  }
}
```

**Consejo de entrevista**: El Circuit Breaker es esencial en arquitecturas de microservicios para evitar fallos en cascada. Sin él, si el servicio de pagos está lento, todas las peticiones de tu app esperarán agotando el thread pool, haciendo que toda la app sea lenta. El patrón se combina habitualmente con **Retry** (reintentos con backoff) y **Fallback** (respuesta degradada). Opossum es la librería más popular en Node.js; en entornos Spring (Java), Hystrix/Resilience4j son los equivalentes. Menciona también el patrón **Bulkhead** (aislar fallos) como complemento.
