El testing en Node.js es fundamental para garantizar la calidad del código. Se divide principalmente en: **unit tests** (funciones individuales), **integration tests** (módulos juntos, incluyendo base de datos) y **e2e tests** (flujo completo de la aplicación). La herramienta más popular es **Jest**, aunque **Vitest** gana popularidad por su velocidad.

## Unit Testing con Jest

```javascript
// services/users.service.test.js
const usersService = require('./users.service');
const User = require('../models/user.model');

// Mock del modelo de base de datos
jest.mock('../models/user.model');

describe('UsersService', () => {
  beforeEach(() => {
    jest.clearAllMocks(); // Limpiar mocks entre tests
  });

  describe('getById', () => {
    it('debe retornar el usuario cuando existe', async () => {
      const mockUser = { id: '1', name: 'Ana', email: 'ana@test.com' };
      User.findById.mockResolvedValue(mockUser);

      const result = await usersService.getById('1');

      expect(User.findById).toHaveBeenCalledWith('1');
      expect(result).toEqual(mockUser);
    });

    it('debe lanzar NotFoundError cuando el usuario no existe', async () => {
      User.findById.mockResolvedValue(null);

      await expect(usersService.getById('999'))
        .rejects.toThrow('Usuario 999 no encontrado');
    });
  });

  describe('create', () => {
    it('debe lanzar ConflictError si el email ya existe', async () => {
      User.findOne.mockResolvedValue({ id: '1', email: 'existing@test.com' });

      await expect(usersService.create({ email: 'existing@test.com', name: 'Test' }))
        .rejects.toThrow('El email ya está registrado');
    });
  });
});
```

## Integration Testing con Supertest

```javascript
// tests/users.routes.test.js
const request = require('supertest');
const mongoose = require('mongoose');
const { MongoMemoryServer } = require('mongodb-memory-server');
const app = require('../app');

let mongoServer;

beforeAll(async () => {
  // Base de datos en memoria para tests de integración
  mongoServer = await MongoMemoryServer.create();
  await mongoose.connect(mongoServer.getUri());
});

afterAll(async () => {
  await mongoose.disconnect();
  await mongoServer.stop();
});

afterEach(async () => {
  // Limpiar entre tests
  await mongoose.connection.dropDatabase();
});

describe('Users API', () => {
  let authToken;

  beforeEach(async () => {
    // Crear usuario y obtener token para tests autenticados
    const res = await request(app)
      .post('/api/auth/register')
      .send({ name: 'Test User', email: 'test@test.com', password: 'password123' });
    authToken = res.body.accessToken;
  });

  describe('GET /api/users', () => {
    it('debe retornar 401 sin autenticación', async () => {
      const res = await request(app).get('/api/users');
      expect(res.status).toBe(401);
    });

    it('debe retornar lista de usuarios autenticado', async () => {
      const res = await request(app)
        .get('/api/users')
        .set('Authorization', `Bearer ${authToken}`);

      expect(res.status).toBe(200);
      expect(res.body).toHaveProperty('users');
      expect(Array.isArray(res.body.users)).toBe(true);
    });
  });

  describe('POST /api/users', () => {
    it('debe crear un nuevo usuario', async () => {
      const newUser = { name: 'Nuevo Usuario', email: 'nuevo@test.com' };

      const res = await request(app)
        .post('/api/users')
        .set('Authorization', `Bearer ${authToken}`)
        .send(newUser);

      expect(res.status).toBe(201);
      expect(res.body).toMatchObject({ name: newUser.name, email: newUser.email });
      expect(res.body).not.toHaveProperty('passwordHash');
    });
  });
});
```

## Configuración de Jest

```javascript
// jest.config.js
module.exports = {
  testEnvironment: 'node',
  setupFilesAfterFramework: ['./tests/setup.js'],
  collectCoverage: true,
  coverageThreshold: {
    global: { branches: 80, functions: 80, lines: 80 }
  },
  testTimeout: 10000
};
```

## Testing de funciones puras

```javascript
// utils/validators.test.js
const { validateEmail, calculateDiscount } = require('./validators');

describe('validateEmail', () => {
  test.each([
    ['valid@email.com', true],
    ['another.valid+email@domain.co', true],
    ['invalid-email', false],
    ['@no-user.com', false],
    ['', false]
  ])('validateEmail("%s") debe retornar %s', (email, expected) => {
    expect(validateEmail(email)).toBe(expected);
  });
});
```

**Consejo de entrevista**: Para una buena estrategia de testing en Node.js: usar `jest.mock()` para aislar dependencias en unit tests, `mongodb-memory-server` para integration tests sin afectar la base de datos real, y `supertest` para testear APIs HTTP. Alcanzar un 80% de cobertura de código es un buen objetivo, pero la calidad de los tests importa más que el porcentaje.
