Node.js puede conectarse con prácticamente cualquier base de datos. Las opciones más comunes son MongoDB (con Mongoose), PostgreSQL (con Prisma o pg), MySQL y Redis. La elección depende de los requisitos del proyecto: estructura de datos, consistencia, escalabilidad y preferencia del equipo.

## MongoDB con Mongoose

```javascript
// config/database.js
const mongoose = require('mongoose');

async function connectDB() {
  try {
    const conn = await mongoose.connect(process.env.MONGODB_URI, {
      maxPoolSize: 10 // Conexiones simultáneas máximas
    });
    console.log(`MongoDB conectado: ${conn.connection.host}`);
  } catch (error) {
    console.error('Error conectando a MongoDB:', error);
    process.exit(1);
  }
}

// Gestionar eventos de conexión
mongoose.connection.on('disconnected', () => {
  console.warn('MongoDB desconectado');
});

module.exports = connectDB;
```

## PostgreSQL con Prisma (ORM moderno)

```javascript
// Definir schema en schema.prisma
// model User {
//   id        Int      @id @default(autoincrement())
//   email     String   @unique
//   name      String
//   posts     Post[]
//   createdAt DateTime @default(now())
// }

// Uso con Prisma Client
const { PrismaClient } = require('@prisma/client');

const prisma = new PrismaClient({
  log: ['query', 'error'] // Log de queries en desarrollo
});

// CRUD con Prisma
async function userOperations() {
  // Crear
  const user = await prisma.user.create({
    data: { email: 'ana@example.com', name: 'Ana García' }
  });

  // Leer con relaciones
  const userWithPosts = await prisma.user.findUnique({
    where: { id: 1 },
    include: { posts: { where: { published: true } } }
  });

  // Paginación
  const users = await prisma.user.findMany({
    where: { name: { contains: 'García' } },
    orderBy: { createdAt: 'desc' },
    skip: 0,
    take: 10
  });

  // Transacciones
  const [post, updatedUser] = await prisma.$transaction([
    prisma.post.create({ data: { title: 'Nuevo post', authorId: 1 } }),
    prisma.user.update({ where: { id: 1 }, data: { postCount: { increment: 1 } } })
  ]);
}
```

## PostgreSQL con pg (query builder)

```javascript
const { Pool } = require('pg');

const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
  max: 20,           // Máximo de conexiones en el pool
  idleTimeoutMillis: 30000,
  connectionTimeoutMillis: 2000,
});

// Query parametrizada (previene SQL injection)
async function getUserById(id) {
  const { rows } = await pool.query(
    'SELECT id, name, email FROM users WHERE id = $1',
    [id] // Parámetro escapado automáticamente
  );
  return rows[0];
}

// Transacción
async function transferFunds(fromId, toId, amount) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    await client.query('UPDATE accounts SET balance = balance - $1 WHERE id = $2', [amount, fromId]);
    await client.query('UPDATE accounts SET balance = balance + $1 WHERE id = $2', [amount, toId]);
    await client.query('COMMIT');
  } catch (error) {
    await client.query('ROLLBACK');
    throw error;
  } finally {
    client.release();
  }
}
```

## Redis para caché

```javascript
const { createClient } = require('redis');

const redis = createClient({ url: process.env.REDIS_URL });
redis.connect();

// Caché con TTL
async function getCachedUser(id) {
  const cacheKey = `user:${id}`;

  const cached = await redis.get(cacheKey);
  if (cached) return JSON.parse(cached);

  const user = await User.findById(id);
  if (user) {
    await redis.setEx(cacheKey, 3600, JSON.stringify(user)); // Expirar en 1 hora
  }
  return user;
}

// Invalidar caché al actualizar
async function updateUser(id, data) {
  const user = await User.findByIdAndUpdate(id, data, { new: true });
  await redis.del(`user:${id}`); // Invalidar caché
  return user;
}
```

**Consejo de entrevista**: Mencionar el **connection pooling** es importante: Node.js debe reutilizar conexiones de base de datos (usando un pool) en lugar de crear una nueva conexión por cada request, lo que sería muy costoso. Tanto `mongoose`, `pg.Pool` como `prisma` gestionan esto automáticamente. Para proyectos nuevos con SQL, Prisma es una excelente opción por su type-safety y sus migraciones.
