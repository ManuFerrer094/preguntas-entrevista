---
title: ¿Qué es el patrón Repository en Node.js?
difficulty: medium
tags: [nodejs, repository-pattern, architecture, clean-code, database]
---

El patrón Repository abstrae el acceso a datos detrás de una interfaz, desacoplando la lógica de negocio de la tecnología de persistencia (MongoDB, PostgreSQL, Redis, etc.). Facilita el testing con mocks y permite cambiar de base de datos sin modificar la lógica de negocio.

## Problema que resuelve

```javascript
// ❌ Sin patrón Repository: la lógica de negocio conoce MongoDB
class UserService {
  async createUser(data) {
    // Lógica de negocio mezclada con acceso a datos
    const existing = await User.findOne({ email: data.email }); // Mongoose directo
    if (existing) throw new Error('Email ya registrado');

    const user = new User(data);
    await user.save();
    return user;
  }
}
// Problemas: difícil de testear, acoplado a Mongoose, si cambias a PostgreSQL
// debes reescribir toda la lógica de negocio
```

## Implementación del patrón Repository

```javascript
// repositories/user.repository.js
class UserRepository {
  async findById(id) {
    return User.findById(id).lean();
  }

  async findByEmail(email) {
    return User.findOne({ email }).lean();
  }

  async findAll({ page = 1, limit = 10, filter = {} } = {}) {
    const skip = (page - 1) * limit;
    const [data, total] = await Promise.all([
      User.find(filter).skip(skip).limit(limit).lean(),
      User.countDocuments(filter)
    ]);
    return { data, total, page, limit };
  }

  async create(data) {
    const user = new User(data);
    return user.save();
  }

  async update(id, data) {
    return User.findByIdAndUpdate(id, data, { new: true }).lean();
  }

  async delete(id) {
    return User.findByIdAndDelete(id).lean();
  }

  async exists({ email }) {
    return User.exists({ email });
  }
}

module.exports = new UserRepository();
```

```javascript
// services/user.service.js — solo lógica de negocio
const userRepository = require('../repositories/user.repository');
const bcrypt = require('bcrypt');

class UserService {
  #repository;

  constructor(repository) {
    this.#repository = repository; // Inyección de dependencias
  }

  async register(data) {
    const exists = await this.#repository.exists({ email: data.email });
    if (exists) throw new ConflictError('El email ya está registrado');

    const passwordHash = await bcrypt.hash(data.password, 12);
    return this.#repository.create({ ...data, password: passwordHash });
  }

  async login(email, password) {
    const user = await this.#repository.findByEmail(email);
    if (!user) throw new UnauthorizedError('Credenciales inválidas');

    const isValid = await bcrypt.compare(password, user.password);
    if (!isValid) throw new UnauthorizedError('Credenciales inválidas');

    return user;
  }

  async getProfile(id) {
    const user = await this.#repository.findById(id);
    if (!user) throw new NotFoundError(`Usuario ${id} no encontrado`);
    return user;
  }
}

module.exports = new UserService(userRepository);
```

## Testing con un repositorio mock

```javascript
// tests/user.service.test.js
const UserService = require('../services/user.service');

// Mock del repositorio — no necesita base de datos real
const mockRepository = {
  exists: jest.fn(),
  create: jest.fn(),
  findByEmail: jest.fn(),
  findById: jest.fn(),
};

const userService = new UserService(mockRepository);

describe('UserService', () => {
  beforeEach(() => jest.clearAllMocks());

  describe('register', () => {
    it('lanza error si el email ya existe', async () => {
      mockRepository.exists.mockResolvedValue(true);

      await expect(userService.register({
        email: 'existing@test.com',
        password: 'pass123'
      })).rejects.toThrow('El email ya está registrado');
    });

    it('crea usuario correctamente', async () => {
      const userData = { email: 'new@test.com', password: 'pass123' };
      const createdUser = { id: '1', ...userData, password: 'hashed' };

      mockRepository.exists.mockResolvedValue(false);
      mockRepository.create.mockResolvedValue(createdUser);

      const user = await userService.register(userData);
      expect(mockRepository.create).toHaveBeenCalledWith(
        expect.objectContaining({ email: userData.email })
      );
      expect(user).toEqual(createdUser);
    });
  });
});
```

## Interfaz genérica con TypeScript

```typescript
// repositories/base.repository.ts
interface IRepository<T, CreateDTO, UpdateDTO> {
  findById(id: string): Promise<T | null>;
  findAll(options?: QueryOptions): Promise<PaginatedResult<T>>;
  create(data: CreateDTO): Promise<T>;
  update(id: string, data: UpdateDTO): Promise<T | null>;
  delete(id: string): Promise<boolean>;
}
```

**Consejo de entrevista**: El patrón Repository es especialmente valioso con el principio de inversión de dependencias (DIP de SOLID): los servicios dependen de abstracciones (interfaces), no de implementaciones concretas. Esto permite reemplazar MongoDB por PostgreSQL (o usar una implementación in-memory en tests) sin cambiar una sola línea de lógica de negocio. En entrevistas, combinar Repository con inyección de dependencias y mencionar la facilidad de testing suele impresionar.
