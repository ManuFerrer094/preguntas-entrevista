title: ¿Qué es EventEmitter en Node.js y cómo usarlo?

`EventEmitter` es la clase base del sistema de eventos de Node.js. Casi todas las APIs de Node.js (streams, HTTP, ficheros) extienden `EventEmitter`. Implementa el patrón Observer permitiendo publicar y suscribirse a eventos.

## Uso básico

```javascript
const { EventEmitter } = require('events');

const emitter = new EventEmitter();

// Suscribirse a un evento
emitter.on('data', (payload) => {
  console.log('Datos recibidos:', payload);
});

// Escuchar solo una vez
emitter.once('connected', () => {
  console.log('Conexión establecida (solo se ejecuta una vez)');
});

// Emitir eventos
emitter.emit('data', { id: 1, value: 'hola' });
emitter.emit('connected');
emitter.emit('connected'); // El listener 'once' ya no se ejecuta

// Eliminar listener específico
const handler = (msg) => console.log(msg);
emitter.on('message', handler);
emitter.off('message', handler); // Equivalente a removeListener
```

## Crear una clase que extiende EventEmitter

```javascript
const { EventEmitter } = require('events');

class Database extends EventEmitter {
  #isConnected = false;
  #connectionString;

  constructor(connectionString) {
    super();
    this.#connectionString = connectionString;
  }

  async connect() {
    try {
      this.emit('connecting', this.#connectionString);
      // Simular conexión asíncrona
      await new Promise(resolve => setTimeout(resolve, 100));
      this.#isConnected = true;
      this.emit('connected', { url: this.#connectionString });
    } catch (error) {
      this.emit('error', error);
    }
  }

  async query(sql, params = []) {
    if (!this.#isConnected) {
      this.emit('error', new Error('No conectado a la base de datos'));
      return null;
    }

    this.emit('query', { sql, params });
    // Ejecutar query...
    const result = { rows: [], rowCount: 0 };
    this.emit('result', result);
    return result;
  }

  disconnect() {
    this.#isConnected = false;
    this.emit('disconnected');
    this.removeAllListeners(); // Limpiar todos los listeners
  }
}

// Uso
const db = new Database('mongodb://localhost:27017/mydb');

db.on('connecting', (url) => console.log(`Conectando a ${url}...`));
db.on('connected', ({ url }) => console.log(`Conectado a ${url}`));
db.on('query', ({ sql }) => console.log(`Ejecutando: ${sql}`));
db.on('error', (err) => console.error(`Error: ${err.message}`));

await db.connect();
await db.query('SELECT * FROM users WHERE id = ?', [1]);
```

## Gestión de errores con EventEmitter

```javascript
const emitter = new EventEmitter();

// IMPORTANTE: si no hay listener para 'error', Node.js lanza una excepción no capturada
// Siempre añadir un listener de error
emitter.on('error', (err) => {
  console.error('Error capturado:', err.message);
});

// Sin listener de error → proceso se cae:
// emitter.emit('error', new Error('Error fatal')); // ❌ Sin listener → crash

// Con listener → manejado correctamente:
emitter.emit('error', new Error('Error controlado')); // ✅
```

## Backpressure y límite de listeners

```javascript
const emitter = new EventEmitter();

// Por defecto, más de 10 listeners en el mismo evento genera un warning
// Aumentar el límite si es necesario
emitter.setMaxListeners(20);

// O para una instancia global
require('events').EventEmitter.defaultMaxListeners = 20;

// Ver cuántos listeners hay para un evento
emitter.on('data', () => {});
emitter.on('data', () => {});
console.log(emitter.listenerCount('data')); // 2
console.log(emitter.eventNames());          // ['data']
```

## Eventos asíncronos con promesas

```javascript
const { EventEmitter, once } = require('events');

class FileProcessor extends EventEmitter {
  async processFile(filePath) {
    const chunks = [];

    this.emit('start', filePath);

    // Simular procesamiento
    for (let i = 0; i < 5; i++) {
      await new Promise(resolve => setTimeout(resolve, 100));
      const chunk = `chunk-${i}`;
      chunks.push(chunk);
      this.emit('progress', { chunk, percent: (i + 1) * 20 });
    }

    this.emit('complete', chunks.join(','));
  }
}

const processor = new FileProcessor();

// Uso con once() para esperar un evento una vez
async function main() {
  const completionPromise = once(processor, 'complete');

  processor.on('progress', ({ percent }) => {
    console.log(`Progreso: ${percent}%`);
  });

  processor.processFile('data.csv');

  const [result] = await completionPromise;
  console.log('Resultado final:', result);
}

main();
```

**Consejo de entrevista**: Siempre añade un listener para el evento `error`; sin él, un `emit('error')` provoca una excepción no capturada que puede tirar el proceso. Para evitar memory leaks, elimina los listeners cuando ya no se necesiten con `off()` o `removeAllListeners()`. La función `once()` de `events` es útil para convertir eventos en promesas.
