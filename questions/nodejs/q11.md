---
title: ¿Qué es el patrón MVC en Node.js?
difficulty: medium

El patrón MVC (Model-View-Controller) es una arquitectura de software que separa la aplicación en tres capas: **Model** (datos y lógica de negocio), **View** (presentación) y **Controller** (coordinación entre Model y View). En APIs REST con Node.js, la "View" generalmente son las respuestas JSON, y la arquitectura se enfoca en separar responsabilidades entre controladores, servicios y modelos.

## Estructura de proyecto con MVC

```
src/
├── controllers/        → Manejo de requests/responses
│   ├── auth.controller.js
│   └── users.controller.js
├── services/           → Lógica de negocio
│   ├── auth.service.js
│   └── users.service.js
├── models/             → Esquemas de base de datos
│   └── user.model.js
├── routes/             → Definición de rutas
│   ├── auth.routes.js
│   └── users.routes.js
├── middlewares/        → Middlewares compartidos
│   ├── auth.middleware.js
│   └── validation.middleware.js
├── repositories/       → Acceso a datos (opcional, capa extra)
│   └── users.repository.js
└── app.js
```

## Model — Mongoose

```javascript
// models/user.model.js
const mongoose = require('mongoose');
const bcrypt = require('bcrypt');

const userSchema = new mongoose.Schema({
  name: { type: String, required: true, trim: true },
  email: { type: String, required: true, unique: true, lowercase: true },
  passwordHash: { type: String, required: true },
  role: { type: String, enum: ['user', 'admin'], default: 'user' },
  isActive: { type: Boolean, default: true }
}, { timestamps: true });

userSchema.pre('save', async function(next) {
  if (!this.isModified('passwordHash')) return next();
  this.passwordHash = await bcrypt.hash(this.passwordHash, 12);
  next();
});

userSchema.methods.comparePassword = function(password) {
  return bcrypt.compare(password, this.passwordHash);
};

// No exponer el hash en las respuestas
userSchema.methods.toJSON = function() {
  const obj = this.toObject();
  delete obj.passwordHash;
  return obj;
};

module.exports = mongoose.model('User', userSchema);
```

## Service — Lógica de negocio

```javascript
// services/users.service.js
const User = require('../models/user.model');
const { NotFoundError, ConflictError } = require('../utils/errors');

class UsersService {
  async getAll(filters = {}) {
    const { page = 1, limit = 10, search } = filters;
    const query = search ? { name: new RegExp(search, 'i') } : {};

    const [users, total] = await Promise.all([
      User.find(query).skip((page - 1) * limit).limit(limit),
      User.countDocuments(query)
    ]);

    return { users, total, pages: Math.ceil(total / limit) };
  }

  async getById(id) {
    const user = await User.findById(id);
    if (!user) throw new NotFoundError(`Usuario ${id} no encontrado`);
    return user;
  }

  async create(data) {
    const exists = await User.findOne({ email: data.email });
    if (exists) throw new ConflictError('El email ya está registrado');

    return User.create(data);
  }

  async update(id, data) {
    const user = await User.findByIdAndUpdate(id, data, { new: true, runValidators: true });
    if (!user) throw new NotFoundError(`Usuario ${id} no encontrado`);
    return user;
  }

  async delete(id) {
    const user = await User.findByIdAndDelete(id);
    if (!user) throw new NotFoundError(`Usuario ${id} no encontrado`);
  }
}

module.exports = new UsersService();
```

## Controller — Manejo HTTP

```javascript
// controllers/users.controller.js
const usersService = require('../services/users.service');

class UsersController {
  async getAll(req, res, next) {
    try {
      const result = await usersService.getAll(req.query);
      res.json(result);
    } catch (error) {
      next(error);
    }
  }

  async getById(req, res, next) {
    try {
      const user = await usersService.getById(req.params.id);
      res.json(user);
    } catch (error) {
      next(error);
    }
  }

  async create(req, res, next) {
    try {
      const user = await usersService.create(req.body);
      res.status(201).json(user);
    } catch (error) {
      next(error);
    }
  }
}

module.exports = new UsersController();
```

## Routes

```javascript
// routes/users.routes.js
const router = require('express').Router();
const usersController = require('../controllers/users.controller');
const { authenticate, authorize } = require('../middlewares/auth.middleware');
const { validateUser } = require('../middlewares/validation.middleware');

router.get('/', authenticate, usersController.getAll);
router.get('/:id', authenticate, usersController.getById);
router.post('/', authenticate, authorize('admin'), validateUser, usersController.create);

module.exports = router;
```

**Consejo de entrevista**: La separación entre Controller y Service es importante: el Controller solo maneja HTTP (req, res, status codes), mientras que el Service contiene la lógica de negocio sin conocer HTTP. Esto hace que el Service sea testeable de forma independiente y reutilizable desde diferentes contextos (WebSocket, CLI, testing).
