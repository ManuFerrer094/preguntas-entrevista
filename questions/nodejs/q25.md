**NestJS** es un framework de Node.js progresivo construido con TypeScript que proporciona una arquitectura robusta inspirada en Angular. Usa decoradores, inyección de dependencias y módulos para crear aplicaciones escalables y mantenibles.

## Arquitectura de NestJS

```
src/
├── app.module.ts        ← Módulo raíz
├── main.ts              ← Punto de entrada
├── usuarios/
│   ├── usuarios.module.ts
│   ├── usuarios.controller.ts
│   ├── usuarios.service.ts
│   ├── dto/
│   │   ├── crear-usuario.dto.ts
│   │   └── actualizar-usuario.dto.ts
│   └── entities/
│       └── usuario.entity.ts
```

## Instalación y estructura básica

```bash
npm install -g @nestjs/cli
nest new mi-proyecto
cd mi-proyecto
```

```typescript
// main.ts — Punto de entrada
import { NestFactory } from '@nestjs/core';
import { AppModule } from './app.module';
import { ValidationPipe } from '@nestjs/common';

async function bootstrap() {
  const app = await NestFactory.create(AppModule);

  // Activar validación global
  app.useGlobalPipes(new ValidationPipe({
    whitelist: true,        // Elimina propiedades no declaradas en DTOs
    forbidNonWhitelisted: true,  // Lanza error si hay propiedades extra
    transform: true,        // Transforma tipos automáticamente (string → number)
  }));

  app.setGlobalPrefix('api/v1');
  app.enableCors();

  await app.listen(3000);
  console.log('🚀 App ejecutándose en http://localhost:3000/api/v1');
}
bootstrap();
```

## Controller — Maneja las rutas HTTP

```typescript
// usuarios.controller.ts
import {
  Controller, Get, Post, Put, Delete,
  Param, Body, Query, HttpCode, HttpStatus,
  UseGuards, ParseIntPipe
} from '@nestjs/common';
import { UsuariosService } from './usuarios.service';
import { CrearUsuarioDto } from './dto/crear-usuario.dto';
import { ActualizarUsuarioDto } from './dto/actualizar-usuario.dto';
import { JwtAuthGuard } from '../auth/jwt-auth.guard';

@Controller('usuarios')  // Prefijo: /api/v1/usuarios
export class UsuariosController {
  constructor(private readonly usuariosService: UsuariosService) {}

  @Get()
  findAll(@Query('activo') activo?: string) {
    return this.usuariosService.findAll({ activo: activo === 'true' });
  }

  @Get(':id')
  findOne(@Param('id', ParseIntPipe) id: number) {  // ParseIntPipe valida y convierte
    return this.usuariosService.findOne(id);
  }

  @Post()
  @HttpCode(HttpStatus.CREATED)
  create(@Body() dto: CrearUsuarioDto) {
    return this.usuariosService.create(dto);
  }

  @Put(':id')
  @UseGuards(JwtAuthGuard)  // Proteger con autenticación
  update(@Param('id', ParseIntPipe) id: number, @Body() dto: ActualizarUsuarioDto) {
    return this.usuariosService.update(id, dto);
  }

  @Delete(':id')
  @UseGuards(JwtAuthGuard)
  @HttpCode(HttpStatus.NO_CONTENT)
  remove(@Param('id', ParseIntPipe) id: number) {
    return this.usuariosService.remove(id);
  }
}
```

## Service — Lógica de negocio

```typescript
// usuarios.service.ts
import { Injectable, NotFoundException, ConflictException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { Usuario } from './entities/usuario.entity';
import { CrearUsuarioDto } from './dto/crear-usuario.dto';

@Injectable()  // Hace la clase inyectable en el sistema DI de NestJS
export class UsuariosService {
  constructor(
    @InjectRepository(Usuario)
    private readonly usuarioRepository: Repository<Usuario>
  ) {}

  async findAll({ activo }: { activo?: boolean } = {}): Promise<Usuario[]> {
    return this.usuarioRepository.find({
      where: activo !== undefined ? { activo } : {},
      order: { createdAt: 'DESC' }
    });
  }

  async findOne(id: number): Promise<Usuario> {
    const usuario = await this.usuarioRepository.findOne({ where: { id } });
    if (!usuario) throw new NotFoundException(`Usuario #${id} no encontrado`);
    return usuario;
  }

  async create(dto: CrearUsuarioDto): Promise<Usuario> {
    const existente = await this.usuarioRepository.findOne({
      where: { email: dto.email }
    });
    if (existente) throw new ConflictException('El email ya está registrado');

    const usuario = this.usuarioRepository.create(dto);
    return this.usuarioRepository.save(usuario);
  }

  async update(id: number, dto: ActualizarUsuarioDto): Promise<Usuario> {
    const usuario = await this.findOne(id);  // Lanza NotFoundException si no existe
    Object.assign(usuario, dto);
    return this.usuarioRepository.save(usuario);
  }

  async remove(id: number): Promise<void> {
    const usuario = await this.findOne(id);
    await this.usuarioRepository.remove(usuario);
  }
}
```

## DTO con validación (class-validator)

```typescript
// dto/crear-usuario.dto.ts
import {
  IsEmail, IsString, MinLength, MaxLength,
  IsOptional, IsEnum, IsNumber, Min, Max
} from 'class-validator';
import { Transform } from 'class-transformer';

export enum RolUsuario {
  ADMIN = 'admin',
  USUARIO = 'usuario',
  MODERADOR = 'moderador'
}

export class CrearUsuarioDto {
  @IsString()
  @MinLength(2, { message: 'El nombre debe tener al menos 2 caracteres' })
  @MaxLength(100)
  @Transform(({ value }) => value?.trim())  // Eliminar espacios
  nombre: string;

  @IsEmail({}, { message: 'El email no es válido' })
  @Transform(({ value }) => value?.toLowerCase())
  email: string;

  @IsString()
  @MinLength(8)
  password: string;

  @IsOptional()
  @IsEnum(RolUsuario)
  rol?: RolUsuario = RolUsuario.USUARIO;

  @IsOptional()
  @IsNumber()
  @Min(0)
  @Max(150)
  edad?: number;
}
```

## Módulo — Agrupa componentes relacionados

```typescript
// usuarios.module.ts
import { Module } from '@nestjs/common';
import { TypeOrmModule } from '@nestjs/typeorm';
import { UsuariosController } from './usuarios.controller';
import { UsuariosService } from './usuarios.service';
import { Usuario } from './entities/usuario.entity';

@Module({
  imports: [TypeOrmModule.forFeature([Usuario])],
  controllers: [UsuariosController],
  providers: [UsuariosService],
  exports: [UsuariosService],  // Exportar para usar en otros módulos
})
export class UsuariosModule {}
```

**Consejo de entrevista**: NestJS destaca en proyectos empresariales por su estructura opinionada y bien documentada. Sus ventajas son: DI nativa (igual que Angular), decoradores para rutas/validación/autenticación, soporte de primera clase para TypeScript, módulos bien organizados. Es ideal cuando el equipo viene de Angular o Java Spring. Para proyectos pequeños, Express o Fastify solos pueden ser suficientes. Menciona que NestJS soporta múltiples transportes (REST, GraphQL, WebSockets, microservicios) manteniendo la misma arquitectura.
