La **arquitectura hexagonal** (también llamada Ports & Adapters o Clean Architecture) separa la lógica de negocio del mundo exterior (bases de datos, APIs, UI), haciendo el código más testeable, mantenible e independiente de frameworks y tecnologías.

## Conceptos fundamentales

```
┌─────────────────────────────────────────────────┐
│                 Adaptadores                      │
│  ┌──────────┐  ┌─────────────────┐  ┌─────────┐ │
│  │  HTTP    │  │   PostgreSQL     │  │  Redis  │ │
│  │ Controller│  │  Repository Impl│  │ Cache   │ │
│  └────┬─────┘  └───────┬─────────┘  └────┬────┘ │
│       │                │                  │      │
│  ┌────▼────────────────▼──────────────────▼────┐ │
│  │              PUERTOS (Interfaces)            │ │
│  │  IUsuarioController  IUsuarioRepository      │ │
│  └────────────────┬─────────────────────────────┘ │
│                   │                               │
│  ┌────────────────▼───────────────────────────┐  │
│  │          DOMINIO (Lógica de negocio)        │  │
│  │  Entities  Use Cases  Domain Services       │  │
│  └────────────────────────────────────────────┘  │
└─────────────────────────────────────────────────┘
```

## Domain Layer — Entidades y Value Objects

```typescript
// domain/value-objects/Email.ts
export class Email {
  private readonly valor: string;

  constructor(email: string) {
    if (!email.includes('@') || email.length < 5) {
      throw new Error(`Email inválido: ${email}`);
    }
    this.valor = email.toLowerCase().trim();
  }

  toString(): string { return this.valor; }
  equals(otro: Email): boolean { return this.valor === otro.valor; }
}

// domain/entities/Usuario.ts
import { Email } from '../value-objects/Email';

export class Usuario {
  private readonly _id: string;
  private _nombre: string;
  private readonly _email: Email;
  private _activo: boolean;
  private readonly _fechaRegistro: Date;

  constructor(params: {
    id: string;
    nombre: string;
    email: string;
    activo?: boolean;
    fechaRegistro?: Date;
  }) {
    this._id = params.id;
    this._nombre = params.nombre;
    this._email = new Email(params.email);
    this._activo = params.activo ?? true;
    this._fechaRegistro = params.fechaRegistro ?? new Date();
  }

  // Getters
  get id() { return this._id; }
  get nombre() { return this._nombre; }
  get email() { return this._email.toString(); }
  get activo() { return this._activo; }

  // Lógica de negocio EN la entidad
  cambiarNombre(nuevoNombre: string): void {
    if (nuevoNombre.trim().length < 2) {
      throw new Error('El nombre debe tener al menos 2 caracteres');
    }
    this._nombre = nuevoNombre.trim();
  }

  desactivar(): void {
    if (!this._activo) throw new Error('El usuario ya está desactivado');
    this._activo = false;
  }

  toPlainObject() {
    return {
      id: this._id,
      nombre: this._nombre,
      email: this.email,
      activo: this._activo,
      fechaRegistro: this._fechaRegistro
    };
  }
}
```

## Domain Layer — Puerto (Interface del Repositorio)

```typescript
// domain/repositories/IUsuarioRepository.ts
import { Usuario } from '../entities/Usuario';

export interface IUsuarioRepository {
  findById(id: string): Promise<Usuario | null>;
  findByEmail(email: string): Promise<Usuario | null>;
  findAll(filtros?: { activo?: boolean }): Promise<Usuario[]>;
```

## Application Layer — Casos de uso

```typescript
// application/use-cases/RegistrarUsuario.ts
import { randomUUID } from 'crypto';
import { IUsuarioRepository } from '../../domain/repositories/IUsuarioRepository';
import { Usuario } from '../../domain/entities/Usuario';
import { IEmailService } from '../../domain/services/IEmailService';

interface RegistrarUsuarioDTO {
  nombre: string;
  email: string;
  password: string;
}

export class RegistrarUsuario {
  constructor(
    private readonly usuarioRepo: IUsuarioRepository,
    private readonly emailService: IEmailService,
    private readonly hashPassword: (pwd: string) => Promise<string>
  ) {}

  async execute(dto: RegistrarUsuarioDTO): Promise<Usuario> {
    // Regla de negocio: email único
    const existente = await this.usuarioRepo.findByEmail(dto.email);
    if (existente) {
      throw new Error('El email ya está registrado');
    }

    const passwordHash = await this.hashPassword(dto.password);
    const usuario = new Usuario({
      id: randomUUID(),
      nombre: dto.nombre,
      email: dto.email,
      // El hash se gestiona por separado, no en la entidad
    });

    const guardado = await this.usuarioRepo.save(usuario);

    // Efecto secundario: enviar email de bienvenida
    await this.emailService.enviarBienvenida(guardado.email, guardado.nombre);

    return guardado;
  }
}
```

## Infrastructure Layer — Adaptador de Repositorio

```typescript
// infrastructure/database/PostgresUsuarioRepository.ts
import { IUsuarioRepository } from '../../domain/repositories/IUsuarioRepository';
import { Usuario } from '../../domain/entities/Usuario';
import { prisma } from '../prisma';

export class PostgresUsuarioRepository implements IUsuarioRepository {
  async findById(id: string): Promise<Usuario | null> {
    const row = await prisma.usuario.findUnique({ where: { id } });
    if (!row) return null;
    return new Usuario(row);  // Mapear de registro DB a entidad de dominio
  }

  async findByEmail(email: string): Promise<Usuario | null> {
    const row = await prisma.usuario.findUnique({ where: { email } });
    return row ? new Usuario(row) : null;
  }

  async findAll(filtros = {}): Promise<Usuario[]> {
    const rows = await prisma.usuario.findMany({ where: filtros });
    return rows.map(row => new Usuario(row));
  }

  async save(usuario: Usuario): Promise<Usuario> {
    const data = usuario.toPlainObject();
    const row = await prisma.usuario.upsert({
      where: { id: data.id },
      create: data,
      update: data
    });
    return new Usuario(row);
  }

  async delete(id: string): Promise<void> {
    await prisma.usuario.delete({ where: { id } });
  }
}
```

## Composición de dependencias (main.ts)

```typescript
// infrastructure/composition-root.ts — Punto donde se ensambla todo
import { PostgresUsuarioRepository } from './database/PostgresUsuarioRepository';
import { SendGridEmailService } from './email/SendGridEmailService';
import { RegistrarUsuario } from '../application/use-cases/RegistrarUsuario';
import { bcrypt } from 'bcryptjs';

export function crearContenedor() {
  const usuarioRepo = new PostgresUsuarioRepository();
  const emailService = new SendGridEmailService(process.env.SENDGRID_KEY!);
  const hashPassword = (pwd: string) => bcrypt.hash(pwd, 10);

  return {
    registrarUsuario: new RegistrarUsuario(usuarioRepo, emailService, hashPassword),
  };
}
```

**Consejo de entrevista**: La arquitectura hexagonal resuelve el problema de acoplamiento entre la lógica de negocio y los detalles técnicos. Los beneficios son: 1) tests unitarios de la lógica de negocio sin necesitar BD ni servidor, 2) intercambiar implementaciones (cambiar Postgres por MongoDB) sin tocar el dominio, 3) código que se entiende por su lógica, no por su tecnología. El costo es más código inicial y curva de aprendizaje. Es más valioso en aplicaciones complejas con mucha lógica de negocio.
