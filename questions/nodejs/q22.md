El módulo `crypto` de Node.js proporciona funcionalidades criptográficas: hashing, HMAC, cifrado simétrico/asimétrico y generación de bytes aleatorios. Es esencial para implementar autenticación y proteger datos sensibles.

## Hashing de contraseñas (con bcrypt o argon2)

```javascript
// ❌ No uses crypto para contraseñas — usa bcrypt o argon2
// crypto.createHash('sha256') no es seguro para contraseñas
// (sin sal, demasiado rápido, susceptible a ataques de diccionario)

const bcrypt = require('bcrypt');

// Hash de contraseña
async function hashPassword(password) {
  const saltRounds = 12; // Mayor número = más lento y más seguro
  return bcrypt.hash(password, saltRounds);
}

// Verificar contraseña
async function verifyPassword(password, hash) {
  return bcrypt.compare(password, hash);
}

// En un servicio de autenticación
const hash = await hashPassword('miContraseñaSegura');
// '$2b$12$...' — incluye la sal embebida
const isValid = await verifyPassword('miContraseñaSegura', hash); // true
```

## Hashing de datos con crypto (SHA-256, MD5)

```javascript
const { createHash, createHmac } = require('crypto');

// Hash simple (para datos no sensibles: fingerprints, checksums)
function sha256(data) {
  return createHash('sha256').update(data).digest('hex');
}

console.log(sha256('hola')); // '4d967a30111bf29f0eba01c448b375c1629b2fed01cdfcc3aed91f1b57d5dd5e'

// Hash de ficheros
const fs = require('fs');
function hashFile(filePath) {
  return new Promise((resolve, reject) => {
    const hash = createHash('sha256');
    const stream = fs.createReadStream(filePath);
    stream.on('data', chunk => hash.update(chunk));
    stream.on('end', () => resolve(hash.digest('hex')));
    stream.on('error', reject);
  });
}

// HMAC: hash con clave secreta (integridad + autenticación)
function createHmacSignature(data, secret) {
  return createHmac('sha256', secret).update(data).digest('hex');
}

function verifyHmacSignature(data, secret, expectedSignature) {
  const signature = createHmacSignature(data, secret);
  // Comparación segura para evitar timing attacks
  const { timingSafeEqual } = require('crypto');
  const sig1 = Buffer.from(signature);
  const sig2 = Buffer.from(expectedSignature);
  return sig1.length === sig2.length && timingSafeEqual(sig1, sig2);
}

// Firmar webhooks (patrón de GitHub/Stripe)
const webhookSecret = process.env.WEBHOOK_SECRET;
const signature = createHmacSignature(req.body, webhookSecret);
// El receptor verifica que la firma coincide
```

## Generación de tokens seguros

```javascript
const { randomBytes, randomUUID } = require('crypto');

// Token aleatorio (para reset de contraseña, tokens de sesión)
function generateToken(bytes = 32) {
  return randomBytes(bytes).toString('hex'); // 64 caracteres hex
}

// UUID v4 nativo (Node.js 14.17+)
function generateId() {
  return randomUUID(); // '9b1deb4d-3b7d-4bad-9bdd-2b0d7b3dcb6d'
}

// Token de reset de contraseña
async function createPasswordResetToken(userId) {
  const token = generateToken();
  const expires = new Date(Date.now() + 60 * 60 * 1000); // 1 hora

  // Guardar el hash del token (no el token en claro) en la BD
  const tokenHash = sha256(token);
  await User.findByIdAndUpdate(userId, {
    resetPasswordToken: tokenHash,
    resetPasswordExpires: expires
  });

  return token; // Enviar este token al usuario por email
}

async function verifyPasswordResetToken(token) {
  const tokenHash = sha256(token);
  return User.findOne({
    resetPasswordToken: tokenHash,
    resetPasswordExpires: { $gt: new Date() }
  });
}
```

## Cifrado simétrico (AES-256-GCM)

```javascript
const { createCipheriv, createDecipheriv, randomBytes } = require('crypto');

const ALGORITHM = 'aes-256-gcm';
const KEY_LENGTH = 32; // 256 bits

// La clave debería venir de una variable de entorno como Buffer de 32 bytes
const key = Buffer.from(process.env.ENCRYPTION_KEY, 'hex'); // 64 hex chars = 32 bytes

function encrypt(plaintext) {
  const iv = randomBytes(12); // 96 bits para GCM
  const cipher = createCipheriv(ALGORITHM, key, iv);

  const encrypted = Buffer.concat([
    cipher.update(plaintext, 'utf8'),
    cipher.final()
  ]);

  const authTag = cipher.getAuthTag(); // Para verificar integridad

  // Devolver IV + authTag + datos cifrados como string
  return Buffer.concat([iv, authTag, encrypted]).toString('base64');
}

function decrypt(encryptedData) {
  const buffer = Buffer.from(encryptedData, 'base64');
  const iv = buffer.subarray(0, 12);
  const authTag = buffer.subarray(12, 28);
  const encrypted = buffer.subarray(28);

  const decipher = createDecipheriv(ALGORITHM, key, iv);
  decipher.setAuthTag(authTag);

  return Buffer.concat([
    decipher.update(encrypted),
    decipher.final()
  ]).toString('utf8');
}

// Uso: cifrar datos sensibles en la base de datos
const encryptedPhone = encrypt('+34 666 777 888');
const decryptedPhone = decrypt(encryptedPhone);
```

**Consejo de entrevista**: Para contraseñas, usa siempre `bcrypt` o `argon2`, nunca SHA-256 sin sal. Para tokens aleatorios, usa `crypto.randomBytes()` en lugar de `Math.random()` (no es criptográficamente seguro). Cuando compares tokens o hashes, usa `timingSafeEqual` para prevenir timing attacks. Para cifrado de datos en la BD (PII), AES-256-GCM es la opción estándar porque incluye autenticación del texto cifrado.
