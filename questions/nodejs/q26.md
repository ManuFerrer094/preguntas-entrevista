Un **job scheduler** permite ejecutar tareas programadas en momentos específicos (cada hora, todos los días a las 3 AM, etc.). En Node.js, la librería más popular es `node-cron`.

## Instalación y sintaxis cron

```bash
npm install node-cron
```

```
Sintaxis cron: * * * * * *
               │ │ │ │ │ │
               │ │ │ │ │ └── Día de la semana (0-7, 0 y 7 = domingo)
               │ │ │ │ └──── Mes (1-12)
               │ │ │ └────── Día del mes (1-31)
               │ │ └──────── Hora (0-23)
               │ └────────── Minuto (0-59)
               └──────────── Segundo (0-59) — opcional en node-cron
```

## Uso básico

```javascript
import cron from 'node-cron';

// Ejecutar cada minuto
cron.schedule('* * * * *', () => {
  console.log('Ejecutando cada minuto:', new Date().toISOString());
});

// Ejecutar cada día a las 3:00 AM
cron.schedule('0 3 * * *', async () => {
  console.log('Limpieza diaria:', new Date().toISOString());
  await limpiarArchivosTemporales();
});

// Ejecutar cada lunes a las 9:00 AM
cron.schedule('0 9 * * 1', async () => {
  await enviarResumenSemanal();
});

// Ejecutar cada hora en punto
cron.schedule('0 * * * *', async () => {
  await sincronizarDatos();
});

// Expresiones especiales
cron.schedule('@hourly', () => { /* cada hora */ });
cron.schedule('@daily', () => { /* cada día a medianoche */ });
cron.schedule('@weekly', () => { /* cada domingo a medianoche */ });
cron.schedule('@monthly', () => { /* primer día del mes */ });
```

## Sistema de jobs estructurado

```javascript
// jobs/limpiezaTemporales.job.js
import cron from 'node-cron';
import { unlink, readdir } from 'fs/promises';
import path from 'path';

const DIRECTORIO_TEMP = './tmp';
const EDAD_MAXIMA_HORAS = 24;

async function limpiarArchivosTemporales() {
  try {
    const archivos = await readdir(DIRECTORIO_TEMP);
    const ahora = Date.now();
    let eliminados = 0;

    for (const archivo of archivos) {
      const ruta = path.join(DIRECTORIO_TEMP, archivo);
      const stats = await fs.stat(ruta);
      const edadHoras = (ahora - stats.mtime.getTime()) / (1000 * 60 * 60);

      if (edadHoras > EDAD_MAXIMA_HORAS) {
        await unlink(ruta);
        eliminados++;
      }
    }

    console.log(`[LimpiezaJob] ${eliminados} archivos eliminados`);
    return { eliminados };
  } catch (error) {
    console.error('[LimpiezaJob] Error:', error.message);
    throw error;
  }
}

export function iniciarJobLimpieza() {
  return cron.schedule('0 2 * * *', limpiarArchivosTemporales, {
    name: 'limpieza-temporales',
    timezone: 'Europe/Madrid'  // Zona horaria específica
  });
}
```

## Gestor centralizado de jobs

```javascript
// jobs/index.js
import cron from 'node-cron';
import { iniciarJobLimpieza } from './limpiezaTemporales.job.js';
import { iniciarJobReportes } from './reportes.job.js';
import { iniciarJobSincronizacion } from './sincronizacion.job.js';

const jobs = new Map();

export function iniciarTodosLosJobs() {
  jobs.set('limpieza', iniciarJobLimpieza());
  jobs.set('reportes', iniciarJobReportes());
  jobs.set('sincronizacion', iniciarJobSincronizacion());

  console.log(`✅ ${jobs.size} jobs programados`);
}

export function detenerJob(nombre) {
  const job = jobs.get(nombre);
  if (job) {
    job.stop();
    console.log(`Job '${nombre}' detenido`);
  }
}

export function listarJobs() {
  return [...jobs.entries()].map(([nombre, task]) => ({
    nombre,
    activo: task.options.scheduled !== false
  }));
}

// Graceful shutdown
process.on('SIGTERM', () => {
  console.log('Deteniendo todos los jobs...');
  jobs.forEach(job => job.stop());
  process.exit(0);
});
```

## Jobs con reintentos y manejo de errores

```javascript
async function ejecutarConReintento(tarea, maxReintentos = 3, esperaMs = 1000) {
  for (let intento = 1; intento <= maxReintentos; intento++) {
    try {
      await tarea();
      return;  // Éxito
    } catch (error) {
      console.error(`Intento ${intento}/${maxReintentos} falló:`, error.message);

      if (intento === maxReintentos) {
        console.error('Todos los reintentos fallaron');
        throw error;  // Re-lanzar después del último intento
      }

      // Espera exponencial entre reintentos
      const espera = esperaMs * Math.pow(2, intento - 1);
      await new Promise(resolve => setTimeout(resolve, espera));
    }
  }
}

// Uso en un job
cron.schedule('0 * * * *', async () => {
  await ejecutarConReintento(
    () => sincronizarConAPIExterna(),
    3,   // 3 reintentos
    2000 // 2s, 4s, 8s (backoff exponencial)
  );
});
```

## Alternativas más avanzadas

```javascript
// BullMQ: para jobs distribuidos con Redis (más robusto para producción)
// npm install bullmq ioredis
import { Queue, Worker } from 'bullmq';

const colaEmails = new Queue('emails', { connection: { host: 'localhost', port: 6379 } });

// Agregar job a la cola
await colaEmails.add('bienvenida', { usuarioId: 123 }, {
  delay: 0,
  attempts: 3,
  backoff: { type: 'exponential', delay: 1000 }
});

// Job repetible (cron con BullMQ)
await colaEmails.add('reporte-diario', {}, {
  repeat: { cron: '0 9 * * *', tz: 'Europe/Madrid' }
});
```

**Consejo de entrevista**: `node-cron` es suficiente para apps de servidor único. Para sistemas distribuidos (múltiples instancias de Node.js), usa **BullMQ** con Redis: garantiza que el job se ejecuta **una sola vez** aunque haya múltiples instancias. Menciona también el problema de jobs de larga duración: si el servidor se reinicia durante la ejecución, el job se cancela sin completarse —BullMQ también soluciona esto con persistencia de estado en Redis.
