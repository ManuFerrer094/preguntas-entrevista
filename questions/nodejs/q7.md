---
title: ¿Qué son los middlewares en Express?
difficulty: medium
tags: [nodejs, express, middleware, http]
---

Los middlewares en Express son funciones que tienen acceso al objeto de solicitud (`req`), la respuesta (`res`) y la siguiente función en el ciclo de solicitud-respuesta (`next`). Son la forma principal de estructurar la lógica de una aplicación Express: cada middleware puede leer/modificar `req` y `res`, terminar el ciclo o pasar al siguiente middleware llamando `next()`.

La firma de un middleware es: `(req, res, next) => void`

## Tipos de middlewares

**1. Middleware de aplicación (global):**

```javascript
const express = require('express');
const app = express();

// Middleware de logging
app.use((req, res, next) => {
  console.log(`${new Date().toISOString()} ${req.method} ${req.url}`);
  next(); // Imprescindible para continuar la cadena
});

// Middleware de parseo de body
app.use(express.json());
app.use(express.urlencoded({ extended: true }));
```

**2. Middleware de ruta:**

```javascript
// Solo para rutas que empiecen con /api
app.use('/api', (req, res, next) => {
  const apiKey = req.headers['x-api-key'];
  if (!apiKey || !isValidApiKey(apiKey)) {
    return res.status(401).json({ error: 'API key inválida' });
  }
  next();
});
```

**3. Middleware de autenticación:**

```javascript
const authMiddleware = async (req, res, next) => {
  const token = req.headers.authorization?.split(' ')[1];

  if (!token) {
    return res.status(401).json({ error: 'Token requerido' });
  }

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = decoded; // Añadir datos al request para middlewares posteriores
    next();
  } catch (error) {
    return res.status(401).json({ error: 'Token inválido' });
  }
};

// Aplicar solo a rutas protegidas
app.get('/profile', authMiddleware, (req, res) => {
  res.json({ user: req.user });
});
```

**4. Middleware de manejo de errores (4 parámetros):**

```javascript
// DEBE tener exactamente 4 parámetros para que Express lo trate como error handler
app.use((err, req, res, next) => {
  console.error(err.stack);

  if (err.name === 'ValidationError') {
    return res.status(400).json({ error: err.message });
  }

  if (err.name === 'UnauthorizedError') {
    return res.status(401).json({ error: 'No autorizado' });
  }

  res.status(500).json({ error: 'Error interno del servidor' });
});
```

## Creando middlewares reutilizables

```javascript
// middleware/rateLimit.js
const rateLimitMiddleware = (options = {}) => {
  const { max = 100, windowMs = 60000 } = options;
  const requests = new Map();

  return (req, res, next) => {
    const ip = req.ip;
    const now = Date.now();
    const userRequests = requests.get(ip) || [];

    // Filtrar requests fuera de la ventana de tiempo
    const recent = userRequests.filter(time => now - time < windowMs);

    if (recent.length >= max) {
      return res.status(429).json({
        error: 'Demasiadas solicitudes. Por favor espera.'
      });
    }

    recent.push(now);
    requests.set(ip, recent);
    next();
  };
};

app.use('/api', rateLimitMiddleware({ max: 50, windowMs: 60000 }));
```

## Orden de middlewares — crucial

```javascript
// El orden importa: se ejecutan en el orden en que se registran
app.use(cors());           // 1. CORS primero
app.use(helmet());         // 2. Seguridad
app.use(morgan('dev'));    // 3. Logging
app.use(express.json());  // 4. Parseo de body
app.use('/api', authMiddleware); // 5. Autenticación para /api

// Rutas
app.use('/api/users', usersRouter);
app.use('/api/products', productsRouter);

// Error handler — SIEMPRE al final
app.use(errorHandler);
```

**Consejo de entrevista**: El middleware es el patrón central de Express. Un error muy común es olvidar llamar a `next()`, lo que hace que la solicitud quede "colgada". También es importante recordar que el middleware de manejo de errores debe tener exactamente 4 parámetros `(err, req, res, next)`, de lo contrario Express no lo reconoce como error handler.
