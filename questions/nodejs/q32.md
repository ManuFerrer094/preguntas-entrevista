La **internacionalización (i18n)** en una API Node.js permite devolver respuestas, mensajes de error y contenido en el idioma preferido del usuario, mejorando la experiencia para audiencias globales.

## Configuración de i18next con Express

```javascript
// config/i18n.js
import i18next from 'i18next';
import Backend from 'i18next-fs-backend';
import middleware from 'i18next-http-middleware';
import { fileURLToPath } from 'url';
import path from 'path';

const __dirname = path.dirname(fileURLToPath(import.meta.url));

await i18next
  .use(Backend)
  .use(middleware.LanguageDetector)  // Detecta idioma del request
  .init({
    backend: {
      loadPath: path.join(__dirname, '../locales/{{lng}}/{{ns}}.json'),
    },
    fallbackLng: 'es',           // Idioma de fallback
    supportedLngs: ['es', 'en', 'fr'],
    preload: ['es', 'en', 'fr'],  // Precargar en el arranque
    ns: ['translation'],
    defaultNS: 'translation',
    detection: {
      order: [
        'header',         // Accept-Language header
        'querystring',    // ?lang=es
        'cookie',         // Cookie 'i18n'
      ],
      lookupHeader: 'accept-language',
      lookupQuerystring: 'lang',
      lookupCookie: 'i18n',
    },
    interpolation: {
      escapeValue: false  // Las APIs no renderizan HTML
    }
  });

export { i18next, middleware };
```

```javascript
// server.js
import express from 'express';
import { i18next, middleware } from './config/i18n.js';

const app = express();

// Aplicar middleware de i18n ANTES de las rutas
app.use(middleware.handle(i18next));

// Ahora req.t está disponible en todos los handlers
app.get('/api/perfil', autenticar, (req, res) => {
  const usuario = req.user;
  res.json({
    mensaje: req.t('exito.sesionIniciada', { nombre: usuario.nombre }),
    usuario
  });
});

// Manejo de errores localizado
app.use((err, req, res, next) => {
  const statusCode = err.statusCode ?? 500;
  const claveError = err.i18nKey ?? 'errores.errorGeneral';

  res.status(statusCode).json({
    error: req.t(claveError, err.i18nParams ?? {}),
    codigo: err.code
  });
});
```

## Errores localizados con clases personalizadas

```javascript
// errors/HttpError.js
export class HttpError extends Error {
  constructor(statusCode, i18nKey, i18nParams = {}) {
    super(i18nKey);
    this.statusCode = statusCode;
    this.i18nKey = i18nKey;
    this.i18nParams = i18nParams;
  }
}

// Uso en servicios/controladores:
throw new HttpError(401, 'errores.tokenExpirado');
throw new HttpError(404, 'errores.recursoNoEncontrado');
throw new HttpError(201, 'exito.usuarioCreado', { nombre: 'Ana' });
```

## Detectar idioma del usuario

```javascript
// El middleware de i18next detecta automáticamente el idioma desde:

// 1. Header Accept-Language (navegadores y apps móviles)
// GET /api/datos
// Accept-Language: es-ES,es;q=0.9,en;q=0.8

// 2. Query parameter
// GET /api/datos?lang=en

// 3. Manualmente en el código
function getT(idioma) {
  return i18next.getFixedT(idioma);
}

// Para emails (fuera del contexto HTTP):
async function enviarEmailBienvenida(usuario) {
  const t = getT(usuario.idioma ?? 'es');  // Idioma preferido del usuario

  await emailService.send({
    para: usuario.email,
    asunto: t('emails.bienvenida.asunto', { appName: 'MiApp' }),
    cuerpo: t('emails.bienvenida.cuerpo', { nombre: usuario.nombre })
  });
}
```

## Formateo de fechas y números por locale

```javascript
// El idioma detectado por i18next también se puede usar para formateo
app.get('/api/estadisticas', autenticar, (req, res) => {
  const locale = req.language ?? 'es-ES';  // e.g. 'es-ES', 'en-US'
  const stats = obtenerEstadisticas();

  const formatearFecha = (fecha) =>
    new Intl.DateTimeFormat(locale, {
      dateStyle: 'long',
      timeStyle: 'short'
    }).format(fecha);

  const formatearMoneda = (importe) =>
    new Intl.NumberFormat(locale, {
      style: 'currency',
      currency: locale.startsWith('en') ? 'USD' : 'EUR'
    }).format(importe);

  res.json({
    totalVentas: formatearMoneda(stats.total),
    ultimaActualizacion: formatearFecha(stats.fecha),
    mensaje: req.t('exito.estadisticasActualizadas')
  });
});
```

**Consejo de entrevista**: En una API REST, la internacionalización aplica a: mensajes de error, emails transaccionales, notificaciones push y contenido dinámico. La detección del idioma se hace por el header `Accept-Language` (estándar HTTP) o por la preferencia del usuario guardada en su perfil. Para emails, usa `i18next.getFixedT(idioma)` fuera del contexto HTTP. Menciona también que los strings en la base de datos (contenido creado por usuarios) se gestionan diferente —normalmente guardando una columna por idioma o usando una tabla de traducciones.
