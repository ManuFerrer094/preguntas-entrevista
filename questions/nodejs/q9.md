Los Worker Threads (módulo `worker_threads`, disponible desde Node.js 10) permiten ejecutar código JavaScript en hilos paralelos dentro del mismo proceso. A diferencia del módulo `cluster` (que crea procesos separados), los Worker Threads comparten memoria con el hilo principal mediante `SharedArrayBuffer` y `Atomics`, lo que los hace ideales para tareas CPU-intensivas.

El event loop de Node.js es single-threaded, lo que significa que una operación de CPU pesada bloquea todas las demás solicitudes. Los Worker Threads resuelven este problema moviendo ese trabajo a hilos separados.

## Caso de uso: cálculo pesado sin bloquear el servidor

```javascript
// worker.js - ejecutado en el hilo worker
const { workerData, parentPort } = require('worker_threads');

function heavyCalculation(data) {
  // Simulación de cálculo intensivo (fibonacci, compresión, etc.)
  let result = 0;
  for (let i = 0; i < data.iterations; i++) {
    result += Math.sqrt(i * data.multiplier);
  }
  return result;
}

const result = heavyCalculation(workerData);
parentPort.postMessage({ result });
```

```javascript
// main.js - hilo principal
const { Worker } = require('worker_threads');
const path = require('path');

function runWorker(data) {
  return new Promise((resolve, reject) => {
    const worker = new Worker(path.join(__dirname, 'worker.js'), {
      workerData: data
    });

    worker.on('message', resolve);
    worker.on('error', reject);
    worker.on('exit', (code) => {
      if (code !== 0) reject(new Error(`Worker terminó con código ${code}`));
    });
  });
}

// El servidor puede seguir respondiendo mientras el worker calcula
const express = require('express');
const app = express();

app.get('/calculate', async (req, res) => {
  try {
    const { result } = await runWorker({
      iterations: 100_000_000,
      multiplier: 2
    });
    res.json({ result });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.listen(3000);
```

## Pool de Workers

Para no crear y destruir workers en cada solicitud (costoso), se usa un pool:

```javascript
const { Worker } = require('worker_threads');
const os = require('os');

class WorkerPool {
  constructor(workerFile, size = os.cpus().length) {
    this.workers = [];
    this.queue = [];

    for (let i = 0; i < size; i++) {
      this.addWorker(workerFile);
    }
  }

  addWorker(workerFile) {
    const worker = new Worker(workerFile);
    this.workers.push({ worker, idle: true });

    worker.on('message', (result) => {
      const workerItem = this.workers.find(w => w.worker === worker);
      if (workerItem) workerItem.idle = true;

      // Resolver la promesa del llamador
      workerItem.resolve(result);

      // Procesar siguiente tarea en la cola
      this.processQueue();
    });
  }

  run(data) {
    return new Promise((resolve, reject) => {
      this.queue.push({ data, resolve, reject });
      this.processQueue();
    });
  }

  processQueue() {
    const idleWorker = this.workers.find(w => w.idle);
    if (!idleWorker || this.queue.length === 0) return;

    const task = this.queue.shift();
    idleWorker.idle = false;
    idleWorker.resolve = task.resolve;
    idleWorker.reject = task.reject;
    idleWorker.worker.postMessage(task.data);
  }
}

const pool = new WorkerPool('./worker.js');
```

## Memoria compartida con SharedArrayBuffer

```javascript
// Compartir memoria entre hilos (muy eficiente, sin serialización)
const sharedBuffer = new SharedArrayBuffer(4); // 4 bytes
const sharedArray = new Int32Array(sharedBuffer);

const worker = new Worker(`
  const { workerData } = require('worker_threads');
  const arr = new Int32Array(workerData.buffer);
  Atomics.add(arr, 0, 1); // Operación atómica, thread-safe
`, { eval: true, workerData: { buffer: sharedBuffer } });
```

**Consejo de entrevista**: Los Worker Threads son la solución para tareas CPU-intensivas en Node.js: procesamiento de imágenes, compresión, criptografía, parsing de archivos grandes. Para operaciones I/O (base de datos, peticiones HTTP), el event loop con async/await sigue siendo más eficiente. En producción, usar una librería como `piscina` que implementa un worker pool optimizado es preferible a hacerlo manualmente.
