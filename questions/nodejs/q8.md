El manejo de errores en Node.js es crucial para construir aplicaciones robustas. Node.js tiene varios mecanismos para capturar errores: try/catch para código síncrono y async/await, callbacks de error, eventos de error, y manejadores de procesos para errores no capturados.

## try/catch con async/await

```javascript
// La forma moderna y recomendada
async function getUserById(id) {
  try {
    const user = await db.users.findById(id);

    if (!user) {
      throw new NotFoundError(`Usuario ${id} no encontrado`);
    }

    return user;
  } catch (error) {
    if (error instanceof NotFoundError) {
      throw error; // Re-lanzar errores conocidos
    }
    // Envolver errores desconocidos con contexto
    throw new DatabaseError('Error al obtener usuario', { cause: error });
  }
}
```

## Clases de error personalizadas

```javascript
class AppError extends Error {
  constructor(message, statusCode = 500, options = {}) {
    super(message, options);
    this.name = this.constructor.name;
    this.statusCode = statusCode;
    this.isOperational = true; // Error conocido y esperado
    Error.captureStackTrace(this, this.constructor);
  }
}

class NotFoundError extends AppError {
  constructor(message) {
    super(message, 404);
  }
}

class ValidationError extends AppError {
  constructor(message, errors) {
    super(message, 400);
    this.errors = errors;
  }
}

class UnauthorizedError extends AppError {
  constructor(message = 'No autorizado') {
    super(message, 401);
  }
}
```

## Manejo centralizado en Express

```javascript
// middleware/errorHandler.js
const errorHandler = (err, req, res, next) => {
  // Log del error
  logger.error({
    message: err.message,
    stack: err.stack,
    url: req.url,
    method: req.method,
    user: req.user?.id
  });

  // Errores operacionales conocidos
  if (err.isOperational) {
    return res.status(err.statusCode).json({
      error: err.message,
      ...(err.errors && { errors: err.errors })
    });
  }

  // Errores de programación o desconocidos
  // En producción, no exponer detalles internos
  res.status(500).json({
    error: process.env.NODE_ENV === 'production'
      ? 'Error interno del servidor'
      : err.message
  });
};

// Wrapper para async route handlers
const asyncHandler = (fn) => (req, res, next) =>
  Promise.resolve(fn(req, res, next)).catch(next);

// Uso
app.get('/users/:id', asyncHandler(async (req, res) => {
  const user = await getUserById(req.params.id); // Si falla, va al errorHandler
  res.json(user);
}));

app.use(errorHandler); // Siempre al final
```

## Manejo de errores asíncronos no capturados

```javascript
// Errores de promesas no capturadas
process.on('unhandledRejection', (reason, promise) => {
  logger.error('Unhandled Rejection:', reason);
  // En producción, cerrar el proceso gracefully
  server.close(() => process.exit(1));
});

// Errores síncronos no capturados (bugs críticos)
process.on('uncaughtException', (error) => {
  logger.error('Uncaught Exception:', error);
  // Siempre cerrar el proceso: el estado puede estar corrompido
  server.close(() => process.exit(1));
});

// Cierre graceful con SIGTERM (señal de Kubernetes, PM2, etc.)
process.on('SIGTERM', () => {
  logger.info('SIGTERM recibido. Cerrando gracefully...');
  server.close(() => {
    logger.info('Servidor cerrado');
    process.exit(0);
  });
});
```

## Manejo en callbacks (estilo antiguo)

```javascript
// Patrón error-first callback (Node.js tradicional)
fs.readFile('file.txt', (err, data) => {
  if (err) {
    if (err.code === 'ENOENT') {
      console.error('Archivo no encontrado');
    } else {
      console.error('Error leyendo archivo:', err);
    }
    return;
  }
  console.log(data.toString());
});

// Convertir callbacks a promesas con util.promisify
const { promisify } = require('util');
const readFileAsync = promisify(fs.readFile);

// O usar la versión promisificada directamente
const { readFile } = require('fs/promises');
```

**Consejo de entrevista**: Las mejores prácticas son: crear clases de error personalizadas con `isOperational: true` para distinguir errores conocidos de bugs, usar un wrapper `asyncHandler` para evitar try/catch repetitivo en cada route, y siempre manejar `unhandledRejection` y `uncaughtException` para evitar que el proceso quede en estado inconsistente.
