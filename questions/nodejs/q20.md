Redis es una base de datos en memoria de alta velocidad usada habitualmente como caché. Almacenar resultados costosos (consultas a BD, llamadas a APIs externas) en Redis reduce la latencia y la carga del servidor.

## Conexión básica a Redis

```javascript
const { createClient } = require('redis');

const redis = createClient({
  url: process.env.REDIS_URL || 'redis://localhost:6379',
  socket: {
    reconnectStrategy: (retries) => {
      if (retries > 10) return new Error('No se puede conectar a Redis');
      return Math.min(retries * 100, 3000); // Backoff exponencial
    }
  }
});

redis.on('error', (err) => console.error('Redis error:', err));
redis.on('connect', () => console.log('Conectado a Redis'));

await redis.connect();
module.exports = redis;
```

## Caché de consultas a base de datos

```javascript
const redis = require('./redis');

const CACHE_TTL = 60 * 5; // 5 minutos en segundos

async function getUserById(id) {
  const cacheKey = `user:${id}`;

  // 1. Intentar obtener de caché
  const cached = await redis.get(cacheKey);
  if (cached) {
    console.log(`Cache HIT para ${cacheKey}`);
    return JSON.parse(cached);
  }

  // 2. Cache MISS: consultar base de datos
  console.log(`Cache MISS para ${cacheKey}`);
  const user = await User.findById(id);

  if (!user) return null;

  // 3. Guardar en caché con TTL
  await redis.setEx(cacheKey, CACHE_TTL, JSON.stringify(user));

  return user;
}

// Invalidar caché al actualizar
async function updateUser(id, data) {
  const user = await User.findByIdAndUpdate(id, data, { new: true });

  // Borrar la entrada de caché para que la próxima lectura sea fresca
  await redis.del(`user:${id}`);

  return user;
}
```

## Middleware de caché para Express

```javascript
const redis = require('./redis');

function cacheMiddleware(ttlSeconds = 300) {
  return async (req, res, next) => {
    // Solo cachear peticiones GET
    if (req.method !== 'GET') return next();

    const cacheKey = `api:${req.originalUrl}`;

    try {
      const cached = await redis.get(cacheKey);

      if (cached) {
        res.set('X-Cache', 'HIT');
        return res.json(JSON.parse(cached));
      }

      // Interceptar res.json para guardar en caché
      const originalJson = res.json.bind(res);
      res.json = async (data) => {
        if (res.statusCode === 200) {
          await redis.setEx(cacheKey, ttlSeconds, JSON.stringify(data));
        }
        res.set('X-Cache', 'MISS');
        return originalJson(data);
      };

      next();
    } catch (err) {
      // Si Redis falla, continuar sin caché
      console.error('Error de caché Redis:', err);
      next();
    }
  };
}

// Uso
app.get('/api/products', cacheMiddleware(300), async (req, res) => {
  const products = await Product.find().lean();
  res.json(products);
});
```

## Caché con patrón Cache-Aside y múltiples claves

```javascript
class CacheService {
  #redis;
  #defaultTTL;

  constructor(redisClient, defaultTTL = 300) {
    this.#redis = redisClient;
    this.#defaultTTL = defaultTTL;
  }

  async getOrSet(key, fetchFn, ttl = this.#defaultTTL) {
    const cached = await this.#redis.get(key);
    if (cached !== null) return JSON.parse(cached);

    const value = await fetchFn();
    if (value !== null && value !== undefined) {
      await this.#redis.setEx(key, ttl, JSON.stringify(value));
    }
    return value;
  }

  async invalidate(pattern) {
    // Borrar todas las claves que coincidan con el patrón
    const keys = await this.#redis.keys(pattern);
    if (keys.length > 0) {
      await this.#redis.del(keys);
    }
  }

  async invalidateByTags(tags) {
    for (const tag of tags) {
      const tagKey = `tag:${tag}`;
      const keys = await this.#redis.sMembers(tagKey);
      if (keys.length > 0) await this.#redis.del([...keys, tagKey]);
    }
  }
}

const cache = new CacheService(redis);

// Uso
const products = await cache.getOrSet(
  `products:category:${categoryId}`,
  () => Product.find({ category: categoryId }).lean(),
  600 // 10 minutos
);

// Invalidar al actualizar un producto
await cache.invalidate('products:*');
```

## Rate limiting con Redis (sliding window)

```javascript
async function checkRateLimit(identifier, maxRequests, windowSeconds) {
  const key = `rate:${identifier}`;
  const now = Date.now();
  const windowStart = now - windowSeconds * 1000;

  const pipeline = redis.multi();
  pipeline.zRemRangeByScore(key, 0, windowStart);   // Eliminar fuera de ventana
  pipeline.zCard(key);                               // Contar peticiones
  pipeline.zAdd(key, { score: now, value: `${now}` }); // Añadir actual
  pipeline.expire(key, windowSeconds);               // TTL del set

  const results = await pipeline.exec();
  const requestCount = results[1];

  return {
    allowed: requestCount < maxRequests,
    remaining: Math.max(0, maxRequests - requestCount - 1)
  };
}
```

**Consejo de entrevista**: El patrón Cache-Aside (lazy loading) es el más común: primero busca en caché, si no está la carga de la fuente y la guarda. Elige TTLs según la frecuencia de actualización de los datos. Para invalidar caché de forma eficiente, puedes usar un sistema de etiquetas (tags) que asocie claves de caché con entidades. En producción, configura Redis en modo cluster o con réplicas para alta disponibilidad.
