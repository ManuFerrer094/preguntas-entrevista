---
title: ¿Cómo implementar rate limiting en una API con Node.js?
difficulty: medium
tags: [nodejs, express, rate-limiting, security, api]
---

El rate limiting limita el número de peticiones que un cliente puede hacer en un periodo de tiempo, protegiendo la API contra abuso, ataques de fuerza bruta y scraping.

## Con express-rate-limit (solución más común)

```javascript
const express = require('express');
const rateLimit = require('express-rate-limit');

const app = express();

// Limiter global: 100 peticiones por 15 minutos
const globalLimiter = rateLimit({
  windowMs: 15 * 60 * 1000, // 15 minutos
  max: 100,
  standardHeaders: true,  // Añade cabeceras RateLimit-* en la respuesta
  legacyHeaders: false,
  message: {
    status: 429,
    error: 'Demasiadas peticiones. Inténtalo en 15 minutos.'
  },
  // Personalizar la clave (por defecto usa IP)
  keyGenerator: (req) => req.ip
});

// Limiter más estricto para autenticación
const authLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 5, // Solo 5 intentos de login
  message: {
    status: 429,
    error: 'Demasiados intentos de inicio de sesión. Inténtalo más tarde.'
  },
  skipSuccessfulRequests: true // No contar peticiones exitosas
});

// Aplicar limiters
app.use('/api/', globalLimiter);
app.use('/api/auth/login', authLimiter);
app.use('/api/auth/register', authLimiter);
```

## Con Redis para múltiples instancias (producción)

```javascript
const rateLimit = require('express-rate-limit');
const RedisStore = require('rate-limit-redis');
const { createClient } = require('redis');

const redisClient = createClient({ url: process.env.REDIS_URL });
await redisClient.connect();

const limiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 100,
  // Redis permite compartir contadores entre múltiples instancias de Node.js
  store: new RedisStore({
    sendCommand: (...args) => redisClient.sendCommand(args),
    prefix: 'rate-limit:'
  })
});
```

## Rate limiting manual con algoritmo de ventana deslizante

```javascript
class SlidingWindowRateLimiter {
  #requests = new Map(); // IP → [timestamps]
  #windowMs;
  #maxRequests;

  constructor(windowMs, maxRequests) {
    this.#windowMs = windowMs;
    this.#maxRequests = maxRequests;
  }

  isAllowed(identifier) {
    const now = Date.now();
    const windowStart = now - this.#windowMs;

    // Obtener o crear el historial de peticiones
    if (!this.#requests.has(identifier)) {
      this.#requests.set(identifier, []);
    }

    const timestamps = this.#requests.get(identifier);

    // Eliminar timestamps fuera de la ventana
    const validTimestamps = timestamps.filter(ts => ts > windowStart);
    this.#requests.set(identifier, validTimestamps);

    if (validTimestamps.length >= this.#maxRequests) {
      return {
        allowed: false,
        retryAfter: Math.ceil((validTimestamps[0] - windowStart) / 1000)
      };
    }

    validTimestamps.push(now);
    return { allowed: true, remaining: this.#maxRequests - validTimestamps.length };
  }
}

// Middleware con el limiter manual
const limiter = new SlidingWindowRateLimiter(60 * 1000, 30); // 30 req/min

app.use((req, res, next) => {
  const result = limiter.isAllowed(req.ip);

  res.set('X-RateLimit-Remaining', result.remaining ?? 0);

  if (!result.allowed) {
    res.set('Retry-After', result.retryAfter);
    return res.status(429).json({
      error: `Rate limit excedido. Espera ${result.retryAfter} segundos.`
    });
  }

  next();
});
```

## Rate limiting por usuario autenticado

```javascript
const userLimiter = rateLimit({
  windowMs: 60 * 1000, // 1 minuto
  max: 60,
  // Usuarios autenticados tienen mayor límite
  keyGenerator: (req) => {
    return req.user?.id || req.ip; // Usar ID de usuario si está autenticado
  },
  skip: (req) => {
    return req.user?.plan === 'premium'; // Saltar para usuarios premium
  }
});
```

**Consejo de entrevista**: Para producción con múltiples instancias, siempre usa Redis como store; los limiters en memoria solo funcionan con una sola instancia de Node.js. Aplica limits más estrictos en endpoints sensibles (login, registro, recuperación de contraseña). El status code correcto para rate limit es `429 Too Many Requests`, y la cabecera `Retry-After` informa al cliente cuándo puede reintentar.
