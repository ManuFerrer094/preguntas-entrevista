La paginación divide colecciones grandes en páginas más pequeñas, reduciendo el tiempo de respuesta y el consumo de memoria. Los dos enfoques principales son **offset/page** (más intuitivo) y **cursor** (más eficiente para grandes datasets).

## Paginación por offset/página

```javascript
// GET /api/products?page=2&limit=10
app.get('/api/products', async (req, res) => {
  const page = Math.max(1, parseInt(req.query.page) || 1);
  const limit = Math.min(100, Math.max(1, parseInt(req.query.limit) || 10));
  const skip = (page - 1) * limit;

  const [products, total] = await Promise.all([
    Product.find()
      .skip(skip)
      .limit(limit)
      .sort({ createdAt: -1 })
      .lean(),
    Product.countDocuments()
  ]);

  const totalPages = Math.ceil(total / limit);

  res.json({
    data: products,
    pagination: {
      page,
      limit,
      total,
      totalPages,
      hasNextPage: page < totalPages,
      hasPrevPage: page > 1,
      nextPage: page < totalPages ? page + 1 : null,
      prevPage: page > 1 ? page - 1 : null
    }
  });
});
```

## Paginación por cursor (Keyset pagination)

```javascript
// GET /api/products?cursor=<lastId>&limit=10
// Más eficiente que offset para grandes colecciones
app.get('/api/products/cursor', async (req, res) => {
  const limit = Math.min(100, parseInt(req.query.limit) || 10);
  const cursor = req.query.cursor; // ID del último elemento de la página anterior

  const query = cursor
    ? { _id: { $gt: cursor } } // Elementos después del cursor
    : {};

  const products = await Product.find(query)
    .sort({ _id: 1 })
    .limit(limit + 1) // Pedir uno más para saber si hay página siguiente
    .lean();

  const hasNextPage = products.length > limit;
  const data = hasNextPage ? products.slice(0, limit) : products;
  const nextCursor = hasNextPage ? data[data.length - 1]._id : null;

  res.json({
    data,
    pagination: {
      hasNextPage,
      nextCursor,
      limit
    }
  });
});
```

## Middleware de paginación reutilizable

```javascript
// middleware/paginate.js
function paginate(defaultLimit = 10, maxLimit = 100) {
  return (req, res, next) => {
    const page = Math.max(1, parseInt(req.query.page) || 1);
    const limit = Math.min(maxLimit, Math.max(1, parseInt(req.query.limit) || defaultLimit));

    req.pagination = {
      page,
      limit,
      skip: (page - 1) * limit
    };

    // Helper para construir la respuesta paginada
    res.paginate = (data, total) => {
      const totalPages = Math.ceil(total / limit);
      const baseUrl = `${req.protocol}://${req.get('host')}${req.path}`;

      res.json({
        data,
        pagination: {
          page,
          limit,
          total,
          totalPages,
          hasNextPage: page < totalPages,
          hasPrevPage: page > 1,
          links: {
            first: `${baseUrl}?page=1&limit=${limit}`,
            last: `${baseUrl}?page=${totalPages}&limit=${limit}`,
            prev: page > 1 ? `${baseUrl}?page=${page - 1}&limit=${limit}` : null,
            next: page < totalPages ? `${baseUrl}?page=${page + 1}&limit=${limit}` : null,
          }
        }
      });
    };

    next();
  };
}

// Uso
app.get('/api/users', paginate(), async (req, res) => {
  const { skip, limit } = req.pagination;

  const [users, total] = await Promise.all([
    User.find().skip(skip).limit(limit).lean(),
    User.countDocuments()
  ]);

  res.paginate(users, total);
});
```

## Paginación con SQL (PostgreSQL/MySQL)

```javascript
// Con pg (node-postgres)
async function paginateSQL(tableName, page, limit, orderBy = 'created_at DESC') {
  const offset = (page - 1) * limit;

  const [dataResult, countResult] = await Promise.all([
    pool.query(
      `SELECT * FROM ${tableName} ORDER BY ${orderBy} LIMIT $1 OFFSET $2`,
      [limit, offset]
    ),
    pool.query(`SELECT COUNT(*) as total FROM ${tableName}`)
  ]);

  const total = parseInt(countResult.rows[0].total);

  return {
    data: dataResult.rows,
    pagination: {
      page,
      limit,
      total,
      totalPages: Math.ceil(total / limit)
    }
  };
}

// Paginación eficiente con keyset en SQL (evita COUNT y OFFSET costosos)
async function keysetPaginateSQL(lastId, limit) {
  const query = lastId
    ? `SELECT * FROM products WHERE id > $1 ORDER BY id ASC LIMIT $2`
    : `SELECT * FROM products ORDER BY id ASC LIMIT $1`;

  const params = lastId ? [lastId, limit + 1] : [limit + 1];
  const result = await pool.query(query, params);

  const hasNextPage = result.rows.length > limit;
  const data = hasNextPage ? result.rows.slice(0, limit) : result.rows;

  return {
    data,
    nextCursor: hasNextPage ? data[data.length - 1].id : null,
    hasNextPage
  };
}
```

**Consejo de entrevista**: La paginación por offset es sencilla pero tiene un problema: el `SKIP/OFFSET` en bases de datos se vuelve lento conforme aumenta el número de páginas porque debe recorrer todos los registros previos. Para colecciones grandes (>100k registros) o feeds en tiempo real, la paginación por cursor es más eficiente y consistente (no se ve afectada por inserciones concurrentes). Limita siempre el tamaño máximo de página para evitar que un cliente descargue toda la base de datos en una petición.
