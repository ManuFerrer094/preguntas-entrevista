WebSocket es un protocolo de comunicación bidireccional en tiempo real sobre una única conexión TCP persistente. A diferencia de HTTP (donde el cliente siempre inicia la comunicación), WebSocket permite que tanto el servidor como el cliente envíen mensajes en cualquier momento. Es ideal para chats, notificaciones en tiempo real, juegos online y dashboards con datos en vivo.

## WebSocket nativo con la librería `ws`

```javascript
const WebSocket = require('ws');
const { createServer } = require('http');
const express = require('express');

const app = express();
const server = createServer(app);
const wss = new WebSocket.Server({ server });

// Almacenar clientes conectados
const clients = new Map();

wss.on('connection', (ws, req) => {
  const userId = req.url.split('?userId=')[1]; // /ws?userId=123
  clients.set(userId, ws);
  console.log(`Cliente ${userId} conectado. Total: ${clients.size}`);

  ws.on('message', (data) => {
    const message = JSON.parse(data.toString());
    console.log(`Mensaje de ${userId}:`, message);

    // Broadcast a todos los clientes conectados
    if (message.type === 'broadcast') {
      clients.forEach((clientWs, clientId) => {
        if (clientWs.readyState === WebSocket.OPEN) {
          clientWs.send(JSON.stringify({
            type: 'message',
            from: userId,
            text: message.text,
            timestamp: new Date().toISOString()
          }));
        }
      });
    }

    // Mensaje privado
    if (message.type === 'private') {
      const targetWs = clients.get(message.to);
      if (targetWs?.readyState === WebSocket.OPEN) {
        targetWs.send(JSON.stringify({ type: 'private', from: userId, text: message.text }));
      }
    }
  });

  ws.on('close', () => {
    clients.delete(userId);
    console.log(`Cliente ${userId} desconectado. Total: ${clients.size}`);
  });

  ws.on('error', (error) => {
    console.error(`Error en cliente ${userId}:`, error);
  });

  // Enviar mensaje de bienvenida
  ws.send(JSON.stringify({ type: 'welcome', message: '¡Conectado!' }));
});

server.listen(3000);
```

## Socket.IO — WebSockets con más funcionalidades

Socket.IO es la librería más popular para WebSockets en Node.js. Añade: rooms (salas), namespaces, reconnección automática, fallback a long-polling y broadcasting.

```javascript
// Servidor
const io = require('socket.io')(server, {
  cors: { origin: 'http://localhost:4200', methods: ['GET', 'POST'] }
});

// Middleware de autenticación
io.use((socket, next) => {
  const token = socket.handshake.auth.token;
  try {
    socket.user = jwt.verify(token, process.env.JWT_SECRET);
    next();
  } catch {
    next(new Error('Token inválido'));
  }
});

io.on('connection', (socket) => {
  console.log(`Usuario ${socket.user.id} conectado`);

  // Unirse a una sala (por ejemplo, una sala de chat)
  socket.on('join-room', (roomId) => {
    socket.join(roomId);
    socket.to(roomId).emit('user-joined', { userId: socket.user.id });
  });

  // Mensaje a una sala
  socket.on('chat-message', ({ roomId, text }) => {
    const message = {
      from: socket.user.id,
      text,
      timestamp: new Date().toISOString()
    };
    io.to(roomId).emit('chat-message', message); // A todos en la sala, incluyendo emisor
    // socket.to(roomId).emit(...)                // A todos EXCEPTO el emisor
  });

  // Notificación de "escribiendo"
  socket.on('typing', ({ roomId }) => {
    socket.to(roomId).emit('user-typing', { userId: socket.user.id });
  });

  socket.on('disconnect', () => {
    console.log(`Usuario ${socket.user.id} desconectado`);
  });
});
```

```javascript
// Cliente (JavaScript del navegador)
const socket = io('http://localhost:3000', {
  auth: { token: localStorage.getItem('token') }
});

socket.emit('join-room', 'room-123');

socket.on('chat-message', (message) => {
  displayMessage(message);
});

socket.on('user-joined', ({ userId }) => {
  showNotification(`${userId} se unió`);
});
```

**Consejo de entrevista**: La diferencia entre WebSocket y HTTP polling es importante: HTTP polling hace requests periódicas (ineficiente), Long-polling mantiene la conexión abierta hasta que hay datos (mejor pero complejo), y WebSocket mantiene una conexión persistente bidireccional (más eficiente para tiempo real). Socket.IO también maneja automáticamente la reconexión cuando se pierde la conexión.
