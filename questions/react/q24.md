---
title: ¿Cómo funciona el hook useImperativeHandle?
difficulty: hard
tags: [react, hooks, useImperativeHandle, forwardRef, ref]
---

`useImperativeHandle` permite personalizar el valor que se expone cuando el padre usa una `ref` sobre un componente hijo. En lugar de exponer el nodo DOM completo, el componente decide exactamente qué métodos o propiedades están disponibles.

## Problema que resuelve

```jsx
// Sin useImperativeHandle, forwardRef expone el DOM directamente
// El padre puede hacer CUALQUIER cosa con el DOM (poco encapsulado)
const InputBasic = forwardRef((props, ref) => (
  <input ref={ref} {...props} />
));

function Parent() {
  const ref = useRef(null);
  // Acceso completo al DOM del input — puede hacer ref.current.style, etc.
  return <InputBasic ref={ref} />;
}
```

## Uso básico de useImperativeHandle

```jsx
const FancyInput = forwardRef(function FancyInput(props, ref) {
  const inputRef = useRef(null);

  // Exponer solo los métodos que queremos dar al padre
  useImperativeHandle(ref, () => ({
    focus() {
      inputRef.current.focus();
    },
    clear() {
      inputRef.current.value = '';
      inputRef.current.focus();
    },
    getValue() {
      return inputRef.current.value;
    }
    // El padre NO puede acceder a inputRef.current directamente
    // ni a otras propiedades del DOM
  }));

  return <input ref={inputRef} className="fancy-input" {...props} />;
});

// Uso en el componente padre
function Form() {
  const inputRef = useRef(null);

  function handleSubmit() {
    const value = inputRef.current.getValue();
    if (!value) {
      inputRef.current.focus();
      return;
    }
    submitData(value);
    inputRef.current.clear();
  }

  return (
    <form onSubmit={handleSubmit}>
      <FancyInput ref={inputRef} placeholder="Introduce tu nombre" />
      <button type="submit">Enviar</button>
    </form>
  );
}
```

## Caso real: componente de video player

```jsx
const VideoPlayer = forwardRef(function VideoPlayer({ src, poster }, ref) {
  const videoRef = useRef(null);
  const [isPlaying, setIsPlaying] = useState(false);

  useImperativeHandle(ref, () => ({
    play() {
      videoRef.current.play();
      setIsPlaying(true);
    },
    pause() {
      videoRef.current.pause();
      setIsPlaying(false);
    },
    seekTo(timeSeconds) {
      videoRef.current.currentTime = timeSeconds;
    },
    get currentTime() {
      return videoRef.current?.currentTime ?? 0;
    },
    get duration() {
      return videoRef.current?.duration ?? 0;
    }
  }), []); // Array de dependencias vacío: la API no cambia

  return (
    <div className="video-container">
      <video ref={videoRef} src={src} poster={poster} />
      <div className="controls">
        <button onClick={() => isPlaying
          ? videoRef.current.pause()
          : videoRef.current.play()
        }>
          {isPlaying ? '⏸' : '▶'}
        </button>
      </div>
    </div>
  );
});

// El componente padre controla el video programáticamente
function Presentation({ slides }) {
  const videoRef = useRef(null);
  const [currentSlide, setCurrentSlide] = useState(0);

  function nextSlide() {
    videoRef.current?.pause();
    setCurrentSlide(s => s + 1);
  }

  function replayFromStart() {
    videoRef.current?.seekTo(0);
    videoRef.current?.play();
  }

  return (
    <div>
      <VideoPlayer
        ref={videoRef}
        src={slides[currentSlide].videoUrl}
      />
      <button onClick={nextSlide}>Siguiente diapositiva</button>
      <button onClick={replayFromStart}>Repetir desde el inicio</button>
    </div>
  );
}
```

## useImperativeHandle con dependencias

```jsx
const SearchInput = forwardRef(function SearchInput({ onSearch }, ref) {
  const inputRef = useRef(null);
  const [query, setQuery] = useState('');

  // Las dependencias controlan cuándo se recalcula el objeto expuesto
  useImperativeHandle(ref, () => ({
    focus: () => inputRef.current?.focus(),
    clear: () => {
      setQuery('');
      onSearch(''); // onSearch está en las dependencias
    },
    get value() {
      return query; // query está en las dependencias
    }
  }), [query, onSearch]); // Recalcular cuando query u onSearch cambien

  return (
    <input
      ref={inputRef}
      value={query}
      onChange={e => {
        setQuery(e.target.value);
        onSearch(e.target.value);
      }}
    />
  );
});
```

## Cuándo usar useImperativeHandle

```jsx
// ✅ Componentes con comportamiento imperativo: media players, editores, mapas
// ✅ Exponer una API pública controlada al componente padre
// ✅ Cuando el padre necesita triggear acciones (focus, scroll, play)

// ❌ Para pasar datos → usa props y callbacks
// ❌ Para estado compartido → usa Context o state lifting
// ❌ Para la mayoría de casos → la comunicación declarativa es preferible
```

**Consejo de entrevista**: `useImperativeHandle` va siempre junto con `forwardRef`. Su uso debe ser excepcional: para componentes de UI con comportamiento imperativo inherente (players, editores de texto enriquecido, mapas, componentes de terceros). Si te encuentras usándolo frecuentemente, probablemente hay una solución más declarativa con props y callbacks. Mencionar esto en la entrevista muestra comprensión de cuándo es apropiado el paradigma imperativo vs declarativo.
