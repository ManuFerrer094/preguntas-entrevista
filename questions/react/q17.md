---
title: ¿Cómo puedes cancelar una petición a una API en useEffect correctamente?
difficulty: medium
tags: [react, useEffect, abort-controller, fetch, cleanup]
---

Cuando un componente se desmonta mientras hay una petición en curso, intentar actualizar el estado genera un warning de memory leak. La solución es cancelar la petición en la función de cleanup de `useEffect` usando `AbortController`.

## El problema sin cancelación

```jsx
function UserProfile({ userId }) {
  const [user, setUser] = useState(null);

  useEffect(() => {
    // ❌ Si el componente se desmonta antes de que termine la petición,
    // setUser se llama en un componente desmontado → warning + posible bug
    fetch(`/api/users/${userId}`)
      .then(res => res.json())
      .then(data => setUser(data));
  }, [userId]);

  return user ? <div>{user.name}</div> : <p>Cargando...</p>;
}
```

## Solución con AbortController

```jsx
function UserProfile({ userId }) {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    const controller = new AbortController();

    async function fetchUser() {
      try {
        setLoading(true);
        setError(null);

        const response = await fetch(`/api/users/${userId}`, {
          signal: controller.signal
        });

        if (!response.ok) {
          throw new Error(`HTTP ${response.status}`);
        }

        const data = await response.json();
        setUser(data);
      } catch (err) {
        if (err.name === 'AbortError') {
          // La petición fue cancelada intencionalmente — no es un error real
          return;
        }
        setError(err.message);
      } finally {
        setLoading(false);
      }
    }

    fetchUser();

    // Función de cleanup: se ejecuta cuando cambia userId o el componente se desmonta
    return () => controller.abort();
  }, [userId]);

  if (loading) return <Spinner />;
  if (error) return <ErrorMessage message={error} />;
  return user ? <div>{user.name}</div> : null;
}
```

## Con múltiples peticiones en paralelo

```jsx
function Dashboard({ userId }) {
  const [data, setData] = useState(null);

  useEffect(() => {
    const controller = new AbortController();
    const { signal } = controller;

    async function loadDashboard() {
      try {
        // Ambas peticiones comparten el mismo signal
        const [user, posts] = await Promise.all([
          fetch(`/api/users/${userId}`, { signal }).then(r => r.json()),
          fetch(`/api/posts?userId=${userId}`, { signal }).then(r => r.json()),
        ]);
        setData({ user, posts });
      } catch (err) {
        if (err.name !== 'AbortError') {
          console.error('Error cargando dashboard:', err);
        }
      }
    }

    loadDashboard();
    return () => controller.abort();
  }, [userId]);

  return data ? <DashboardView data={data} /> : <Skeleton />;
}
```

## Custom hook reutilizable

```jsx
function useFetch(url) {
  const [state, setState] = useState({
    data: null,
    loading: true,
    error: null
  });

  useEffect(() => {
    if (!url) return;

    const controller = new AbortController();

    setState(prev => ({ ...prev, loading: true, error: null }));

    fetch(url, { signal: controller.signal })
      .then(res => {
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        return res.json();
      })
      .then(data => setState({ data, loading: false, error: null }))
      .catch(err => {
        if (err.name === 'AbortError') return;
        setState({ data: null, loading: false, error: err.message });
      });

    return () => controller.abort();
  }, [url]);

  return state;
}

// Uso
function ProductList({ categoryId }) {
  const { data: products, loading, error } = useFetch(
    `/api/products?category=${categoryId}`
  );

  if (loading) return <Spinner />;
  if (error) return <p>Error: {error}</p>;
  return <ul>{products?.map(p => <li key={p.id}>{p.name}</li>)}</ul>;
}
```

## Con React Query (evita el boilerplate)

```jsx
import { useQuery } from '@tanstack/react-query';

// React Query gestiona la cancelación automáticamente con AbortController
function UserProfile({ userId }) {
  const { data: user, isLoading, error } = useQuery({
    queryKey: ['user', userId],
    queryFn: async ({ signal }) => {
      const response = await fetch(`/api/users/${userId}`, { signal });
      if (!response.ok) throw new Error(`HTTP ${response.status}`);
      return response.json();
    }
  });

  if (isLoading) return <Spinner />;
  if (error) return <ErrorMessage message={error.message} />;
  return <div>{user?.name}</div>;
}
```

**Consejo de entrevista**: El `AbortController` se crea dentro del `useEffect`, no fuera. La función de cleanup llama a `controller.abort()`. Siempre verifica `err.name === 'AbortError'` para distinguir una cancelación intencional de un error real. En proyectos con muchas peticiones asíncronas, considera usar React Query o SWR, que gestionan cancelación, caché y revalidación automáticamente.
