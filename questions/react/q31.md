El patrón **Compound Components** permite crear componentes complejos que trabajan juntos compartiendo estado implícito mediante Context, dando al consumidor control total sobre la estructura del markup sin necesidad de pasar decenas de props.

## El problema que resuelve

```jsx
// ❌ Prop drilling: muchos props, poca flexibilidad
<Select
  opciones={opciones}
  valorSeleccionado={valor}
  onChange={setValor}
  placeholder="Selecciona..."
  mostrarIcono={true}
  iconoFiltro={true}
  mostrarContador={true}
  renderOpcion={(op) => <CustomOpcion op={op} />}
  renderGrupo={(grupo) => <h3>{grupo.nombre}</h3>}
  // 20 props más...
/>

// ✅ Compound Components: estructura declarativa
<Select valor={valor} onChange={setValor}>
  <Select.Trigger placeholder="Selecciona..." />
  <Select.Menu>
    <Select.Search />
    {opciones.map(op => (
      <Select.Option key={op.id} valor={op.valor}>
        <CustomOpcion op={op} />
      </Select.Option>
    ))}
  </Select.Menu>
</Select>
```

## Implementación con Context

```jsx
import { createContext, useContext, useState, useCallback } from 'react';

// Contexto compartido entre todos los sub-componentes
const SelectContext = createContext(null);

function useSelectContext() {
  const ctx = useContext(SelectContext);
  if (!ctx) throw new Error('useSelectContext debe usarse dentro de <Select>');
  return ctx;
}

// Componente raíz: gestiona el estado compartido
function Select({ children, valor, onChange }) {
  const [abierto, setAbierto] = useState(false);

  const seleccionar = useCallback((nuevoValor) => {
    onChange(nuevoValor);
    setAbierto(false);
  }, [onChange]);

  return (
    <SelectContext.Provider value={{ abierto, setAbierto, valor, seleccionar }}>
      <div className="select-container">
        {children}
      </div>
    </SelectContext.Provider>
  );
}

// Sub-componente Trigger
Select.Trigger = function SelectTrigger({ placeholder }) {
  const { abierto, setAbierto, valor } = useSelectContext();

  return (
    <button
      className={`select-trigger ${abierto ? 'abierto' : ''}`}
      onClick={() => setAbierto(!abierto)}
      aria-expanded={abierto}
    >
      {valor ?? placeholder ?? 'Selecciona...'}
      <span>{abierto ? '▲' : '▼'}</span>
    </button>
  );
};

// Sub-componente Menu
Select.Menu = function SelectMenu({ children }) {
  const { abierto } = useSelectContext();

  if (!abierto) return null;

  return (
    <ul className="select-menu" role="listbox">
      {children}
    </ul>
  );
};

// Sub-componente Option
Select.Option = function SelectOption({ children, valor: opcionValor }) {
  const { seleccionar, valor } = useSelectContext();
  const estaSeleccionado = valor === opcionValor;

  return (
    <li
      role="option"
      aria-selected={estaSeleccionado}
      className={`select-option ${estaSeleccionado ? 'seleccionado' : ''}`}
      onClick={() => seleccionar(opcionValor)}
    >
      {children}
    </li>
  );
};
```

## Ejemplo real: Acordeón expandible

```jsx
const AcordeonContext = createContext(null);

function Acordeon({ children, multiple = false }) {
  const [abiertos, setAbiertos] = useState(new Set());

  const toggle = (id) => {
    setAbiertos(prev => {
      const nuevos = multiple ? new Set(prev) : new Set();
      if (prev.has(id)) {
        nuevos.delete(id);
      } else {
        nuevos.add(id);
      }
      return nuevos;
    });
  };

  return (
    <AcordeonContext.Provider value={{ abiertos, toggle }}>
      <div className="acordeon">{children}</div>
    </AcordeonContext.Provider>
  );
}

Acordeon.Item = function AcordeonItem({ children, id }) {
  const { abiertos, toggle } = useContext(AcordeonContext);
  const estaAbierto = abiertos.has(id);

  return (
    <div className={`acordeon-item ${estaAbierto ? 'abierto' : ''}`}>
      {typeof children === 'function'
        ? children({ estaAbierto, toggle: () => toggle(id) })
        : children
      }
    </div>
  );
};

// Uso flexible:
<Acordeon multiple>
  <Acordeon.Item id="faq1">
    {({ estaAbierto, toggle }) => (
      <>
        <button onClick={toggle}>
          ¿Qué es React? {estaAbierto ? '−' : '+'}
        </button>
        {estaAbierto && <p>React es una biblioteca de UI...</p>}
      </>
    )}
  </Acordeon.Item>
  <Acordeon.Item id="faq2">
    {({ estaAbierto, toggle }) => (
      <>
        <button onClick={toggle}>
          ¿Qué son los hooks? {estaAbierto ? '−' : '+'}
        </button>
        {estaAbierto && <p>Los hooks son funciones especiales de React...</p>}
      </>
    )}
  </Acordeon.Item>
</Acordeon>
```

## Cuándo usar Compound Components

```
✅ Usar cuando:
- El componente tiene múltiples partes relacionadas (Trigger, Content, Footer)
- El consumidor necesita control sobre el markup (accesibilidad, estilos)
- Hay estado compartido entre las partes
- Quieres evitar prop drilling en componentes complejos

❌ Evitar cuando:
- El componente es simple (1-2 props)
- Los sub-componentes siempre aparecen juntos con el mismo orden
- No hay flexibilidad de layout necesaria
```

**Consejo de entrevista**: El patrón Compound Components aparece en las mejores librerías de UI: Radix UI, Headless UI, Reach UI. La clave es el uso de `Context` para compartir estado sin prop drilling y el error al usar fuera del contexto (`throw new Error`). Menciona también la variante con `React.Children.map` para proyectos sin Context (más limitada), y la variante con render props para máxima flexibilidad.
