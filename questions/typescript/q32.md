El **patrón Factory** (Fábrica) es un patrón creacional que proporciona una interfaz para crear objetos sin especificar su clase concreta. En TypeScript, se pueden implementar de forma tipada y segura mediante funciones de fábrica, clases abstractas o el patrón Abstract Factory.

## Factory Function (más sencillo)

```typescript
interface Notificacion {
  tipo: string;
  mensaje: string;
  enviar(): Promise<void>;
}

class NotificacionEmail implements Notificacion {
  tipo = 'email';
  constructor(
    public mensaje: string,
    private destinatario: string
  ) {}

  async enviar(): Promise<void> {
    await emailService.send(this.destinatario, this.mensaje);
    console.log(`Email enviado a ${this.destinatario}: ${this.mensaje}`);
  }
}

class NotificacionSMS implements Notificacion {
  tipo = 'sms';
  constructor(
    public mensaje: string,
    private telefono: string
  ) {}

  async enviar(): Promise<void> {
    await smsService.send(this.telefono, this.mensaje);
    console.log(`SMS enviado a ${this.telefono}: ${this.mensaje}`);
  }
}

class NotificacionPush implements Notificacion {
  tipo = 'push';
  constructor(
    public mensaje: string,
    private tokenDispositivo: string
  ) {}

  async enviar(): Promise<void> {
    await pushService.send(this.tokenDispositivo, this.mensaje);
  }
}

// Factory Function: crea la notificación según el tipo
type TipoNotificacion = 'email' | 'sms' | 'push';

interface OpcionesNotificacion {
  tipo: TipoNotificacion;
  mensaje: string;
  destinatario: string;
}

function crearNotificacion(opciones: OpcionesNotificacion): Notificacion {
  switch (opciones.tipo) {
    case 'email':
      return new NotificacionEmail(opciones.mensaje, opciones.destinatario);
    case 'sms':
      return new NotificacionSMS(opciones.mensaje, opciones.destinatario);
    case 'push':
      return new NotificacionPush(opciones.mensaje, opciones.destinatario);
    default:
      const _exhaustivo: never = opciones.tipo;
      throw new Error(`Tipo de notificación no soportado: ${_exhaustivo}`);
  }
}

// Uso:
const notificacion = crearNotificacion({
  tipo: 'email',
  mensaje: 'Tu pedido ha sido enviado',
  destinatario: 'usuario@ejemplo.com'
});
await notificacion.enviar();
```

## Factory Method (con clase base abstracta)

```typescript
abstract class SerializadorBase<T> {
  // Template Method: usa el Factory Method internamente
  serializar(datos: T[]): string {
    const preparados = datos.map(d => this.transformar(d));
    return this.formatear(preparados);
  }

  // Factory Method: las subclases deciden cómo transformar
  protected abstract transformar(dato: T): Record<string, unknown>;

  // Factory Method: las subclases deciden el formato de salida
  protected abstract formatear(datos: Record<string, unknown>[]): string;
}

class SerializadorJSON<T extends object> extends SerializadorBase<T> {
  protected transformar(dato: T): Record<string, unknown> {
    return dato as Record<string, unknown>;
  }

  protected formatear(datos: Record<string, unknown>[]): string {
    return JSON.stringify(datos, null, 2);
  }
}

class SerializadorCSV extends SerializadorBase<Record<string, string | number>> {
  protected transformar(dato: Record<string, string | number>) {
    return dato;
  }

  protected formatear(datos: Record<string, unknown>[]): string {
    if (!datos.length) return '';
    const cabeceras = Object.keys(datos[0]).join(',');
    const filas = datos.map(d => Object.values(d).join(','));
    return [cabeceras, ...filas].join('\n');
  }
}
```

## Abstract Factory (fábrica de fábricas)

```typescript
// Abstract Factory: crea familias de objetos relacionados
interface BotonUI { render(): string; onClick(): void; }
interface InputUI { render(): string; getValue(): string; }
interface ModalUI { render(): string; close(): void; }

// Fábrica abstracta
interface FabricaUI {
  crearBoton(texto: string): BotonUI;
  crearInput(placeholder: string): InputUI;
  crearModal(titulo: string, contenido: string): ModalUI;
}

// Familia Material Design
class BotonMaterial implements BotonUI {
  constructor(private texto: string) {}
  render() { return `<button class="md-button">${this.texto}</button>`; }
  onClick() { console.log(`Material click: ${this.texto}`); }
}

class FabricaMaterial implements FabricaUI {
  crearBoton(texto: string) { return new BotonMaterial(texto); }
  crearInput(placeholder: string) { return new InputMaterial(placeholder); }
  crearModal(titulo: string, contenido: string) { return new ModalMaterial(titulo, contenido); }
}

// Familia Bootstrap
class FabricaBootstrap implements FabricaUI {
  crearBoton(texto: string) { return new BotonBootstrap(texto); }
  crearInput(placeholder: string) { return new InputBootstrap(placeholder); }
  crearModal(titulo: string, contenido: string) { return new ModalBootstrap(titulo, contenido); }
}

// El cliente trabaja con la fábrica abstracta
function crearFormulario(fabrica: FabricaUI, config: FormConfig) {
  const boton = fabrica.crearBoton(config.textoBoton);
  const inputs = config.campos.map(c => fabrica.crearInput(c.placeholder));
  const modal = fabrica.crearModal(config.titulo, config.descripcion);

  // El formulario funciona igual independientemente del tema visual
  return { boton, inputs, modal };
}

// Cambiar el tema: solo cambia la fábrica
const formularioMaterial = crearFormulario(new FabricaMaterial(), config);
const formularioBootstrap = crearFormulario(new FabricaBootstrap(), config);
```

## Factory con registro dinámico

```typescript
// Factory con registro: permite añadir tipos sin modificar el factory
class RepositorioFactory {
  private static registro = new Map<string, new (...args: any[]) => IRepositorio>();

  static registrar(tipo: string, clase: new (...args: any[]) => IRepositorio) {
    this.registro.set(tipo, clase);
  }

  static crear(tipo: string, ...args: any[]): IRepositorio {
    const Clase = this.registro.get(tipo);
    if (!Clase) throw new Error(`Repositorio no registrado: ${tipo}`);
    return new Clase(...args);
  }
}

// Registro de implementaciones
RepositorioFactory.registrar('mysql', RepositorioMySQL);
RepositorioFactory.registrar('mongo', RepositorioMongo);
RepositorioFactory.registrar('memoria', RepositorioMemoria);

// Uso:
const repo = RepositorioFactory.crear(process.env.DB_TYPE ?? 'memoria');
```

**Consejo de entrevista**: El patrón Factory es más valioso cuando el tipo exacto del objeto a crear se conoce en runtime (configuración, variables de entorno). La versión con `switch` exhaustivo y el tipo `never` como default asegura que si se añade un nuevo caso, TypeScript emitirá un error de compilación si no se maneja. La Abstract Factory es ideal para Design Systems donde necesitas intercambiar familias completas de componentes (tema claro/oscuro, plataforma nativa/web).
