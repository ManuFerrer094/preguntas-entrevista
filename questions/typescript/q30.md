Los **decoradores de acceso** (*Accessor Decorators*) introducidos en TypeScript 5 son un tipo especial de decorador que puede aplicarse a los accessors `get`/`set` de una clase, permitiendo añadir lógica de observación, validación, transformación o logging de forma declarativa.

> **Nota**: TypeScript 5 implementa decoradores según la propuesta Stage 3 de ECMAScript, diferente a los decoradores experimentales de versiones anteriores. Requiere `"experimentalDecorators": false` (o simplemente no configurarlo) para usar los nuevos.

## Accessor Decorator básico

```typescript
// Un Accessor Decorator recibe: target (clase o prototipo), context (metadatos)
// Para auto-accessor (el más común en TS5)

function observable<This, Value>(
  target: ClassAccessorDecoratorTarget<This, Value>,
  context: ClassAccessorDecoratorContext<This, Value>
): ClassAccessorDecoratorResult<This, Value> {
  return {
    // Sobreescribir el getter
    get(this: This): Value {
      return target.get.call(this);
    },
    // Sobreescribir el setter con logging
    set(this: This, valor: Value): void {
      const valorAnterior = target.get.call(this);
      target.set.call(this, valor);
      console.log(`[${String(context.name)}] ${String(valorAnterior)} → ${String(valor)}`);
    },
    // Inicialización
    init(this: This, valorInicial: Value): Value {
      return valorInicial;
    }
  };
}

class Usuario {
  // `accessor` es una nueva keyword de TS5 que genera get/set automáticamente
  @observable
  accessor nombre = 'Ana';

  @observable
  accessor email = 'ana@ejemplo.com';
}

const usuario = new Usuario();
usuario.nombre = 'Carlos';  // Log: "[nombre] Ana → Carlos"
usuario.email = 'carlos@ejemplo.com';  // Log: "[email] ana@ejemplo.com → carlos@ejemplo.com"
```

## Decorador de validación

```typescript
function validarRango(min: number, max: number) {
  return function<This>(
    target: ClassAccessorDecoratorTarget<This, number>,
    context: ClassAccessorDecoratorContext<This, number>
  ): ClassAccessorDecoratorResult<This, number> {
    return {
      set(this: This, valor: number) {
        if (valor < min || valor > max) {
          throw new RangeError(
            `${String(context.name)} debe estar entre ${min} y ${max}, recibido: ${valor}`
          );
        }
        target.set.call(this, valor);
      }
    };
  };
}

function noVacio(
  target: ClassAccessorDecoratorTarget<unknown, string>,
  context: ClassAccessorDecoratorContext<unknown, string>
): ClassAccessorDecoratorResult<unknown, string> {
  return {
    set(this: unknown, valor: string) {
      if (!valor.trim()) {
        throw new Error(`${String(context.name)} no puede estar vacío`);
      }
      target.set.call(this, valor.trim());
    }
  };
}

class Producto {
  @noVacio
  accessor nombre = '';

  @validarRango(0, 99999)
  accessor precio = 0;

  @validarRango(0, 1000)
  accessor stock = 0;
}

const producto = new Producto();
producto.nombre = '  ';         // ❌ Error: nombre no puede estar vacío
producto.precio = -50;          // ❌ RangeError: precio debe estar entre 0 y 99999
producto.stock = 5;             // ✅ OK
```

## Decorador de caché (memoización de accessor)

```typescript
function memoizar<This, Value>(
  target: ClassAccessorDecoratorTarget<This, Value>,
  context: ClassAccessorDecoratorContext<This, Value>
): ClassAccessorDecoratorResult<This, Value> {
  let cachedValue: Value;
  let calculado = false;

  return {
    get(this: This): Value {
      if (!calculado) {
        cachedValue = target.get.call(this);
        calculado = true;
        console.log(`[${String(context.name)}] calculado por primera vez`);
      }
      return cachedValue;
    },
    set(this: This, valor: Value) {
      calculado = false;  // Invalidar caché al setear
      target.set.call(this, valor);
    }
  };
}
```

## Decorador de clase, método y field (TypeScript 5)

```typescript
// Decorador de clase
function sellable<T extends { new(...args: any[]): {} }>(
  target: T,
  context: ClassDecoratorContext
) {
  return class extends target {
    enVenta = true;
    vendidoEl?: Date;
  };
}

// Decorador de método
function deprecado(mensaje: string) {
  return function(target: Function, context: ClassMethodDecoratorContext) {
    return function(this: unknown, ...args: unknown[]) {
      console.warn(`⚠️ ${String(context.name)} está deprecado: ${mensaje}`);
      return target.apply(this, args);
    };
  };
}

@sellable
class Articulo {
  @deprecado('Usa obtenerNombreCompleto() en su lugar')
  getNombre() {
    return this.nombre;
  }

  accessor nombre = '';
}
```

## Diferencia: decoradores legacy vs Stage 3

```typescript
// ❌ Decoradores legacy (experimentalDecorators: true) — evitar en proyectos nuevos
function LegacyDecorator(target: any, key: string, descriptor: PropertyDescriptor) {
  // ...
}

// ✅ Decoradores Stage 3 (TypeScript 5+, sin experimentalDecorators)
function ModernDecorator<This, Value>(
  target: ClassAccessorDecoratorTarget<This, Value>,
  context: ClassAccessorDecoratorContext<This, Value>
) {
  // context incluye metadatos ricos: name, kind, static, private, addInitializer...
  context.addInitializer(function(this: This) {
    console.log(`Inicializando ${String(context.name)}`);
  });
}
```

**Consejo de entrevista**: Los decoradores Stage 3 de TypeScript 5 son la versión estándar y futura de los decoradores en JavaScript. La nueva keyword `accessor` genera automáticamente getter, setter y un campo privado de respaldo, eliminando la necesidad de definirlos manualmente. El `context.addInitializer()` permite ejecutar código durante la inicialización de la clase. Menciona que Angular sigue usando decoradores legacy (`experimentalDecorators: true`) pero que el equipo está migrando hacia los Stage 3 en versiones futuras.
