Usar TypeScript con React añade tipado estático a componentes, props, hooks y eventos, previniendo errores en tiempo de compilación y mejorando la experiencia de desarrollo con autocompletado.

## Tipar componentes funcionales

```tsx
import { FC, ReactNode, PropsWithChildren } from 'react';

// Forma 1: tipos inline (la más usada en código moderno)
interface TarjetaProps {
  titulo: string;
  descripcion?: string;           // Prop opcional
  variante: 'primaria' | 'secundaria' | 'peligro';
  onClick?: () => void;
  children: ReactNode;             // Cualquier contenido renderizable
}

function Tarjeta({ titulo, descripcion, variante, onClick, children }: TarjetaProps) {
  return (
    <div className={`tarjeta tarjeta-${variante}`} onClick={onClick}>
      <h3>{titulo}</h3>
      {descripcion && <p>{descripcion}</p>}
      {children}
    </div>
  );
}

// Forma 2: PropsWithChildren (añade children automáticamente)
type ContenedorProps = PropsWithChildren<{
  maxWidth?: number;
  centrado?: boolean;
}>;

function Contenedor({ children, maxWidth = 1200, centrado = false }: ContenedorProps) {
  return (
    <div style={{ maxWidth, margin: centrado ? '0 auto' : undefined }}>
      {children}
    </div>
  );
}
```

## Tipar useState y useReducer

```tsx
import { useState, useReducer } from 'react';

interface Usuario {
  id: number;
  nombre: string;
  email: string;
  rol: 'admin' | 'usuario' | 'moderador';
}

function PanelUsuario() {
  // Estado tipado
  const [usuario, setUsuario] = useState<Usuario | null>(null);
  const [cargando, setCargando] = useState<boolean>(false);
  const [errores, setErrores] = useState<string[]>([]);

  // Con valor inicial tipado automáticamente (sin genérico necesario)
  const [contador, setContador] = useState(0);  // infiere number
  const [nombre, setNombre] = useState('');       // infiere string
}

// useReducer tipado
type Accion =
  | { type: 'CARGAR' }
  | { type: 'CARGAR_EXITO'; payload: Usuario[] }
  | { type: 'CARGAR_ERROR'; error: string }
  | { type: 'ELIMINAR'; id: number };

interface Estado {
  usuarios: Usuario[];
  cargando: boolean;
  error: string | null;
}

function reducer(estado: Estado, accion: Accion): Estado {
  switch (accion.type) {
    case 'CARGAR':
      return { ...estado, cargando: true, error: null };
    case 'CARGAR_EXITO':
      return { ...estado, cargando: false, usuarios: accion.payload };
    case 'CARGAR_ERROR':
      return { ...estado, cargando: false, error: accion.error };
    case 'ELIMINAR':
      return { ...estado, usuarios: estado.usuarios.filter(u => u.id !== accion.id) };
    default:
      return estado;
  }
}
```

## Tipar eventos de React

```tsx
import { ChangeEvent, FormEvent, MouseEvent, KeyboardEvent } from 'react';

function Formulario() {
  // Eventos tipados con el elemento correcto
  function handleChange(e: ChangeEvent<HTMLInputElement>) {
    console.log(e.target.value);     // string
    console.log(e.target.checked);   // boolean (checkbox)
  }

  function handleSelectChange(e: ChangeEvent<HTMLSelectElement>) {
    console.log(e.target.value);
  }

  function handleTextAreaChange(e: ChangeEvent<HTMLTextAreaElement>) {
    console.log(e.target.value);
  }

  function handleSubmit(e: FormEvent<HTMLFormElement>) {
    e.preventDefault();
    const formData = new FormData(e.currentTarget);
    console.log(formData.get('email'));
  }

  function handleClick(e: MouseEvent<HTMLButtonElement>) {
    console.log(e.clientX, e.clientY);  // Coordenadas del click
  }

  function handleKeyDown(e: KeyboardEvent<HTMLInputElement>) {
    if (e.key === 'Enter') { /* ... */ }
  }

  return (
    <form onSubmit={handleSubmit}>
      <input onChange={handleChange} onKeyDown={handleKeyDown} />
      <button onClick={handleClick}>Enviar</button>
    </form>
  );
}
```

## Custom Hooks tipados

```tsx
import { useState, useEffect, useCallback } from 'react';

// Hook genérico para fetch
interface EstadoFetch<T> {
  datos: T | null;
  cargando: boolean;
  error: Error | null;
  recargar: () => void;
}

function useFetch<T>(url: string): EstadoFetch<T> {
  const [datos, setDatos] = useState<T | null>(null);
  const [cargando, setCargando] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  const [contador, setContador] = useState(0);  // Para forzar recargas

  useEffect(() => {
    const controller = new AbortController();
    setCargando(true);

    fetch(url, { signal: controller.signal })
      .then(res => {
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        return res.json() as Promise<T>;
      })
      .then(data => {
        setDatos(data);
        setError(null);
      })
      .catch(err => {
        if (err.name !== 'AbortError') setError(err);
      })
      .finally(() => setCargando(false));

    return () => controller.abort();
  }, [url, contador]);

  const recargar = useCallback(() => setContador(c => c + 1), []);

  return { datos, cargando, error, recargar };
}

// Uso tipado:
const { datos: usuarios, cargando } = useFetch<Usuario[]>('/api/usuarios');
// datos es Usuario[] | null (no any)
```

## Tipar Context API

```tsx
import { createContext, useContext, ReactNode, useState } from 'react';

interface TemaContextType {
  tema: 'claro' | 'oscuro';
  toggleTema: () => void;
}

// createContext con valor inicial null y tipo explícito
const TemaContext = createContext<TemaContextType | null>(null);

export function TemaProvider({ children }: { children: ReactNode }) {
  const [tema, setTema] = useState<'claro' | 'oscuro'>('claro');
  const toggleTema = () => setTema(t => t === 'claro' ? 'oscuro' : 'claro');

  return (
    <TemaContext.Provider value={{ tema, toggleTema }}>
      {children}
    </TemaContext.Provider>
  );
}

// Custom hook con validación del contexto
export function useTema(): TemaContextType {
  const ctx = useContext(TemaContext);
  if (!ctx) throw new Error('useTema debe usarse dentro de TemaProvider');
  return ctx;
}
```

## Tipar refs

```tsx
import { useRef, forwardRef } from 'react';

function MedirComponente() {
  // Ref al elemento DOM
  const divRef = useRef<HTMLDivElement>(null);
  const inputRef = useRef<HTMLInputElement>(null);

  const medir = () => {
    if (divRef.current) {
      console.log(divRef.current.getBoundingClientRect());
    }
    inputRef.current?.focus();  // Opcional chaining seguro
  };
}

// forwardRef tipado
const InputRef = forwardRef<HTMLInputElement, { placeholder?: string }>(
  ({ placeholder }, ref) => (
    <input ref={ref} placeholder={placeholder} />
  )
);
```

**Consejo de entrevista**: Los tipos más importantes de React en TypeScript son: `ReactNode` (cualquier cosa renderizable), `ReactElement` (elemento JSX específico), `FC<Props>` (componente funcional), y los tipos de eventos (`ChangeEvent`, `FormEvent`, `MouseEvent`). El patrón de usar generics en hooks (`useFetch<T>`) elimina los `any` y propaga el tipado correctamente. Evita `FC<>` para componentes que no necesitan `children` —es más preciso definir el tipo directamente en los parámetros.
