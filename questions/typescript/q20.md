El patrón Builder construye objetos complejos paso a paso con una interfaz fluida (method chaining). TypeScript lo potencia con tipos que verifican en compilación que todos los campos requeridos han sido proporcionados antes de llamar a `build()`.

## Builder básico

```typescript
class QueryBuilder {
  private table: string = '';
  private conditions: string[] = [];
  private columns: string[] = ['*'];
  private limitValue?: number;
  private offsetValue?: number;
  private orderByClause?: string;

  from(table: string): this {
    this.table = table;
    return this; // Retorna this para encadenamiento
  }

  select(...columns: string[]): this {
    this.columns = columns;
    return this;
  }

  where(condition: string): this {
    this.conditions.push(condition);
    return this;
  }

  limit(n: number): this {
    this.limitValue = n;
    return this;
  }

  offset(n: number): this {
    this.offsetValue = n;
    return this;
  }

  orderBy(column: string, direction: 'ASC' | 'DESC' = 'ASC'): this {
    this.orderByClause = `${column} ${direction}`;
    return this;
  }

  build(): string {
    if (!this.table) throw new Error('Table is required');

    let query = `SELECT ${this.columns.join(', ')} FROM ${this.table}`;

    if (this.conditions.length > 0) {
      query += ` WHERE ${this.conditions.join(' AND ')}`;
    }
    if (this.orderByClause) query += ` ORDER BY ${this.orderByClause}`;
    if (this.limitValue !== undefined) query += ` LIMIT ${this.limitValue}`;
    if (this.offsetValue !== undefined) query += ` OFFSET ${this.offsetValue}`;

    return query;
  }
}

// Uso fluido
const query = new QueryBuilder()
  .from('users')
  .select('id', 'name', 'email')
  .where('active = true')
  .where('age >= 18')
  .orderBy('name')
  .limit(10)
  .offset(20)
  .build();
// "SELECT id, name, email FROM users WHERE active = true AND age >= 18 ORDER BY name ASC LIMIT 10 OFFSET 20"
```

## Builder type-safe con campos requeridos (using generics)

```typescript
// Técnica para garantizar en compile-time que los campos obligatorios se han proporcionado

type BuilderState<T> = Partial<T>;

// Marca los campos que han sido proporcionados como Required
type Complete<T, K extends keyof T> = BuilderState<T> & Required<Pick<T, K>>;

interface EmailConfig {
  to: string;       // Requerido
  from: string;     // Requerido
  subject: string;  // Requerido
  body: string;     // Requerido
  cc?: string;
  bcc?: string;
  attachments?: string[];
}

class EmailBuilder<TProvided extends Partial<EmailConfig> = {}> {
  private config: TProvided;

  private constructor(config: TProvided) {
    this.config = config;
  }

  static create(): EmailBuilder<{}> {
    return new EmailBuilder({});
  }

  to(email: string): EmailBuilder<TProvided & { to: string }> {
    return new EmailBuilder({ ...this.config, to: email });
  }

  from(email: string): EmailBuilder<TProvided & { from: string }> {
    return new EmailBuilder({ ...this.config, from: email });
  }

  subject(text: string): EmailBuilder<TProvided & { subject: string }> {
    return new EmailBuilder({ ...this.config, subject: text });
  }

  body(html: string): EmailBuilder<TProvided & { body: string }> {
    return new EmailBuilder({ ...this.config, body: html });
  }

  cc(email: string): this {
    return new EmailBuilder({ ...this.config, cc: email }) as this;
  }

  // build() solo está disponible cuando todos los campos requeridos han sido proporcionados
  build(
    this: EmailBuilder<TProvided & { to: string; from: string; subject: string; body: string }>
  ): EmailConfig {
    return this.config as EmailConfig;
  }
}

// ✅ Funciona: todos los campos requeridos están presentes
const email = EmailBuilder.create()
  .to('usuario@ejemplo.com')
  .from('noreply@miapp.com')
  .subject('Bienvenido')
  .body('<h1>Hola!</h1>')
  .build();

// ❌ Error en compilación: falta 'body' para llamar a build()
const incomplete = EmailBuilder.create()
  .to('usuario@ejemplo.com')
  .from('noreply@miapp.com')
  .subject('Bienvenido')
  .build(); // Error: 'build' no existe en el tipo actual
```

## Builder para objetos de test (Test Data Builder)

```typescript
interface User {
  id: string;
  name: string;
  email: string;
  age: number;
  role: 'user' | 'admin';
  createdAt: Date;
}

class UserBuilder {
  private user: User = {
    id: 'test-id-1',
    name: 'Test User',
    email: 'test@example.com',
    age: 25,
    role: 'user',
    createdAt: new Date('2024-01-01')
  };

  withId(id: string): this {
    this.user = { ...this.user, id };
    return this;
  }

  withName(name: string): this {
    this.user = { ...this.user, name };
    return this;
  }

  asAdmin(): this {
    this.user = { ...this.user, role: 'admin' };
    return this;
  }

  underage(): this {
    this.user = { ...this.user, age: 16 };
    return this;
  }

  build(): User {
    return { ...this.user };
  }
}

// En tests: conciso y expresivo
const adminUser = new UserBuilder().withName('Ana').asAdmin().build();
const youngUser = new UserBuilder().underage().build();
const specificUser = new UserBuilder()
  .withId('user-123')
  .withName('Luis')
  .build();
```

**Consejo de entrevista**: El patrón Builder es ideal para objetos con muchos parámetros opcionales donde el constructor se volvería inmanejable. El Builder type-safe usando genéricos es la versión TypeScript avanzada que garantiza en tiempo de compilación que los campos requeridos se proporcionan. El Test Data Builder es una variante muy práctica que mejora la legibilidad de los tests al crear objetos de prueba con valores por defecto sensatos.
