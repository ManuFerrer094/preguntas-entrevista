---
title: ¿Qué son @HostListener y @HostBinding en Angular?
difficulty: medium
tags: [angular, hostlistener, hostbinding, directives, decorators]
---

`@HostListener` y `@HostBinding` son decoradores que permiten a las directivas y componentes interactuar con su **elemento huésped** (host element): el elemento del DOM al que está aplicada la directiva o el componente.

## @HostListener — Escuchar eventos del elemento huésped

`@HostListener` vincula un método del componente/directiva a un evento del elemento huésped, sin necesidad de `addEventListener` manual.

```typescript
@Directive({
  selector: '[appResaltar]',
  standalone: true
})
export class ResaltarDirective {
  @HostListener('mouseenter')
  onMouseEnter(): void {
    console.log('Ratón entró en el elemento');
  }

  @HostListener('mouseleave')
  onMouseLeave(): void {
    console.log('Ratón salió del elemento');
  }

  // Acceder al evento nativo con $event
  @HostListener('click', ['$event'])
  onClick(event: MouseEvent): void {
    console.log('Click en posición:', event.clientX, event.clientY);
    event.stopPropagation();
  }

  // Escuchar eventos del documento o ventana
  @HostListener('document:keydown.escape')
  onEscapeKey(): void {
    console.log('Tecla Escape pulsada');
  }

  @HostListener('window:resize', ['$event'])
  onWindowResize(event: Event): void {
    console.log('Ventana redimensionada:', (event.target as Window).innerWidth);
  }
}
```

## @HostBinding — Vincular propiedades del elemento huésped

`@HostBinding` enlaza una propiedad o atributo del elemento huésped con una propiedad del componente/directiva.

```typescript
@Directive({
  selector: '[appResaltar]',
  standalone: true
})
export class ResaltarDirective {
  @HostBinding('style.backgroundColor') colorFondo = 'transparent';
  @HostBinding('style.cursor') cursor = 'default';
  @HostBinding('class.resaltado') estaResaltado = false;
  @HostBinding('attr.aria-selected') ariaSelected = false;

  @HostListener('mouseenter')
  onEnter(): void {
    this.colorFondo = 'lightyellow';
    this.cursor = 'pointer';
    this.estaResaltado = true;
    this.ariaSelected = true;
  }

  @HostListener('mouseleave')
  onLeave(): void {
    this.colorFondo = 'transparent';
    this.cursor = 'default';
    this.estaResaltado = false;
    this.ariaSelected = false;
  }
}
```

## Ejemplo completo: directiva de efecto hover

```typescript
@Directive({
  selector: '[appEfectoHover]',
  standalone: true
})
export class EfectoHoverDirective {
  @Input('appEfectoHover') colorResaltado = 'lightyellow';
  @Input() colorTextoHover = 'inherit';
  @Input() escala = 1.02;

  @HostBinding('style.transition')
  readonly transicion = 'all 0.2s ease-in-out';

  @HostBinding('style.backgroundColor') colorFondo = 'transparent';
  @HostBinding('style.color') colorTexto = 'inherit';
  @HostBinding('style.transform') transformacion = 'scale(1)';
  @HostBinding('style.boxShadow') sombra = 'none';

  @HostListener('mouseenter')
  onEnter(): void {
    this.colorFondo = this.colorResaltado;
    this.colorTexto = this.colorTextoHover;
    this.transformacion = `scale(${this.escala})`;
    this.sombra = '0 4px 8px rgba(0,0,0,0.15)';
  }

  @HostListener('mouseleave')
  onLeave(): void {
    this.colorFondo = 'transparent';
    this.colorTexto = 'inherit';
    this.transformacion = 'scale(1)';
    this.sombra = 'none';
  }
}
```

```html
<!-- Uso de la directiva -->
<div appEfectoHover colorResaltado="#e3f2fd" [escala]="1.05">
  Tarjeta con efecto hover
</div>

<button [appEfectoHover]="'#c8e6c9'" colorTextoHover="#1b5e20">
  Botón animado
</button>
```

## Directiva de permiso con @HostBinding

```typescript
@Directive({
  selector: '[appPermiso]',
  standalone: true
})
export class PermisoDirective implements OnInit {
  @Input('appPermiso') permisoRequerido!: string;

  @HostBinding('class.sin-permiso') sinPermiso = false;
  @HostBinding('attr.disabled') deshabilitado: boolean | null = null;
  @HostBinding('style.opacity') opacidad = '1';

  constructor(private authService: AuthService) {}

  ngOnInit(): void {
    const tienePermiso = this.authService.tienePermiso(this.permisoRequerido);
    this.sinPermiso = !tienePermiso;
    this.deshabilitado = tienePermiso ? null : true;
    this.opacidad = tienePermiso ? '1' : '0.5';
  }
}
```

```html
<button [appPermiso]="'admin:delete'">Eliminar usuario</button>
```

## host en los metadatos del componente (alternativa)

Los `@HostListener` y `@HostBinding` también se pueden definir en los metadatos del componente con la propiedad `host`:

```typescript
@Component({
  selector: 'app-card',
  standalone: true,
  host: {
    'class': 'card',                              // Clase estática
    '[class.activa]': 'estaActiva',               // HostBinding
    '(click)': 'onClick($event)',                  // HostListener
    '[attr.role]': '"article"',                    // Atributo estático
    '[style.border-color]': 'colorBorde'          // Style binding
  },
  template: `<ng-content></ng-content>`
})
export class CardComponent {
  estaActiva = false;
  colorBorde = '#ddd';

  onClick(e: MouseEvent): void { /* ... */ }
}
```

**Consejo de entrevista**: `@HostBinding` y `@HostListener` son más legibles cuando se usan en directivas individuales, mientras que la propiedad `host` en el decorador es útil cuando se quiere ver todos los bindings del host de un vistazo. Una ventaja de `@HostListener` sobre `addEventListener` es que Angular gestiona automáticamente la limpieza de los listeners al destruir la directiva/componente, previniendo memory leaks.
