---
title: ¿Qué es el HttpClientTesting en Angular?
difficulty: medium
tags: [angular, testing, httpclient, http, mocks]
---

`HttpClientTestingModule` (y el moderno `provideHttpClientTesting()`) proporciona herramientas para **interceptar y simular peticiones HTTP** en tests de Angular sin hacer peticiones reales al servidor. Permite verificar que los servicios realizan las peticiones correctas y manejan las respuestas adecuadamente.

## Configuración básica

```typescript
// Con TestBed moderno (Angular 15+)
import { TestBed } from '@angular/core/testing';
import { provideHttpClient } from '@angular/common/http';
import { provideHttpClientTesting, HttpTestingController } from '@angular/common/http/testing';

describe('UsuarioService', () => {
  let service: UsuarioService;
  let httpMock: HttpTestingController;

  beforeEach(() => {
    TestBed.configureTestingModule({
      providers: [
        UsuarioService,
        provideHttpClient(),
        provideHttpClientTesting()  // ✅ Intercepta todas las peticiones HTTP
      ]
    });

    service = TestBed.inject(UsuarioService);
    httpMock = TestBed.inject(HttpTestingController);
  });

  afterEach(() => {
    // Verifica que no queden peticiones pendientes sin verificar
    httpMock.verify();
  });
});
```

## Testear una petición GET

```typescript
it('debería obtener la lista de usuarios', () => {
  const usuariosMock: Usuario[] = [
    { id: 1, nombre: 'Ana García' },
    { id: 2, nombre: 'Carlos López' }
  ];

  // Llamar al servicio
  service.obtenerUsuarios().subscribe(usuarios => {
    expect(usuarios).toEqual(usuariosMock);
    expect(usuarios.length).toBe(2);
  });

  // Interceptar la petición HTTP
  const req = httpMock.expectOne('/api/usuarios');
  expect(req.request.method).toBe('GET');

  // Simular la respuesta del servidor
  req.flush(usuariosMock);
});
```

## Testear petición POST con body y headers

```typescript
it('debería crear un usuario con los datos correctos', () => {
  const nuevoUsuario = { nombre: 'María Pérez', email: 'maria@ejemplo.com' };
  const respuestaMock = { id: 3, ...nuevoUsuario };

  service.crearUsuario(nuevoUsuario).subscribe(usuario => {
    expect(usuario.id).toBe(3);
    expect(usuario.nombre).toBe('María Pérez');
  });

  const req = httpMock.expectOne('/api/usuarios');
  expect(req.request.method).toBe('POST');
  expect(req.request.body).toEqual(nuevoUsuario);
  expect(req.request.headers.get('Content-Type')).toBe('application/json');

  req.flush(respuestaMock);
});
```

## Simular errores HTTP

```typescript
it('debería manejar un error 404', () => {
  service.obtenerUsuario(999).subscribe({
    next: () => fail('Debería haber lanzado un error'),
    error: (error) => {
      expect(error.status).toBe(404);
      expect(error.statusText).toBe('Not Found');
    }
  });

  const req = httpMock.expectOne('/api/usuarios/999');

  // Simular respuesta de error
  req.flush('Usuario no encontrado', {
    status: 404,
    statusText: 'Not Found'
  });
});

it('debería manejar un error de red', () => {
  service.obtenerUsuarios().subscribe({
    next: () => fail('Debería haber lanzado un error'),
    error: (error) => {
      expect(error.type).toBe(HttpEventType.Response);
    }
  });

  const req = httpMock.expectOne('/api/usuarios');
  req.error(new ProgressEvent('Network error'));
});
```

## Verificar query params y múltiples peticiones

```typescript
it('debería buscar con los parámetros correctos', () => {
  service.buscar({ q: 'Angular', pagina: 2 }).subscribe();

  const req = httpMock.expectOne(r =>
    r.url === '/api/buscar' &&
    r.params.get('q') === 'Angular' &&
    r.params.get('pagina') === '2'
  );

  expect(req.request.method).toBe('GET');
  req.flush([]);
});

it('debería manejar múltiples peticiones', () => {
  service.obtenerUsuario(1).subscribe();
  service.obtenerUsuario(2).subscribe();

  // Verificar que se hicieron exactamente 2 peticiones
  const reqs = httpMock.match('/api/usuarios');
  expect(reqs.length).toBe(0); // url sin ID

  const req1 = httpMock.expectOne('/api/usuarios/1');
  const req2 = httpMock.expectOne('/api/usuarios/2');

  req1.flush({ id: 1, nombre: 'Ana' });
  req2.flush({ id: 2, nombre: 'Carlos' });
});
```

## Testear reintentos con retry

```typescript
it('debería reintentar en caso de error', () => {
  let intentos = 0;

  service.obtenerConRetry().subscribe({
    next: (data) => expect(data).toBeTruthy(),
    error: () => fail('No debería fallar')
  });

  // Primer intento falla
  httpMock.expectOne('/api/datos').error(new ProgressEvent('error'));
  intentos++;

  // Segundo intento (retry) falla
  httpMock.expectOne('/api/datos').error(new ProgressEvent('error'));
  intentos++;

  // Tercer intento tiene éxito
  httpMock.expectOne('/api/datos').flush({ data: 'ok' });

  expect(intentos).toBe(2);
});
```

**Consejo de entrevista**: `httpMock.verify()` en `afterEach` es crucial: detecta peticiones esperadas que no se verificaron o peticiones inesperadas que se realizaron. Esto previene tests que "pasan" pero con comportamientos incorrectos. Para interceptores HTTP, también se puede usar `httpMock` para verificar que los headers de autenticación se añaden correctamente.
