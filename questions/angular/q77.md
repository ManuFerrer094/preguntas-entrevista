`ViewContainerRef` es una referencia a un **contenedor de vistas** en el DOM donde Angular puede insertar dinámicamente componentes, templates o vistas embebidas en tiempo de ejecución, sin necesidad de declararlos en el template estático.

## createComponent() — Insertar componente dinámicamente

```typescript
import {
  Component, ViewChild, ViewContainerRef,
  ComponentRef, inject, afterNextRender
} from '@angular/core';

@Component({
  standalone: true,
  template: `
    <button (click)="cargarGrafico()">Cargar Gráfico</button>
    <!-- Punto de inserción dinámico -->
    <ng-container #contenedor></ng-container>
  `
})
export class PanelComponent {
  @ViewChild('contenedor', { read: ViewContainerRef })
  contenedor!: ViewContainerRef;

  private componentRef: ComponentRef<GraficoComponent> | null = null;

  async cargarGrafico() {
    // Importación dinámica (lazy loading)
    const { GraficoComponent } = await import('./grafico/grafico.component');

    this.contenedor.clear();  // Limpiar el contenedor

    // Crear e insertar el componente
    this.componentRef = this.contenedor.createComponent(GraficoComponent);

    // Pasar datos (equivalente a @Input)
    this.componentRef.setInput('datos', this.datos);
    this.componentRef.setInput('titulo', 'Ventas 2024');

    // Suscribirse a outputs
    this.componentRef.instance.cierreClick.subscribe(() => this.cerrar());
  }

  cerrar() {
    this.componentRef?.destroy();
    this.contenedor.clear();
  }
}
```

## Sistema de modales dinámicos

```typescript
// modal.service.ts
@Injectable({ providedIn: 'root' })
export class ModalService {
  private vcr!: ViewContainerRef;

  // Registrar el contenedor raíz (desde AppComponent)
  registrar(vcr: ViewContainerRef) {
    this.vcr = vcr;
  }

  abrir<T>(componente: Type<T>, datos?: Partial<T>): ComponentRef<T> {
    this.vcr.clear();
    const ref = this.vcr.createComponent(componente);

    // Inyectar datos como inputs
    if (datos) {
      Object.entries(datos).forEach(([key, value]) => {
        ref.setInput(key, value);
      });
    }

    return ref;
  }

  cerrar() {
    this.vcr.clear();
  }
}

// app.component.ts
@Component({
  standalone: true,
  template: `
    <router-outlet />
    <ng-container #modalContainer></ng-container>
  `
})
export class AppComponent implements AfterViewInit {
  @ViewChild('modalContainer', { read: ViewContainerRef })
  modalContainer!: ViewContainerRef;

  private modalService = inject(ModalService);

  ngAfterViewInit() {
    this.modalService.registrar(this.modalContainer);
  }
}

// Uso en cualquier componente:
@Component({ standalone: true })
export class ProductoComponent {
  private modalService = inject(ModalService);

  verDetalle(producto: Producto) {
    const ref = this.modalService.abrir(ProductoDetalleComponent, {
      producto,
      modo: 'detalle'
    });

    ref.instance.cerrar.subscribe(() => this.modalService.cerrar());
  }
}
```

## createEmbeddedView() — Insertar ng-template dinámicamente

```typescript
@Component({
  standalone: true,
  template: `
    <ng-template #tmplCargando>
      <div class="spinner">Cargando...</div>
    </ng-template>

    <ng-template #tmplError let-mensaje="mensaje">
      <div class="error">{{ mensaje }}</div>
    </ng-template>

    <ng-container #contenedor></ng-container>
    <button (click)="mostrarEstado('cargando')">Cargando</button>
    <button (click)="mostrarEstado('error')">Error</button>
  `
})
export class EstadoComponent {
  @ViewChild('contenedor', { read: ViewContainerRef }) vcr!: ViewContainerRef;
  @ViewChild('tmplCargando') tmplCargando!: TemplateRef<void>;
  @ViewChild('tmplError') tmplError!: TemplateRef<{ mensaje: string }>;

  mostrarEstado(estado: 'cargando' | 'error') {
    this.vcr.clear();

    if (estado === 'cargando') {
      this.vcr.createEmbeddedView(this.tmplCargando);
    } else {
      // Pasar contexto al template
      this.vcr.createEmbeddedView(this.tmplError, { mensaje: 'No se pudo cargar' });
    }
  }
}
```

## Sistema de notificaciones tipo Toast

```typescript
@Injectable({ providedIn: 'root' })
export class ToastService {
  private vcr!: ViewContainerRef;
  private refs: ComponentRef<ToastComponent>[] = [];

  registrar(vcr: ViewContainerRef) { this.vcr = vcr; }

  mostrar(mensaje: string, tipo: 'info' | 'error' | 'exito' = 'info') {
    const ref = this.vcr.createComponent(ToastComponent);
    ref.setInput('mensaje', mensaje);
    ref.setInput('tipo', tipo);

    this.refs.push(ref);

    // Auto-cerrar después de 3 segundos
    setTimeout(() => this.cerrar(ref), 3000);
    return ref;
  }

  cerrar(ref: ComponentRef<ToastComponent>) {
    const idx = this.refs.indexOf(ref);
    if (idx > -1) {
      this.refs.splice(idx, 1);
      ref.destroy();
    }
  }
}
```

**Consejo de entrevista**: `ViewContainerRef` es la base de sistemas avanzados de UI como modales, toasts, tooltips y overlays dinámicos. La diferencia con `*ngIf` es que los componentes dinámicos se crean completamente en runtime, permitiendo lazy loading y composición dinámica. En Angular moderno, `setInput()` es la forma correcta de pasar datos (respeta el ciclo de vida y el change detection), en lugar de acceder directamente a `instance.propiedad`.
