Los **formularios dinámicos** en Angular son formularios construidos programáticamente en tiempo de ejecución a partir de una definición de datos (JSON, objeto de configuración), en lugar de definirlos estáticamente en el template. Son ideales para formularios configurables, builders de encuestas, o formularios gestionados desde el backend.

## Definición del modelo de campos

```typescript
// form-field.model.ts
export interface CampoFormulario {
  key: string;
  tipo: 'text' | 'email' | 'number' | 'select' | 'checkbox' | 'textarea' | 'date';
  etiqueta: string;
  valor?: unknown;
  requerido?: boolean;
  minLength?: number;
  maxLength?: number;
  opciones?: { valor: string; texto: string }[];  // Para select
  orden?: number;
}
```

## Servicio que genera el FormGroup dinámicamente

```typescript
// dynamic-form.service.ts
@Injectable({ providedIn: 'root' })
export class DynamicFormService {
  generarFormGroup(campos: CampoFormulario[]): FormGroup {
    const controles: Record<string, FormControl> = {};

    campos.forEach(campo => {
      const validadores: ValidatorFn[] = [];

      if (campo.requerido) validadores.push(Validators.required);
      if (campo.tipo === 'email') validadores.push(Validators.email);
      if (campo.minLength) validadores.push(Validators.minLength(campo.minLength));
      if (campo.maxLength) validadores.push(Validators.maxLength(campo.maxLength));

      controles[campo.key] = new FormControl(
        campo.valor ?? null,
        validadores
      );
    });

    return new FormGroup(controles);
  }
}
```

## Componente de campo dinámico individual

```typescript
// dynamic-field.component.ts
@Component({
  selector: 'app-dynamic-field',
  standalone: true,
  imports: [ReactiveFormsModule, NgSwitch, NgSwitchCase, NgForOf],
  template: `
    <div class="campo" [formGroup]="formGroup">
      <label [for]="campo.key">
        {{ campo.etiqueta }}
        <span *ngIf="campo.requerido" class="requerido">*</span>
      </label>

      @switch (campo.tipo) {
        @case ('select') {
          <select [id]="campo.key" [formControlName]="campo.key">
            <option value="">-- Seleccionar --</option>
            @for (op of campo.opciones; track op.valor) {
              <option [value]="op.valor">{{ op.texto }}</option>
            }
          </select>
        }
        @case ('textarea') {
          <textarea [id]="campo.key" [formControlName]="campo.key" rows="4"></textarea>
        }
        @case ('checkbox') {
          <input type="checkbox" [id]="campo.key" [formControlName]="campo.key">
        }
        @default {
          <input [type]="campo.tipo" [id]="campo.key" [formControlName]="campo.key">
        }
      }

      @if (control?.invalid && control?.touched) {
        <div class="error">
          @if (control?.errors?.['required']) { <span>Este campo es requerido.</span> }
          @if (control?.errors?.['email']) { <span>Email no válido.</span> }
          @if (control?.errors?.['minlength']) {
            <span>Mínimo {{ campo.minLength }} caracteres.</span>
          }
        </div>
      }
    </div>
  `
})
export class DynamicFieldComponent {
  @Input({ required: true }) campo!: CampoFormulario;
  @Input({ required: true }) formGroup!: FormGroup;

  get control() {
    return this.formGroup.get(this.campo.key);
  }
}
```

## Formulario dinámico completo

```typescript
// dynamic-form.component.ts
@Component({
  selector: 'app-dynamic-form',
  standalone: true,
  imports: [ReactiveFormsModule, DynamicFieldComponent],
  template: `
    <form [formGroup]="form" (ngSubmit)="onSubmit()">
      @for (campo of camposOrdenados; track campo.key) {
        <app-dynamic-field [campo]="campo" [formGroup]="form" />
      }
      <button type="submit" [disabled]="form.invalid">Enviar</button>
    </form>
  `
})
export class DynamicFormComponent implements OnInit {
  @Input({ required: true }) campos: CampoFormulario[] = [];
  @Output() formSubmit = new EventEmitter<Record<string, unknown>>();

  form!: FormGroup;

  private formService = inject(DynamicFormService);

  get camposOrdenados() {
    return [...this.campos].sort((a, b) => (a.orden ?? 0) - (b.orden ?? 0));
  }

  ngOnInit() {
    this.form = this.formService.generarFormGroup(this.campos);
  }

  onSubmit() {
    if (this.form.valid) {
      this.formSubmit.emit(this.form.value);
    } else {
      this.form.markAllAsTouched();
    }
  }
}
```

## Uso con configuración desde el backend

```typescript
// Configuración que podría venir de una API:
const camposEncuesta: CampoFormulario[] = [
  { key: 'nombre',    tipo: 'text',     etiqueta: 'Nombre completo', requerido: true, minLength: 2, orden: 1 },
  { key: 'email',     tipo: 'email',    etiqueta: 'Correo electrónico', requerido: true, orden: 2 },
  { key: 'edad',      tipo: 'number',   etiqueta: 'Edad', orden: 3 },

// En el template:

```

**Consejo de entrevista**: Los formularios dinámicos son un patrón avanzado que combina `FormGroup`, `FormControl`, validadores dinámicos y renderizado condicional. La clave está en separar el modelo de datos (definición del campo) de la presentación (componente de campo). Este patrón es muy común en sistemas de CMS, builders de formularios y plataformas SaaS. Menciona también que `FormArray` puede usarse para listas de campos repetibles (como añadir múltiples teléfonos).
