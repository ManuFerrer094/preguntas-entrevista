La paginación en Razor Pages con EF Core combina el patrón PageModel con consultas LINQ paginadas, evitando cargar todos los registros en memoria.

**Clase helper `PaginatedList<T>`:**

```csharp
public class PaginatedList<T> : List<T>
{
    public int PageIndex { get; }
    public int TotalPages { get; }
    public bool HasPreviousPage => PageIndex > 1;
    public bool HasNextPage => PageIndex < TotalPages;

    private PaginatedList(List<T> items, int count, int pageIndex, int pageSize)
    {
        PageIndex = pageIndex;
        TotalPages = (int)Math.Ceiling(count / (double)pageSize);
        AddRange(items);
    }

    public static async Task<PaginatedList<T>> CreateAsync(
        IQueryable<T> source, int pageIndex, int pageSize)
    {
        var count = await source.CountAsync();
        var items = await source.Skip((pageIndex - 1) * pageSize).Take(pageSize).ToListAsync();
        return new PaginatedList<T>(items, count, pageIndex, pageSize);
    }
}
```

**PageModel:**

```csharp
public class IndexModel : PageModel
{
    private readonly AppDbContext _db;
    public PaginatedList<Producto> Productos { get; set; }

    [BindProperty(SupportsGet = true)]
    public int PageIndex { get; set; } = 1;

    public async Task OnGetAsync()
    {
        Productos = await PaginatedList<Producto>.CreateAsync(
            _db.Productos.OrderBy(p => p.Nombre), PageIndex, 10);
    }
}
```

**Vista Razor:**

```razor
@foreach (var p in Model.Productos)
{
    <tr><td>@p.Nombre</td></tr>
}

<nav>
    @if (Model.Productos.HasPreviousPage)
    {
        <a asp-page="./Index" asp-route-pageIndex="@(Model.PageIndex - 1)">Anterior</a>
    }
    @if (Model.Productos.HasNextPage)
    {
        <a asp-page="./Index" asp-route-pageIndex="@(Model.PageIndex + 1)">Siguiente</a>
    }
</nav>
```
