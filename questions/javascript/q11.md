El objeto `Proxy` en JavaScript permite interceptar y redefinir operaciones fundamentales de un objeto: acceso a propiedades, asignación, eliminación, invocación de funciones, y más. Es la base de la metaprogramación en JavaScript y se usa internamente en Vue 3 para su sistema de reactividad.

Un Proxy envuelve un objeto objetivo (`target`) con un manejador (`handler`) que define "trampas" (traps) para interceptar operaciones.

## Sintaxis y trampas básicas

```javascript
const target = { name: 'Ana', age: 28 };

const handler = {
  // Interceptar lecturas de propiedades
  get(target, property, receiver) {
    console.log(`Leyendo: ${String(property)}`);
    return Reflect.get(target, property, receiver);
  },

  // Interceptar escrituras de propiedades
  set(target, property, value, receiver) {
    console.log(`Escribiendo ${String(property)} = ${value}`);

    // Validación
    if (property === 'age' && typeof value !== 'number') {
      throw new TypeError('La edad debe ser un número');
    }

    return Reflect.set(target, property, value, receiver);
  },

  // Interceptar eliminación de propiedades
  deleteProperty(target, property) {
    if (property === 'name') {
      throw new Error('No se puede eliminar el nombre');
    }
    return Reflect.deleteProperty(target, property);
  },

  // Interceptar el operador 'in'
  has(target, property) {
    if (property === 'secret') return false; // Ocultar propiedad
    return Reflect.has(target, property);
  }
};

const proxy = new Proxy(target, handler);

proxy.name;      // "Leyendo: name" → 'Ana'
proxy.age = 30;  // "Escribiendo age = 30"
proxy.age = 'treinta'; // ❌ TypeError
```

## Caso de uso: Validación de objetos

```javascript
function createValidatedObject(schema) {
  return new Proxy({}, {
    set(target, property, value) {
      const validator = schema[property];

      if (!validator) {
        throw new Error(`Propiedad '${String(property)}' no permitida`);
      }

      const error = validator(value);
      if (error) throw new TypeError(error);

      target[property] = value;
      return true;
    }
  });
}

const userSchema = {
  name: (v) => typeof v !== 'string' ? 'name debe ser string' : null,
  age: (v) => (typeof v !== 'number' || v < 0) ? 'age debe ser número positivo' : null,
  email: (v) => !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v) ? 'email inválido' : null
};

const user = createValidatedObject(userSchema);
user.name = 'Ana';         // ✅
user.age = 28;             // ✅
// user.age = -5;           // ❌ TypeError: age debe ser número positivo
// user.unknown = 'test';   // ❌ Error: Propiedad 'unknown' no permitida
```

## Caso de uso: Objeto reactivo (como Vue 3)

```javascript
function reactive(obj, onChange) {
  return new Proxy(obj, {
    get(target, key) {
      const value = target[key];
      // Si el valor es un objeto, hacerlo también reactivo
      if (typeof value === 'object' && value !== null) {
        return reactive(value, onChange);
      }
      return value;
    },
    set(target, key, value) {
      const oldValue = target[key];
      target[key] = value;
      if (oldValue !== value) {
        onChange(key, value, oldValue);
      }
      return true;
    }
  });
}

const state = reactive({ count: 0, user: { name: 'Ana' } }, (key, newVal) => {
  console.log(`${key} cambió a: ${newVal}`);
  // Re-renderizar componente...
});

state.count = 1;         // "count cambió a: 1"
state.user.name = 'Carlos'; // "name cambió a: Carlos"
```

## Proxy para logging automático

```javascript
function createLoggingProxy(service, logger) {
  return new Proxy(service, {
    get(target, property) {
      const original = target[property];
      if (typeof original !== 'function') return original;

      return async function(...args) {
        const start = Date.now();
        try {
          const result = await original.apply(target, args);
          logger.info(`${property} completado en ${Date.now() - start}ms`);
          return result;
        } catch (error) {
          logger.error(`${property} falló: ${error.message}`);
          throw error;
        }
      };
    }
  });
}

const loggedUserService = createLoggingProxy(userService, logger);
await loggedUserService.getUser('123'); // Loggea automáticamente duración y errores
```

**Consejo de entrevista**: El Proxy es la base del sistema de reactividad de Vue 3 y se usa en muchos frameworks para observar cambios de estado. Sus principales casos de uso son: validación, logging automático, memoización, control de acceso, e implementación de patrones reactivos. Mencionar que `Reflect` se usa junto con `Proxy` para mantener el comportamiento por defecto es un detalle que demuestra conocimiento profundo.
