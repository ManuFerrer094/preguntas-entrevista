---
title: ¿Cuál es la diferencia entre null y undefined en JavaScript?
difficulty: easy
tags: [javascript, null, undefined, types, primitives]
---

`null` y `undefined` representan la ausencia de valor, pero con semánticas distintas: `undefined` es la ausencia de valor por defecto del lenguaje, mientras que `null` es una asignación intencional que indica "sin valor".

## Diferencias fundamentales

```javascript
// undefined: el lenguaje asigna este valor automáticamente
let x;
console.log(x);             // undefined — variable declarada pero no inicializada

function greet(name) {
  console.log(name);        // undefined — parámetro no pasado
}
greet();

const obj = { a: 1 };
console.log(obj.b);         // undefined — propiedad inexistente

function noReturn() {}
console.log(noReturn());    // undefined — función sin return

// null: el programador asigna explícitamente "sin valor"
let selectedUser = null;    // Indicación intencional de "ningún usuario seleccionado"
let response = null;        // La respuesta aún no ha llegado
```

## Comparaciones

```javascript
// Igualdad abstracta (==): null y undefined son iguales entre sí
console.log(null == undefined);   // true
console.log(null == null);        // true
console.log(undefined == undefined); // true

// null y undefined NO son iguales a ningún otro valor con ==
console.log(null == 0);           // false
console.log(null == false);       // false
console.log(null == '');          // false
console.log(undefined == 0);     // false
console.log(undefined == false);  // false

// Igualdad estricta (===): son distintos
console.log(null === undefined);  // false
console.log(null === null);       // true
console.log(undefined === undefined); // true

// typeof
console.log(typeof null);         // 'object' — bug histórico de JS
console.log(typeof undefined);    // 'undefined'
```

## Comportamiento en operaciones aritméticas

```javascript
// null se convierte a 0 en contextos numéricos
console.log(null + 1);       // 1
console.log(null * 5);       // 0
console.log(Number(null));   // 0

// undefined se convierte a NaN
console.log(undefined + 1); // NaN
console.log(undefined * 5); // NaN
console.log(Number(undefined)); // NaN

// En JSON
JSON.stringify({ a: null, b: undefined });
// '{"a":null}' — undefined desaparece en JSON
JSON.stringify([null, undefined]);
// '[null,null]' — undefined se convierte a null en arrays JSON
```

## Patrones de comprobación

```javascript
// Comprobar ambos a la vez (patrón idiomático)
function processUser(user) {
  if (user == null) {  // Comprueba tanto null como undefined
    return 'Sin usuario';
  }
  return user.name;
}

processUser(null);       // 'Sin usuario'
processUser(undefined);  // 'Sin usuario'
processUser({ name: 'Ana' }); // 'Ana'

// Nullish coalescing (??) — solo actúa con null/undefined
const value1 = null ?? 'default';      // 'default'
const value2 = undefined ?? 'default'; // 'default'
const value3 = 0 ?? 'default';         // 0 (0 no es null/undefined)
const value4 = '' ?? 'default';        // '' ('' no es null/undefined)

// OR (||) — actúa con cualquier valor falsy
const value5 = 0 || 'default';         // 'default' (diferencia clave)
const value6 = '' || 'default';        // 'default'

// Optional chaining (?.) — evita errores con null/undefined
const user = null;
console.log(user?.name);         // undefined (no lanza error)
console.log(user?.address?.city); // undefined

// Nullish assignment (??=)
let config = null;
config ??= { theme: 'dark' }; // Solo asigna si es null/undefined
console.log(config); // { theme: 'dark' }
```

## Cuándo usar cada uno

```javascript
// ✅ Usar null cuando quieres indicar explícitamente "sin valor"
class UserService {
  #currentUser = null;  // null: usuario no autenticado

  login(user) {
    this.#currentUser = user;
  }

  logout() {
    this.#currentUser = null; // Restablece a "sin usuario"
  }

  getCurrentUser() {
    return this.#currentUser; // Devuelve null o un objeto User
  }
}

// ✅ undefined queda para el comportamiento por defecto del lenguaje
// No asignes undefined manualmente — usa null en su lugar
// ❌ No recomendado:
let selectedId = undefined; // Usa null en su lugar
```

**Consejo de entrevista**: La regla práctica es: usa `null` cuando quieras indicar intencionalmente "sin valor" (lo asignas tú), deja `undefined` para el lenguaje. Para comprobar ambos, usa `== null` o el operador `??`. Mencionar el bug `typeof null === 'object'` y la diferencia de comportamiento de `??` vs `||` demuestra un conocimiento sólido.
