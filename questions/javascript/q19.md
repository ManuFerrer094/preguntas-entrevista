El Garbage Collector (GC) de JavaScript libera automáticamente la memoria que ya no es accesible desde el código. El algoritmo principal moderno es **Mark and Sweep**.

## Algoritmo Mark and Sweep

```javascript
// El GC parte de las "raíces" (variables globales, stack de llamadas)
// y marca todo lo accesible. Lo no marcado se libera.

let obj = { name: 'Ana' };  // obj es accesible → marcado
let ref = obj;               // ref también apunta al mismo objeto

obj = null; // obj ya no apunta al objeto
// ref todavía apunta → el objeto sigue siendo accesible, NO se libera

ref = null; // Ahora ninguna variable apunta al objeto
// El objeto queda inalcanzable → el GC lo puede liberar
```

## Memory Leaks más comunes

```javascript
// 1. Variables globales accidentales
function badFunction() {
  leak = 'Soy una variable global accidental'; // sin var/let/const
}

// 2. Closures que retienen referencias innecesarias
function createLeak() {
  const largeArray = new Array(1000000).fill('datos');

  return function() {
    // Este closure retiene largeArray aunque no lo use
    console.log('función simple');
  };
}
const fn = createLeak(); // largeArray nunca se libera

// Solución: no capturar lo que no se necesita
function createOptimized() {
  const largeArray = new Array(1000000).fill('datos');
  const result = processData(largeArray); // Extraer lo necesario
  // largeArray puede ser liberado, closure solo captura result
  return function() {
    return result;
  };
}

// 3. Event listeners no eliminados
const element = document.getElementById('btn');
const handler = () => console.log('click');
element.addEventListener('click', handler);

// Sin esto, el handler (y todo su closure) nunca se libera:
element.removeEventListener('click', handler);

// 4. Referencias circulares (ya manejadas por Mark and Sweep, pero
//    pueden causar problemas en WeakMap/WeakSet incorrectos)
function circularReference() {
  const a = {};
  const b = {};
  a.ref = b;
  b.ref = a;
  // Cuando a y b salgan de scope, el GC las libera correctamente
  // porque Mark and Sweep detecta que son inalcanzables desde raíces
}
```

## WeakRef y FinalizationRegistry (ES2021)

```javascript
// WeakRef: referencia débil que no impide la recolección
let user = { name: 'Ana', data: new Array(100000) };
const weakRef = new WeakRef(user);

user = null; // El objeto puede ser recogido por el GC

// En algún momento posterior:
const obj = weakRef.deref();
if (obj) {
  console.log(obj.name); // Todavía existe
} else {
  console.log('El objeto fue recogido por el GC');
}

// FinalizationRegistry: callback cuando un objeto es recogido
const registry = new FinalizationRegistry((value) => {
  console.log(`Objeto con clave "${value}" fue recogido por el GC`);
});

let resource = { connection: 'DB connection' };
registry.register(resource, 'db-connection');

resource = null; // Cuando el GC lo recoja, ejecutará el callback
```

## Usar WeakMap para caché sin memory leaks

```javascript
// ✅ WeakMap: las entradas se liberan cuando la clave no es accesible
const cache = new WeakMap();

function processUser(user) {
  if (cache.has(user)) {
    return cache.get(user); // Cache hit
  }

  const result = expensiveComputation(user);
  cache.set(user, result);
  return result;
}

// Cuando el objeto user ya no sea accesible desde fuera,
// el GC puede liberar tanto la clave como el valor del WeakMap

// ❌ Map normal retiene la referencia indefinidamente
const badCache = new Map();
function processUserBad(user) {
  if (!badCache.has(user)) {
    badCache.set(user, expensiveComputation(user));
  }
  return badCache.get(user);
  // user nunca se libera mientras badCache exista
}
```

## Herramientas para detectar memory leaks

```javascript
// En Node.js: monitorizar el heap
const { memoryUsage } = process;

function checkMemory() {
  const { heapUsed, heapTotal, external } = memoryUsage();
  console.log(`Heap usado: ${Math.round(heapUsed / 1024 / 1024)} MB`);
  console.log(`Heap total: ${Math.round(heapTotal / 1024 / 1024)} MB`);
}

// Forzar recolección en Node.js (solo para diagnóstico)
// node --expose-gc script.js
if (global.gc) {
  global.gc();
  checkMemory();
}
```

**Consejo de entrevista**: Los memory leaks más frecuentes son event listeners no eliminados, timers (`setInterval`) no limpiados, y closures que retienen referencias grandes. Usa `WeakMap` y `WeakSet` cuando las claves sean objetos para evitar retener referencias innecesarias. En el navegador, Chrome DevTools Memory tab permite tomar heap snapshots y comparar para detectar leaks.
