Las clases en JavaScript (ES6) son "azúcar sintáctica" sobre el sistema de prototipos existente. No introducen un nuevo modelo de herencia orientada a objetos; simplemente ofrecen una sintaxis más clara y familiar para trabajar con prototipos y constructores.

## Sintaxis de clases

```javascript
class Animal {
  // Propiedad de clase (estática)
  static count = 0;

  // Campos privados (ES2022)
  #name;
  #sound;

  constructor(name, sound) {
    this.#name = name;
    this.#sound = sound;
    Animal.count++;
  }

  // Método de instancia
  speak() {
    return `${this.#name} dice: ${this.#sound}!`;
  }

  // Getter
  get name() {
    return this.#name;
  }

  // Setter con validación
  set name(value) {
    if (typeof value !== 'string' || value.trim().length === 0) {
      throw new Error('El nombre debe ser un string no vacío');
    }
    this.#name = value.trim();
  }

  // Método estático — pertenece a la clase, no a la instancia
  static create(name, sound) {
    return new Animal(name, sound);
  }

  toString() {
    return `Animal(${this.#name})`;
  }
}

const dog = new Animal('Rex', 'Guau');
console.log(dog.speak()); // "Rex dice: Guau!"
console.log(dog.name);    // "Rex" (getter)
console.log(Animal.count); // 1 (propiedad estática)
const cat = Animal.create('Whiskers', 'Miau'); // Método estático
```

## Herencia con extends

```javascript
class Dog extends Animal {
  #breed;

  constructor(name, breed) {
    super(name, 'Guau'); // Llamar al constructor del padre PRIMERO
    this.#breed = breed;
  }

  // Sobrescribir método del padre
  speak() {
    return `${super.speak()} (${this.#breed})`; // super.método() para llamar al padre
  }

  fetch() {
    return `${this.name} trae la pelota!`;
  }

  get info() {
    return `${this.name} - ${this.#breed}`;
  }
}

const labrador = new Dog('Buddy', 'Labrador');
console.log(labrador.speak()); // "Buddy dice: Guau! (Labrador)"
console.log(labrador instanceof Dog);    // true
console.log(labrador instanceof Animal); // true
```

## Mixins — herencia múltiple simulada

JavaScript no soporta herencia múltiple, pero se puede simular con mixins:

```javascript
const Serializable = (Base) => class extends Base {
  serialize() {
    return JSON.stringify(this);
  }

  static deserialize(json) {
    return Object.assign(new this(), JSON.parse(json));
  }
};

const Timestamped = (Base) => class extends Base {
  #createdAt = new Date();
  #updatedAt = new Date();

  get createdAt() { return this.#createdAt; }

  touch() { this.#updatedAt = new Date(); }
};

class User extends Timestamped(Serializable(class {})) {
  constructor(name, email) {
    super();
    this.name = name;
    this.email = email;
  }
}

const user = new User('Ana', 'ana@test.com');
user.serialize(); // Método de Serializable
user.createdAt;   // Propiedad de Timestamped
```

## Clases vs funciones constructoras (lo que hay debajo)

```javascript
// ES6 class
class Rectangle {
  constructor(width, height) {
    this.width = width;
    this.height = height;
  }
  area() { return this.width * this.height; }
}

// Equivalente con prototipos (lo que hace JavaScript internamente)
function Rectangle(width, height) {
  this.width = width;
  this.height = height;
}
Rectangle.prototype.area = function() {
  return this.width * this.height;
};

// Ambos son equivalentes:
const r1 = new Rectangle(5, 3);
r1.area(); // 15

// Las clases son funciones
console.log(typeof Rectangle); // 'function'
console.log(Rectangle.prototype.constructor === Rectangle); // true
```

**Consejo de entrevista**: Aunque las clases son sintaxis sobre prototipos, en la práctica moderna se usan clases directamente y no hay necesidad de manipular prototipos directamente. Los puntos clave son: `extends` para herencia, `super()` debe llamarse primero en el constructor hijo, los campos privados con `#` son verdaderamente privados (no accesibles ni con Proxy), y los métodos estáticos pertenecen a la clase, no a las instancias.
