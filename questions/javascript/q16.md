---
title: ¿Cómo puedes cancelar una petición fetch con AbortController?
difficulty: medium
tags: [javascript, fetch, abort-controller, async, api]
---

`AbortController` es la API nativa del navegador (y Node.js 16+) que permite cancelar peticiones `fetch` en vuelo, evitando actualizaciones de estado en componentes desmontados y condiciones de carrera.

## Uso básico

```javascript
const controller = new AbortController();
const signal = controller.signal;

fetch('/api/users', { signal })
  .then(res => res.json())
  .then(data => console.log(data))
  .catch(err => {
    if (err.name === 'AbortError') {
      console.log('Petición cancelada');
    } else {
      console.error('Error real:', err);
    }
  });

// Cancelar la petición
controller.abort();
```

## Cancelación con timeout

```javascript
async function fetchWithTimeout(url, timeoutMs = 5000) {
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), timeoutMs);

  try {
    const response = await fetch(url, { signal: controller.signal });
    if (!response.ok) throw new Error(`HTTP ${response.status}`);
    return await response.json();
  } catch (err) {
    if (err.name === 'AbortError') {
      throw new Error(`Tiempo de espera agotado después de ${timeoutMs}ms`);
    }
    throw err;
  } finally {
    clearTimeout(timeoutId); // Limpiar siempre el timeout
  }
}
```

## Cancelar la petición anterior en búsquedas en tiempo real

```javascript
let currentController = null;

async function searchUsers(query) {
  // Cancelar la búsqueda anterior si todavía está en curso
  if (currentController) {
    currentController.abort();
  }

  currentController = new AbortController();

  try {
    const response = await fetch(`/api/users?q=${query}`, {
      signal: currentController.signal
    });
    return await response.json();
  } catch (err) {
    if (err.name === 'AbortError') {
      return null; // Búsqueda cancelada intencionalmente
    }
    throw err;
  }
}

// Solo se procesa la última búsqueda activa
document.getElementById('search').addEventListener('input', async (e) => {
  const results = await searchUsers(e.target.value);
  if (results) updateUI(results);
});
```

## Cancelar múltiples peticiones a la vez

```javascript
const controller = new AbortController();
const { signal } = controller;

// El mismo signal sirve para múltiples peticiones
const [users, posts, comments] = await Promise.all([
  fetch('/api/users', { signal }).then(r => r.json()),
  fetch('/api/posts', { signal }).then(r => r.json()),
  fetch('/api/comments', { signal }).then(r => r.json()),
]);

// Cancela todas las peticiones simultáneamente
controller.abort();
```

**Consejo de entrevista**: `AbortController` es fundamental para evitar memory leaks y condiciones de carrera. Recuerda siempre distinguir el `AbortError` del resto de errores, ya que la cancelación es una operación intencionada. En React, se usa habitualmente en el cleanup de `useEffect`; en Vue, en `onUnmounted`.
