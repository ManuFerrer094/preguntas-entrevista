La **inmutabilidad** significa que los datos no se modifican una vez creados. En lugar de mutar un objeto o array existente, se crean nuevas versiones con los cambios aplicados. Es un principio fundamental en React, Redux y la programación funcional.

## Por qué importa la inmutabilidad

```javascript
// ❌ Mutación: modifica el objeto original
function actualizarNombre(usuario, nuevoNombre) {
  usuario.nombre = nuevoNombre;  // Muta el objeto original
  return usuario;
}

const usuario = { id: 1, nombre: 'Ana', email: 'ana@ejemplo.com' };
const actualizado = actualizarNombre(usuario, 'Carlos');

console.log(usuario === actualizado);    // true — ¡es el MISMO objeto!
console.log(usuario.nombre);             // 'Carlos' — ¡el original fue modificado!

// ✅ Inmutable: crea una nueva versión
function actualizarNombreImmutable(usuario, nuevoNombre) {
  return { ...usuario, nombre: nuevoNombre };  // Nuevo objeto
}

const original = { id: 1, nombre: 'Ana', email: 'ana@ejemplo.com' };
const actualizado2 = actualizarNombreImmutable(original, 'Carlos');

console.log(original === actualizado2);  // false — son objetos diferentes
console.log(original.nombre);            // 'Ana' — original intacto
console.log(actualizado2.nombre);        // 'Carlos' — nueva versión
```

## Patrones inmutables para objetos

```javascript
const usuario = {
  id: 1,
  nombre: 'Ana',
  perfil: {
    ciudad: 'Madrid',
    hobbies: ['música', 'lectura']
  }
};

// Actualizar propiedad de primer nivel
const nuevoBusuario = { ...usuario, nombre: 'Carlos' };

// Actualizar propiedad anidada (spread anidado)
const nuevaUbicacion = {
  ...usuario,
  perfil: {
    ...usuario.perfil,
    ciudad: 'Barcelona'
  }
};

// Eliminar una propiedad con destructuring
const { nombre, ...sinNombre } = usuario;
// sinNombre = { id: 1, perfil: { ... } }

// Añadir propiedad
const conFecha = { ...usuario, fechaActualizacion: new Date() };
```

## Patrones inmutables para arrays

```javascript
const tareas = [
  { id: 1, texto: 'Comprar leche', completada: false },
  { id: 2, texto: 'Llamar al médico', completada: false },
  { id: 3, texto: 'Hacer ejercicio', completada: true }
];

// Añadir elemento (al final / al inicio)
const conNuevaTarea = [...tareas, { id: 4, texto: 'Nueva tarea', completada: false }];
const conTareaAlInicio = [{ id: 4, texto: 'Urgente', completada: false }, ...tareas];

// Insertar en posición específica
function insertarEnPosicion(array, indice, elemento) {
  return [...array.slice(0, indice), elemento, ...array.slice(indice)];
}

// Eliminar por índice
const sinTarea2 = tareas.filter(t => t.id !== 2);

// Actualizar un elemento
const tareaCompletada = tareas.map(t =>
  t.id === 2 ? { ...t, completada: true } : t
);

// Mover un elemento (reordenar)
function moverElemento(array, desdeIndice, hastaIndice) {
  const resultado = [...array];
  const [elemento] = resultado.splice(desdeIndice, 1);
  resultado.splice(hastaIndice, 0, elemento);
  return resultado;
}
```

## Immer — Inmutabilidad con sintaxis mutable

```javascript
import { produce } from 'immer';

const estado = {
  usuarios: [
    { id: 1, nombre: 'Ana', rol: 'admin' },
    { id: 2, nombre: 'Carlos', rol: 'usuario' }
  ],
  configuracion: { tema: 'claro', idioma: 'es' }
};

// produce(estadoActual, recipe) → nuevo estado sin mutar el original
const nuevoEstado = produce(estado, borrador => {
  // Puedes usar mutaciones directas (Immer las convierte en inmutables internamente)
  borrador.usuarios[0].nombre = 'Ana García';
  borrador.usuarios.push({ id: 3, nombre: 'María', rol: 'usuario' });
  borrador.configuracion.tema = 'oscuro';
  // Immer crea un nuevo objeto con estos cambios, sin modificar `estado`
});

console.log(estado.usuarios[0].nombre);      // 'Ana' — original intacto
console.log(nuevoEstado.usuarios[0].nombre); // 'Ana García' — nuevo estado

// Currified version para reducers
const reducer = produce((borrador, accion) => {
  switch (accion.type) {
    case 'TOGGLE_TAREA':
      const tarea = borrador.tareas.find(t => t.id === accion.id);
      if (tarea) tarea.completada = !tarea.completada;
      break;
    case 'ELIMINAR_TAREA':
      borrador.tareas = borrador.tareas.filter(t => t.id !== accion.id);
      break;
  }
});
```

## structuredClone — Copia profunda nativa (ES2022)

```javascript
// Alternativa nativa a JSON.parse(JSON.stringify(...))
// Soporta: Date, Map, Set, ArrayBuffer, undefined, circular references

const original = {
  nombre: 'Ana',
  fecha: new Date(),
  mapa: new Map([['a', 1]]),
  conjunto: new Set([1, 2, 3]),
  datos: { anidado: { profundo: 42 } }
};

// ❌ JSON.parse/stringify: pierde Date, Map, Set, undefined
const clon1 = JSON.parse(JSON.stringify(original));
console.log(clon1.fecha instanceof Date);  // false — se convirtió a string
console.log(clon1.mapa);                   // {} — Map se pierde

// ✅ structuredClone: copia profunda correcta
const clon2 = structuredClone(original);
console.log(clon2.fecha instanceof Date);  // true ✅
console.log(clon2.mapa instanceof Map);    // true ✅

// Modificar el clon no afecta al original
clon2.datos.anidado.profundo = 99;
console.log(original.datos.anidado.profundo);  // 42 ✅
```

**Consejo de entrevista**: La inmutabilidad es fundamental en React porque permite a React detectar cambios por comparación de referencia (`===`) en lugar de comparación profunda, lo que hace los re-renders más eficientes. Menciona Immer como la librería que simplifica la escritura de código inmutable: permite escribir actualizaciones con la sintaxis mutable natural de JavaScript, y Immer se encarga de crear el nuevo objeto inmutable mediante Proxies. Redux Toolkit usa Immer internamente.
