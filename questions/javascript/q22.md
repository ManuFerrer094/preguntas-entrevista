title: ¿Qué es el patrón Observer en JavaScript?

El patrón Observer (también conocido como Publisher-Subscriber o Pub/Sub) define una relación de suscripción donde un sujeto notifica automáticamente a todos sus observadores cuando cambia su estado. Es la base de sistemas de eventos, programación reactiva y librerías como RxJS.

## Implementación básica

```javascript
class EventEmitter {
  #listeners = new Map();

  on(event, callback) {
    if (!this.#listeners.has(event)) {
      this.#listeners.set(event, new Set());
    }
    this.#listeners.get(event).add(callback);
    // Retornar función para desuscribirse (patrón limpio)
    return () => this.off(event, callback);
  }

  off(event, callback) {
    this.#listeners.get(event)?.delete(callback);
  }

  once(event, callback) {
    const wrapper = (...args) => {
      callback(...args);
      this.off(event, wrapper);
    };
    return this.on(event, wrapper);
  }

  emit(event, ...args) {
    this.#listeners.get(event)?.forEach(cb => cb(...args));
  }
}

// Uso
const bus = new EventEmitter();

const unsubscribe = bus.on('userLoggedIn', (user) => {
  console.log(`Bienvenido, ${user.name}`);
});

bus.emit('userLoggedIn', { name: 'Ana', role: 'admin' });
// 'Bienvenido, Ana'

unsubscribe(); // Ya no recibe eventos
bus.emit('userLoggedIn', { name: 'Luis' }); // Sin efecto
```

## Patrón Observable (estilo RxJS simplificado)

```javascript
class Observable {
  #subscribeFn;

  constructor(subscribeFn) {
    this.#subscribeFn = subscribeFn;
  }

  subscribe(observer) {
    // observer puede ser una función o { next, error, complete }
    const normalizedObserver = typeof observer === 'function'
      ? { next: observer, error: console.error, complete: () => {} }
      : { next: () => {}, error: console.error, complete: () => {}, ...observer };

    return this.#subscribeFn(normalizedObserver);
  }

  // Operador map
  map(transformFn) {
    return new Observable(observer => {
      return this.subscribe({
        next: value => observer.next(transformFn(value)),
        error: err => observer.error(err),
        complete: () => observer.complete()
      });
    });
  }

  // Operador filter
  filter(predicateFn) {
    return new Observable(observer => {
      return this.subscribe({
        next: value => predicateFn(value) && observer.next(value),
        error: err => observer.error(err),
        complete: () => observer.complete()
      });
    });
  }
}

// Observable de intervalo
function interval(ms) {
  return new Observable(observer => {
    let count = 0;
    const id = setInterval(() => observer.next(count++), ms);
    return () => clearInterval(id); // Función de cleanup
  });
}

const subscription = interval(1000)
  .filter(n => n % 2 === 0)
  .map(n => `Tick par: ${n}`)
  .subscribe(console.log);

// Después de 5 segundos, desuscribirse
setTimeout(subscription, 5000);
```

## Store reactivo (similar a Redux/MobX simplificado)

```javascript
function createStore(initialState, reducer) {
  let state = initialState;
  const observers = new Set();

  return {
    getState() {
      return state;
    },

    dispatch(action) {
      state = reducer(state, action);
      observers.forEach(observer => observer(state));
    },

    subscribe(observer) {
      observers.add(observer);
      return () => observers.delete(observer); // Unsubscribe
    }
  };
}

// Uso
const store = createStore(
  { count: 0, user: null },
  (state, action) => {
    switch (action.type) {
      case 'INCREMENT': return { ...state, count: state.count + 1 };
      case 'SET_USER': return { ...state, user: action.payload };
      default: return state;
    }
  }
);

const unsubscribe = store.subscribe(state => {
  console.log('Estado actualizado:', state);
});

store.dispatch({ type: 'INCREMENT' });
store.dispatch({ type: 'SET_USER', payload: { name: 'Ana' } });
unsubscribe();
```

## Observer nativo: MutationObserver, IntersectionObserver

```javascript
// MutationObserver: observa cambios en el DOM
const observer = new MutationObserver((mutations) => {
  mutations.forEach(mutation => {
    console.log('Cambio detectado:', mutation.type, mutation.target);
  });
});

observer.observe(document.getElementById('app'), {
  childList: true,
  subtree: true,
  attributes: true
});

// Dejar de observar
observer.disconnect();

// IntersectionObserver: lazy loading de imágenes
const lazyLoader = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      const img = entry.target;
      img.src = img.dataset.src;
      lazyLoader.unobserve(img);
    }
  });
}, { threshold: 0.1 });

document.querySelectorAll('img[data-src]').forEach(img => {
  lazyLoader.observe(img);
});
```

**Consejo de entrevista**: El patrón Observer es fundamental en JavaScript por su naturaleza asíncrona y orientada a eventos. La clave es siempre proveer una forma de desuscribirse para evitar memory leaks. En entrevistas, mencionar la diferencia entre Observer (el sujeto conoce a sus observadores) y Mediador/Pub-Sub (los participantes se comunican a través de un intermediario) muestra un conocimiento profundo.
