title: ¿Qué son los Tagged Template Literals en JavaScript?

Los Tagged Template Literals permiten procesar un template literal con una función. La función "tag" recibe las partes estáticas del string y los valores interpolados por separado, permitiendo transformaciones personalizadas.

## Sintaxis básica

```javascript
function tag(strings, ...values) {
  console.log(strings); // Array de partes estáticas (strings entre ${})
  console.log(values);  // Array de valores interpolados

  // Reconstruir el string (comportamiento por defecto de template literal)
  return strings.reduce((result, str, i) => {
    return result + str + (values[i] !== undefined ? values[i] : '');
  }, '');
}

const name = 'Ana';
const age = 28;
const result = tag`Hola, ${name}. Tienes ${age} años.`;
// strings: ['Hola, ', '. Tienes ', ' años.']
// values:  ['Ana', 28]
```

## Caso de uso: sanitización SQL (prevención de inyección)

```javascript
function sql(strings, ...values) {
  const sanitized = values.map(value => {
    if (typeof value === 'string') {
      // Escapar caracteres peligrosos
      return `'${value.replace(/'/g, "''")}'`;
    }
    return value;
  });

  return strings.reduce((query, str, i) => {
    return query + str + (sanitized[i] !== undefined ? sanitized[i] : '');
  }, '');
}

const userInput = "O'Reilly"; // Entrada del usuario con comilla simple
const id = 42;
const query = sql`SELECT * FROM users WHERE name = ${userInput} AND id = ${id}`;
// "SELECT * FROM users WHERE name = 'O''Reilly' AND id = 42"
```

## Caso de uso: internacionalización (i18n)

```javascript
const translations = {
  es: { greeting: 'Hola, {0}. Tienes {1} mensajes.' },
  en: { greeting: 'Hello, {0}. You have {1} messages.' }
};

function i18n(key) {
  return function(strings, ...values) {
    let template = translations['es'][key] || strings.join('{}');
    return values.reduce((str, val, i) => {
      return str.replace(`{${i}}`, val);
    }, template);
  };
}

const name = 'Ana';
const count = 5;
console.log(i18n('greeting')`${name}${count}`);
// 'Hola, Ana. Tienes 5 mensajes.'
```

## Caso de uso: CSS-in-JS (similar a styled-components)

```javascript
function css(strings, ...values) {
  const style = strings.reduce((acc, str, i) => {
    return acc + str + (values[i] !== undefined ? values[i] : '');
  }, '');

  const className = `css-${Math.random().toString(36).slice(2, 8)}`;
  const styleEl = document.createElement('style');
  styleEl.textContent = `.${className} { ${style} }`;
  document.head.appendChild(styleEl);

  return className;
}

const primaryColor = '#007bff';
const padding = 16;
const buttonClass = css`
  background-color: ${primaryColor};
  padding: ${padding}px ${padding * 2}px;
  border-radius: 4px;
  color: white;
`;

document.querySelector('button').className = buttonClass;
```

## Caso de uso: highlight de código HTML

```javascript
function highlight(strings, ...values) {
  return strings.reduce((result, str, i) => {
    const value = values[i] !== undefined
      ? `<mark>${String(values[i]).replace(/</g, '&lt;')}</mark>`
      : '';
    return result + str + value;
  }, '');
}

const searchTerm = 'JavaScript';
const text = 'Aprende JavaScript y domina la web con JavaScript';
// No aplica directamente sobre el texto, pero ilustra el concepto:
const html = highlight`Resultado: ${searchTerm} encontrado`;
// 'Resultado: <mark>JavaScript</mark> encontrado'
```

## String.raw — el tag incorporado más común

```javascript
// String.raw evita el procesamiento de secuencias de escape
const path = String.raw`C:\Users\Ana\Documents`;
console.log(path); // 'C:\Users\Ana\Documents' (no interpreta \U, \A, \D)

// Sin String.raw:
const path2 = `C:\Users\Ana\Documents`;
console.log(path2); // 'C:UsersAnaDocuments' (procesa \U, \A, \D como escapes)

// Útil para expresiones regulares complejas
const regex = new RegExp(String.raw`\d{4}-\d{2}-\d{2}`);
console.log(regex.test('2024-01-15')); // true
```

**Consejo de entrevista**: Los tagged template literals son la base de librerías populares como `styled-components`, `graphql-tag`, y `sql-template-strings`. Entender cómo funcionan internamente demuestra un conocimiento avanzado de JavaScript. La clave es que la función tag recibe un array de strings estáticos (siempre `strings.length === values.length + 1`) y los valores interpolados por separado.
