---
title: ¿Qué es el manejo de errores con try/catch en JavaScript?
difficulty: easy
tags: [javascript, error-handling, try-catch, exceptions]
---

El bloque `try/catch/finally` es el mecanismo principal de JavaScript para manejar errores de forma controlada. Permite capturar excepciones que ocurren durante la ejecución de código y ejecutar código de recuperación o limpieza, evitando que la aplicación se detenga abruptamente.

## Estructura básica

```javascript
try {
  // Código que puede lanzar un error
  const result = JSON.parse(invalidJson); // Lanza SyntaxError
  processData(result);
} catch (error) {
  // Se ejecuta si ocurre un error en el bloque try
  console.error('Error:', error.message);
  console.error('Tipo:', error.name);    // 'SyntaxError'
  console.error('Stack:', error.stack);
} finally {
  // Siempre se ejecuta, haya error o no
  // Ideal para limpieza de recursos
  closeConnection();
  hideLoadingSpinner();
}
```

## Tipos de errores nativos

```javascript
// RangeError: valor fuera del rango permitido
const arr = new Array(-1); // ❌ RangeError

// ReferenceError: acceder a variable no declarada
console.log(undeclaredVar); // ❌ ReferenceError

// SyntaxError: código con sintaxis inválida (en runtime con eval)
eval('const {'); // ❌ SyntaxError

// TypeError: operación en tipo incorrecto
null.property; // ❌ TypeError
undefined();   // ❌ TypeError

// URIError
decodeURIComponent('%'); // ❌ URIError
```

## Errores personalizados

```javascript
class AppError extends Error {
  constructor(message, statusCode = 500, code = 'INTERNAL_ERROR') {
    super(message);
    this.name = this.constructor.name;
    this.statusCode = statusCode;
    this.code = code;
    // Mantener el stack trace correcto en V8
    if (Error.captureStackTrace) {
      Error.captureStackTrace(this, this.constructor);
    }
  }
}

class ValidationError extends AppError {
  constructor(message, fields) {
    super(message, 400, 'VALIDATION_ERROR');
    this.fields = fields;
  }
}

class NotFoundError extends AppError {
  constructor(resource) {
    super(`${resource} no encontrado`, 404, 'NOT_FOUND');
  }
}

// Usar instanceof para identificar el tipo de error
try {
  const user = await getUser(id);
  if (!user) throw new NotFoundError('Usuario');
  validateUser(user); // Puede lanzar ValidationError
} catch (error) {
  if (error instanceof NotFoundError) {
    return res.status(404).json({ error: error.message });
  }
  if (error instanceof ValidationError) {
    return res.status(400).json({ error: error.message, fields: error.fields });
  }
  throw error; // Re-lanzar errores no manejados
}
```

## try/catch con async/await

```javascript
// Patrón 1: try/catch tradicional
async function fetchUserData(id) {
  try {
    const response = await fetch(`/api/users/${id}`);
    if (!response.ok) throw new Error(`HTTP ${response.status}`);
    return await response.json();
  } catch (error) {
    console.error('Error fetching user:', error);
    return null; // Valor por defecto
  }
}

// Patrón 2: función helper que evita try/catch repetitivo
async function tryCatch(promise) {
  try {
    const data = await promise;
    return [null, data];
  } catch (error) {
    return [error, null];
  }
}

// Uso elegante (inspirado en Go)
const [error, user] = await tryCatch(fetchUser(id));
if (error) {
  console.error('No se pudo obtener el usuario:', error.message);
  return;
}
console.log(user.name);
```

## finally para limpieza garantizada

```javascript
async function uploadFile(file) {
  let tempPath;

  try {
    tempPath = await saveTempFile(file);
    const result = await processAndUpload(tempPath);
    await db.saveRecord(result);
    return result;
  } catch (error) {
    console.error('Error en upload:', error);
    throw error; // Re-lanzar para que el caller sepa del error
  } finally {
    // Siempre limpiar el archivo temporal
    if (tempPath) {
      await fs.unlink(tempPath).catch(() => {}); // Ignorar error de limpieza
    }
  }
}
```

## Errores en Promises sin async/await

```javascript
// Con .catch()
fetch('/api/data')
  .then(res => res.json())
  .then(data => processData(data))
  .catch(error => {
    console.error('Error:', error.message);
  })
  .finally(() => {
    hideLoader();
  });

// Capturar promesas no manejadas globalmente
window.addEventListener('unhandledrejection', event => {
  console.error('Promesa rechazada sin manejar:', event.reason);
  event.preventDefault(); // Evitar que el navegador muestre el error
});

// En Node.js
process.on('unhandledRejection', (reason) => {
  logger.error('Unhandled Rejection:', reason);
  process.exit(1);
});
```

**Consejo de entrevista**: Puntos clave: (1) el bloque `finally` siempre se ejecuta, incluso si hay un `return` en el `try` o `catch`; (2) un `catch` vacío (`catch {}`) oculta errores silenciosamente — siempre hacer algo con el error; (3) re-lanzar con `throw error` (no `throw new Error(error)`) para preservar el stack trace original; (4) con `async/await`, el try/catch captura tanto errores síncronos como rechazos de promesas.
