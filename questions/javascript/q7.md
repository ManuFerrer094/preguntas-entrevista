Los iteradores y generadores son mecanismos de JavaScript para controlar la iteración de forma personalizada. Los iteradores implementan el protocolo de iteración, y los generadores son una forma especial de funciones que producen iteradores de manera más conveniente.

## El protocolo de iteración

Un objeto es **iterable** si implementa el método `Symbol.iterator` que devuelve un **iterador**: un objeto con el método `next()` que devuelve `{ value, done }`.

```javascript
// Crear un iterador manualmente
function createRangeIterator(start, end) {
  let current = start;

  return {
    [Symbol.iterator]() { return this; }, // Iterable además de iterador
    next() {
      if (current <= end) {
        return { value: current++, done: false };
      }
      return { value: undefined, done: true };
    }
  };
}

const range = createRangeIterator(1, 5);

for (const num of range) { // for...of usa Symbol.iterator
  console.log(num); // 1, 2, 3, 4, 5
}

// Spread también usa el protocolo de iteración
const numbers = [...createRangeIterator(1, 5)]; // [1, 2, 3, 4, 5]
```

## Generadores — la forma elegante

Los generadores son funciones especiales (definidas con `function*`) que pueden pausar su ejecución con `yield` y reanudarla luego. Cada llamada a `next()` ejecuta hasta el siguiente `yield`.

```javascript
// Generador equivalente al iterador anterior
function* range(start, end) {
  for (let i = start; i <= end; i++) {
    yield i; // Pausa y devuelve el valor
  }
}

for (const num of range(1, 5)) {
  console.log(num); // 1, 2, 3, 4, 5
}

// Generador infinito — sin while(true) bloqueante
function* naturalNumbers() {
  let n = 1;
  while (true) {
    yield n++;
  }
}

// Tomar solo los primeros N valores
function take(iterable, n) {
  const result = [];
  for (const value of iterable) {
    result.push(value);
    if (result.length === n) break;
  }
  return result;
}

console.log(take(naturalNumbers(), 5)); // [1, 2, 3, 4, 5]
```

## Generadores con datos bidireccionales

Los generadores también pueden **recibir** valores a través de `next(value)`:

```javascript
function* calculator() {
  let result = 0;
  while (true) {
    const input = yield result; // yield devuelve result y recibe el siguiente valor
    if (input === null) break;
    result += input;
  }
  return result;
}

const calc = calculator();
calc.next();      // Iniciar el generador, devuelve { value: 0, done: false }
calc.next(10);    // Suma 10 → { value: 10, done: false }
calc.next(5);     // Suma 5  → { value: 15, done: false }
calc.next(null);  // Terminar → { value: 15, done: true }
```

## Generadores asíncronos

```javascript
async function* paginatedFetch(url) {
  let page = 1;
  let hasMore = true;

  while (hasMore) {
    const response = await fetch(`${url}?page=${page}`);
    const { data, total, perPage } = await response.json();

    yield data; // Emitir cada página de resultados

    page++;
    hasMore = page * perPage < total;
  }
}

// Procesar páginas de una API sin cargar todo en memoria
for await (const page of paginatedFetch('/api/users')) {
  processPage(page);
}
```

## Uso en el mundo real: lazy evaluation

```javascript
// Procesar archivos grandes línea por línea (sin cargar todo en memoria)
const { createReadStream } = require('fs');
const { createInterface } = require('readline');

async function* readLines(filePath) {
  const fileStream = createReadStream(filePath);
  const rl = createInterface({ input: fileStream, crlfDelay: Infinity });

  for await (const line of rl) {
    yield line;
  }
}

async function processCSV(filePath) {
  for await (const line of readLines(filePath)) {
    const [name, email, age] = line.split(',');
    await processRecord({ name, email, age });
  }
}
```

**Consejo de entrevista**: Los generadores son poderosos para representar secuencias infinitas, implementar lazy evaluation (procesar datos bajo demanda sin cargar todo en memoria) y crear máquinas de estado. La librería Redux-Saga los usa extensamente para gestionar side effects. El punto clave es que `yield` suspende la ejecución sin bloquear el hilo, lo que los hace fundamentalmente diferentes a los bucles normales.
